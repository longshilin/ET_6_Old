50 private WireType GetWireType(ProtoTypeCode code, DataFormat format, ref Type type, out int modelKey)
51 {
52 modelKey = -1;
53 if (Helpers.IsEnum(type))
54 {
55 modelKey = GetKey(ref type);
56 return WireType.Variant;
57 }
58 switch (code)
59 {
60 case ProtoTypeCode.Int64:
61 case ProtoTypeCode.UInt64:
62 return format == DataFormat.FixedSize ? WireType.Fixed64 : WireType.Variant;
63 case ProtoTypeCode.Int16:
64 case ProtoTypeCode.Int32:
65 case ProtoTypeCode.UInt16:
66 case ProtoTypeCode.UInt32:
67 case ProtoTypeCode.Boolean:
68 case ProtoTypeCode.SByte:
69 case ProtoTypeCode.Byte:
70 case ProtoTypeCode.Char:
71 return format == DataFormat.FixedSize ? WireType.Fixed32 : WireType.Variant;
72 case ProtoTypeCode.Double:
73 return WireType.Fixed64;
74 case ProtoTypeCode.Single:
75 return WireType.Fixed32;
76 case ProtoTypeCode.String:
77 case ProtoTypeCode.DateTime:
78 case ProtoTypeCode.Decimal:
79 case ProtoTypeCode.ByteArray:
80 case ProtoTypeCode.TimeSpan:
81 case ProtoTypeCode.Guid:
82 case ProtoTypeCode.Uri:
83 return WireType.String;
84 }
85
86 if ((modelKey = GetKey(ref type)) >= 0)
87 {
88 return WireType.String;
89 }
90 return WireType.None;
91 }

103 internal bool TrySerializeAuxiliaryType(ProtoWriter writer, Type type, DataFormat format, int tag, object value, bool isInsideList, object parentList)
104 {
105 if (type == null) { type = value.GetType(); }
106
107 ProtoTypeCode typecode = Helpers.GetTypeCode(type);
108 // note the "ref type" here normalizes against proxies
109 int modelKey;
110 WireType wireType = GetWireType(typecode, format, ref type, out modelKey);
111
112
113 if (modelKey >= 0)
114 { // write the header, but defer to the model
115 if (Helpers.IsEnum(type))
116 { // no header
117 Serialize(modelKey, value, writer);
118 return true;
119 }
120 else
121 {
122 ProtoWriter.WriteFieldHeader(tag, wireType, writer);
123 switch (wireType)
124 {
125 case WireType.None:
126 throw ProtoWriter.CreateException(writer);
127 case WireType.StartGroup:
128 case WireType.String:
129 // needs a wrapping length etc
130 SubItemToken token = ProtoWriter.StartSubItem(value, writer);
131 Serialize(modelKey, value, writer);
132 ProtoWriter.EndSubItem(token, writer);
133 return true;
134 default:
135 Serialize(modelKey, value, writer);
136 return true;
137 }
138 }
139 }
140
141 if(wireType != WireType.None) {
142 ProtoWriter.WriteFieldHeader(tag, wireType, writer);
143 }
144 switch(typecode) {
145 case ProtoTypeCode.Int16: ProtoWriter.WriteInt16((short)value, writer); return true;
146 case ProtoTypeCode.Int32: ProtoWriter.WriteInt32((int)value, writer); return true;
147 case ProtoTypeCode.Int64: ProtoWriter.WriteInt64((long)value, writer); return true;
148 case ProtoTypeCode.UInt16: ProtoWriter.WriteUInt16((ushort)value, writer); return true;
149 case ProtoTypeCode.UInt32: ProtoWriter.WriteUInt32((uint)value, writer); return true;
150 case ProtoTypeCode.UInt64: ProtoWriter.WriteUInt64((ulong)value, writer); return true;
151 case ProtoTypeCode.Boolean: ProtoWriter.WriteBoolean((bool)value, writer); return true;
152 case ProtoTypeCode.SByte: ProtoWriter.WriteSByte((sbyte)value, writer); return true;
153 case ProtoTypeCode.Byte: ProtoWriter.WriteByte((byte)value, writer); return true;
154 case ProtoTypeCode.Char: ProtoWriter.WriteUInt16((ushort)(char)value, writer); return true;
155 case ProtoTypeCode.Double: ProtoWriter.WriteDouble((double)value, writer); return true;
156 case ProtoTypeCode.Single: ProtoWriter.WriteSingle((float)value, writer); return true;
157 case ProtoTypeCode.DateTime:
158 if (SerializeDateTimeKind())
159 BclHelpers.WriteDateTimeWithKind((DateTime)value, writer);
160 else
161 BclHelpers.WriteDateTime((DateTime)value, writer);
162 return true;
163 case ProtoTypeCode.Decimal: BclHelpers.WriteDecimal((decimal)value, writer); return true;
164 case ProtoTypeCode.String: ProtoWriter.WriteString((string)value, writer); return true;
165 case ProtoTypeCode.ByteArray: ProtoWriter.WriteBytes((byte[])value, writer); return true;
166 case ProtoTypeCode.TimeSpan: BclHelpers.WriteTimeSpan((TimeSpan)value, writer); return true;
167 case ProtoTypeCode.Guid: BclHelpers.WriteGuid((Guid)value, writer); return true;
168 case ProtoTypeCode.Uri: ProtoWriter.WriteString(((Uri)value).OriginalString, writer); return true;
169 }
170
171 // by now, we should have covered all the simple cases; if we wrote a field-header, we have
172 // forgotten something!
173 Helpers.DebugAssert(wireType == WireType.None);
174
175 // now attempt to handle sequences (including arrays and lists)
176 if (value is IEnumerable)
177 {
178 var sequence = value as IEnumerable;
179 if (isInsideList) throw CreateNestedListsNotSupported(parentList?.GetType());
180 foreach (object item in sequence)
181 {
182 if (item == null) { throw new NullReferenceException(); }
183 if (!TrySerializeAuxiliaryType(writer, null, format, tag, item, true, sequence))
184 {
185 ThrowUnexpectedType(item.GetType());
186 }
187 }
188 return true;
189 }
190 return false;
191 }

337 private object DeserializeWithLengthPrefix(Stream source, object value, Type type, PrefixStyle style, int expectedField, Serializer.TypeResolver resolver, out long bytesRead, out bool haveObject, SerializationContext context)
338 {
339#if FEAT_IKVM
340 throw new NotSupportedException();
341#else
342 haveObject = false;
343 bool skip;
344 long len;
345 bytesRead = 0;
346 if (type == null && (style != PrefixStyle.Base128 || resolver == null))
347 {
348 throw new InvalidOperationException("A type must be provided unless base-128 prefixing is being used in combination with a resolver");
349 }
350 do
351 {
352
353 bool expectPrefix = expectedField > 0 || resolver != null;
354 int actualField;
355 int tmpBytesRead;
356 len = ProtoReader.ReadLongLengthPrefix(source, expectPrefix, style, out actualField, out tmpBytesRead);
357 if (tmpBytesRead == 0) return value;
358 bytesRead += tmpBytesRead;
359 if (len < 0) return value;
360
361 switch (style)
362 {
363 case PrefixStyle.Base128:
364 if (expectPrefix && expectedField == 0 && type == null && resolver != null)
365 {
366 type = resolver(actualField);
367 skip = type == null;
368 }
369 else { skip = expectedField != actualField; }
370 break;
371 default:
372 skip = false;
373 break;
374 }
375
376 if (skip)
377 {
378 if (len == long.MaxValue) throw new InvalidOperationException();
379 ProtoReader.Seek(source, len, null);
380 bytesRead += len;
381 }
382 } while (skip);
383
384 ProtoReader reader = null;
385 try
386 {
387 reader = ProtoReader.Create(source, this, context, len);
388 int key = GetKey(ref type);
389 if (key >= 0 && !Helpers.IsEnum(type))
390 {
391 value = Deserialize(key, value, reader);
392 }
393 else
394 {
395 if (!(TryDeserializeAuxiliaryType(reader, DataFormat.Default, Serializer.ListItemTag, type, ref value, true, false, true, false, null) || len == 0))
396 {
397 TypeModel.ThrowUnexpectedType(type); // throws
398 }
399 }
400 bytesRead += reader.LongPosition;
401 haveObject = true;
402 return value;
403 }
404 finally
405 {
406 ProtoReader.Recycle(reader);
407 }
408#endif

842 internal static Type GetListItemType(TypeModel model, Type listType)
843 {
844 Helpers.DebugAssert(listType != null);
845
846#if WINRT
847 TypeInfo listTypeInfo = listType.GetTypeInfo();
848 if (listType == typeof(string) || listType.IsArray
849 || !typeof(IEnumerable).GetTypeInfo().IsAssignableFrom(listTypeInfo)) return null;
850#else
851 if (listType == model.MapType(typeof(string)) || listType.IsArray
852 || !model.MapType(typeof(IEnumerable)).IsAssignableFrom(listType)) return null;
853 if (listType.FullName == "System.String") return null;
854#endif
855
856
857 if (listType is ILRuntimeWrapperType)
858 {
859 return ((ILRuntimeWrapperType)listType).CLRType.GenericArguments[0].Value.ReflectionType;
860 }
861
862 BasicList candidates = new BasicList();
863#if WINRT
864 foreach (MethodInfo method in listType.GetRuntimeMethods())
865#else
866 foreach (MethodInfo method in listType.GetMethods())
867#endif
868 {
869 if (method.IsStatic || method.Name != "Add") continue;
870 ParameterInfo[] parameters = method.GetParameters();
871 Type paramType;
872 if (parameters.Length == 1 && !candidates.Contains(paramType = parameters[0].ParameterType))
873 {
874 candidates.Add(paramType);
875 }
876 }
877
878 string name = listType.Name;
879 bool isQueueStack = name != null && (name.IndexOf("Queue") >= 0 || name.IndexOf("Stack") >= 0);
880#if !NO_GENERICS
881 if(!isQueueStack)
882 {
883 TestEnumerableListPatterns(model, candidates, listType);
884#if WINRT
885 foreach (Type iType in listTypeInfo.ImplementedInterfaces)
886 {
887 TestEnumerableListPatterns(model, candidates, iType);
888 }
889#else
890 foreach (Type iType in listType.GetInterfaces())
891 {
892 TestEnumerableListPatterns(model, candidates, iType);
893 }
894#endif
895 }
896#endif
897
898#if WINRT
899 // more convenient GetProperty overload not supported on all platforms
900 foreach (PropertyInfo indexer in listType.GetRuntimeProperties())
901 {
902 if (indexer.Name != "Item" || candidates.Contains(indexer.PropertyType)) continue;
903 ParameterInfo[] args = indexer.GetIndexParameters();
904 if (args.Length != 1 || args[0].ParameterType != typeof(int)) continue;
905 MethodInfo getter = indexer.GetMethod;
906 if (getter == null || getter.IsStatic) continue;
907 candidates.Add(indexer.PropertyType);
908 }
909#else
910 // more convenient GetProperty overload not supported on all platforms
911 foreach (PropertyInfo indexer in listType.GetProperties(BindingFlags.Instance | BindingFlags.Public | BindingFlags.NonPublic))
912 {
913 if (indexer.Name != "Item" || candidates.Contains(indexer.PropertyType)) continue;
914 ParameterInfo[] args = indexer.GetIndexParameters();
915 if (args.Length != 1 || args[0].ParameterType != model.MapType(typeof(int))) continue;
916 candidates.Add(indexer.PropertyType);
917 }
918#endif
919
920 switch (candidates.Count)
921 {
922 case 0:
923 return null;
924 case 1:
925 if ((Type)candidates[0] == listType) return null; // recursive
926 return (Type)candidates[0];
927 case 2:
928 if ((Type)candidates[0] != listType && CheckDictionaryAccessors(model, (Type)candidates[0], (Type)candidates[1])) return (Type)candidates[0];
929 if ((Type)candidates[1] != listType && CheckDictionaryAccessors(model, (Type)candidates[1], (Type)candidates[0])) return (Type)candidates[1];
930 break;
931 }
932
933 return null;
934 }

937 {
938
939#if WINRT || COREFX
940 TypeInfo iTypeInfo = iType.GetTypeInfo();
941 if (iTypeInfo.IsGenericType)
942 {
943 Type typeDef = iTypeInfo.GetGenericTypeDefinition();
944 if(
945 typeDef == model.MapType(typeof(System.Collections.Generic.IEnumerable<>))
946 || typeDef == model.MapType(typeof(System.Collections.Generic.ICollection<>))
947 || typeDef.GetTypeInfo().FullName == "System.Collections.Concurrent.IProducerConsumerCollection`1")
948 {
949
950 Type[] iTypeArgs = iTypeInfo.GenericTypeArguments;
951 if (!candidates.Contains(iTypeArgs[0]))
952 {
953 candidates.Add(iTypeArgs[0]);
954 }
955 }
956 }
957#elif !NO_GENERICS
958 if (iType.IsGenericType)
959 {
960 Type typeDef = iType.GetGenericTypeDefinition();
961 if (typeDef == model.MapType(typeof(System.Collections.Generic.IEnumerable<>))
962 || typeDef == model.MapType(typeof(System.Collections.Generic.ICollection<>))
