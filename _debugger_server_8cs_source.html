117 void OnReceive(DebugMessageType type, byte[] buffer)
118 {
119 if (clientSocket == null || clientSocket.Disconnected)
120 return;
121 System.IO.MemoryStream ms = new System.IO.MemoryStream(buffer);
122 System.IO.BinaryReader br = new System.IO.BinaryReader(ms);
123
124 switch (type)
125 {
126 case DebugMessageType.CSAttach:
127 {
128 SendAttachResult();
129 }
130 break;
131 case DebugMessageType.CSBindBreakpoint:
132 {
133 CSBindBreakpoint msg = new Protocol.CSBindBreakpoint();
134 msg.BreakpointHashCode = br.ReadInt32();
135 msg.IsLambda = br.ReadBoolean();
136 msg.TypeName = br.ReadString();
137 msg.MethodName = br.ReadString();
138 msg.StartLine = br.ReadInt32();
139 msg.EndLine = br.ReadInt32();
140 TryBindBreakpoint(msg);
141 }
142 break;
143 case DebugMessageType.CSDeleteBreakpoint:
144 {
145 CSDeleteBreakpoint msg = new Protocol.CSDeleteBreakpoint();
146 msg.BreakpointHashCode = br.ReadInt32();
147 ds.DeleteBreakpoint(msg.BreakpointHashCode);
148 }
149 break;
150 case DebugMessageType.CSExecute:
151 {
152 CSExecute msg = new Protocol.CSExecute();
153 msg.ThreadHashCode = br.ReadInt32();
154 ds.ExecuteThread(msg.ThreadHashCode);
155 }
156 break;
157 case DebugMessageType.CSStep:
158 {
159 CSStep msg = new CSStep();
160 msg.ThreadHashCode = br.ReadInt32();
161 msg.StepType = (StepTypes)br.ReadByte();
162 ds.StepThread(msg.ThreadHashCode, msg.StepType);
163 }
164 break;
165 case DebugMessageType.CSResolveVariable:
166 {
167 CSResolveVariable msg = new CSResolveVariable();
168 msg.ThreadHashCode = br.ReadInt32();
169 msg.Variable = ReadVariableReference(br);
170 VariableInfo info;
171 try
172 {
173 object res;
174 info = ds.ResolveVariable(msg.ThreadHashCode, msg.Variable, out res);
175 }
176 catch (Exception ex)
177 {
178 info = VariableInfo.GetException(ex);
179 }
180 if (info.Type != VariableTypes.Pending)
181 SendSCResolveVariableResult(info);
182 }
183 break;
184 case DebugMessageType.CSResolveIndexAccess:
185 {
186 CSResolveIndexer msg = new CSResolveIndexer();
187 msg.ThreadHashCode = br.ReadInt32();
188 msg.Body = ReadVariableReference(br);
189 msg.Index = ReadVariableReference(br);
190
191 VariableInfo info;
192 try
193 {
194 object res;
195 info = ds.ResolveIndexAccess(msg.ThreadHashCode, msg.Body, msg.Index, out res);
196 }
197 catch(Exception ex)
198 {
199 info = VariableInfo.GetException(ex);
200 }
201 if (info.Type != VariableTypes.Pending)
202 SendSCResolveVariableResult(info);
203 }
204 break;
205 case DebugMessageType.CSEnumChildren:
206 {
207 int thId = br.ReadInt32();
208 var parent = ReadVariableReference(br);
209
210 VariableInfo[] info = null;
211 try
212 {
213 info = ds.EnumChildren(thId, parent);
214 }
215 catch(Exception ex)
216 {
217 info = new VariableInfo[] { VariableInfo.GetException(ex) };
218 }
219 if (info != null)
220 SendSCEnumChildrenResult(info);
221 }
222 break;
223 }
224
225 }

227 VariableReference ReadVariableReference(System.IO.BinaryReader br)
228 {
229 VariableReference res = null;
230 if (br.ReadBoolean())
231 {
232 res = new Debugger.VariableReference();
233 res.Address = br.ReadInt64();
234 res.Type = (VariableTypes)br.ReadByte();
235 res.Offset = br.ReadInt32();
236 res.Name = br.ReadString();
237 res.Parent = ReadVariableReference(br);

241 {
242 res.Parameters[i] = ReadVariableReference(br);
243 }
244 }
245 return res;

270 {

272 Mono.Cecil.CustomAttribute ca = null;
273 found = null;
274 foreach (var attr in mDef.CustomAttributes)
275 {
276 switch (attr.AttributeType.FullName)
277 {
278 case "System.Runtime.CompilerServices.AsyncStateMachineAttribute":
279 case "System.Runtime.CompilerServices.IteratorStateMachineAttribute":

303 void TryBindBreakpoint(CSBindBreakpoint msg)
304 {
305 var domain = ds.AppDomain;
306 SCBindBreakpointResult res = new Protocol.SCBindBreakpointResult();
307 res.BreakpointHashCode = msg.BreakpointHashCode;
308 IType type;
309 if (msg.IsLambda)
310 {
311 ILMethod found = null;
312 foreach (var i in domain.LoadedTypes.ToArray())
313 {
314 var vt = i.Value as ILType;
315 if (vt != null)
316 {
317 if (vt.FullName.Contains(msg.TypeName))
318 {
319 foreach (var j in vt.GetMethods())
320 {
321 if (j.Name.Contains(string.Format("<{0}>", msg.MethodName)))
322 {
323 ILMethod ilm = (ILMethod)j;
324 if (ilm.StartLine <= (msg.StartLine + 1) && ilm.EndLine >= (msg.StartLine + 1))
325 {
326 found = ilm;
327 break;
328 }
329 else if (CheckCompilerGeneratedStateMachine(ilm, domain, msg.StartLine, out found))
330 {
331 break;
332 }
333 }
334 }
335 }
336 }
337 if (found != null)
338 break;
339 }
340 if (found != null)
341 {
342 ds.SetBreakPoint(found.GetHashCode(), msg.BreakpointHashCode, msg.StartLine);
343 res.Result = BindBreakpointResults.OK;
344 }
345 else
346 {
347 res.Result = BindBreakpointResults.CodeNotFound;
348 }
349 }
350 else
351 {
352 if (domain.LoadedTypes.TryGetValue(msg.TypeName, out type))
353 {
354 if (type is ILType)
355 {
356 ILType it = (ILType)type;
357 ILMethod found = null;
358 if (msg.MethodName == ".ctor")
359 {
360 foreach (var i in it.GetConstructors())
361 {
362 ILMethod ilm = (ILMethod)i;
363 if (ilm.StartLine <= (msg.StartLine + 1) && ilm.EndLine >= (msg.StartLine + 1))
364 {
365 found = ilm;
366 break;
367 }
368 }
369 }
370 else if (msg.MethodName == ".cctor")
371 {
372 ILMethod ilm = it.GetStaticConstroctor() as ILMethod;
373 if (ilm.StartLine <= (msg.StartLine + 1) && ilm.EndLine >= (msg.StartLine + 1))
374 {
375 found = ilm;
376 }
377 }
378 else
379 {
380 foreach (var i in it.GetMethods())
381 {
382 if (i.Name == msg.MethodName)
383 {
384 ILMethod ilm = (ILMethod)i;
385 if (ilm.StartLine <= (msg.StartLine + 1) && ilm.EndLine >= (msg.StartLine + 1))
386 {
387 found = ilm;
388 break;
389 }
390 else if(CheckCompilerGeneratedStateMachine(ilm, domain, msg.StartLine, out found))
391 {
392 break;
393 }
394 }
395 }
396 }
397 if (found != null)
398 {
399 ds.SetBreakPoint(found.GetHashCode(), msg.BreakpointHashCode, msg.StartLine);
400 res.Result = BindBreakpointResults.OK;
401 }
402 else
403 {
404 res.Result = BindBreakpointResults.CodeNotFound;
405 }
406 }
407 else
408 {
409 res.Result = BindBreakpointResults.TypeNotFound;
410 }
411 }
412 else
413 {
414 res.Result = BindBreakpointResults.TypeNotFound;
415 }
416 }
417 SendSCBindBreakpointResult(res);
418 }
