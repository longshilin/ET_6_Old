575 void InitializeBaseType ()
576 {
577 if ( definition != null && definition.BaseType != null )
578 {
579 bool specialProcess = false;
580 List<int> spIdx = null;
581 if ( definition.BaseType.IsGenericInstance )
582 {
583 GenericInstanceType git = definition.BaseType as GenericInstanceType;
584 var elementType = appdomain.GetType ( git.ElementType, this, null );
585 if ( elementType is CLRType )
586 {
587 for ( int i = 0; i < git.GenericArguments.Count; i++ )
588 {
589 var ga = git.GenericArguments [ i ];
590 if ( ga == typeRef )
591 {
592 specialProcess = true;
593 if ( spIdx == null )
594 spIdx = new List<int> ();
595 spIdx.Add ( i );
596 }
597 }
598 }
599 }
600 if ( specialProcess )
601 {
602 //如果泛型参数是自身，则必须要特殊处理，否则会StackOverflow
603 var elementType = appdomain.GetType ( ( ( GenericInstanceType ) definition.BaseType ).ElementType, this, null );
604 foreach ( var i in appdomain.CrossBindingAdaptors )
605 {
606 if ( i.Key.IsGenericType && !i.Key.IsGenericTypeDefinition )
607 {
608 var gd = i.Key.GetGenericTypeDefinition ();
609 if ( gd == elementType.TypeForCLR )
610 {
611 var ga = i.Key.GetGenericArguments ();
612 bool match = true;
613 foreach ( var j in spIdx )
614 {
615 if ( ga [ j ] != i.Value.AdaptorType )
616 {
617 match = false;
618 break;
619 }
620 }
621 if ( match )
622 {
623 baseType = i.Value;
624 break;
625 }
626 }
627 }
628 }
629 if ( baseType == null )
630 {
631 throw new TypeLoadException ( "Cannot find Adaptor for:" + definition.BaseType.FullName );
632 }
633 }
634 else
635 {
636 baseType = appdomain.GetType ( definition.BaseType, this, null );
637 if ( baseType is CLRType )
638 {
639 if ( baseType.TypeForCLR == typeof ( Enum ) || baseType.TypeForCLR == typeof ( object ) || baseType.TypeForCLR == typeof ( ValueType ) || baseType.TypeForCLR == typeof ( System.Enum ) )
640 {//都是这样，无所谓
641 baseType = null;
642 }
643 else if ( baseType.TypeForCLR == typeof ( MulticastDelegate ) )
644 {
645 baseType = null;
646 isDelegate = true;
647 }
648 else
649 {
650 CrossBindingAdaptor adaptor;
651 if ( appdomain.CrossBindingAdaptors.TryGetValue ( baseType.TypeForCLR, out adaptor ) )
652 {
653 baseType = adaptor;
654 }
655 else
656 throw new TypeLoadException ( "Cannot find Adaptor for:" + baseType.TypeForCLR.ToString () );
657 //继承了其他系统类型
658 //env.logger.Log_Error("ScriptType:" + Name + " Based On a SystemType:" + BaseType.Name);
659 //HasSysBase = true;
660 //throw new Exception("不得继承系统类型，脚本类型系统和脚本类型系统是隔离的");
661 }
662 }
663 }
664 }
665 var curBase = baseType;
666 while ( curBase is ILType )
667 {
668 curBase = curBase.BaseType;
669 }
670 firstCLRBaseType = curBase;
671 baseTypeInitialized = true;

712 {
713 methods = new Dictionary<string, List<ILMethod>> ();
714 constructors = new List<ILMethod> ();
715 if ( definition == null )
716 return;
717 if ( definition.HasCustomAttributes )
718 {
719 for ( int i = 0; i < definition.CustomAttributes.Count; i++ )
720 {
721 int f;
722 if ( definition.CustomAttributes [ i ].GetJITFlags ( AppDomain, out f ) )
723 {
724 this.jitFlags = f;
725 break;
726 }
727 }
728 }
729 foreach ( var i in definition.Methods )
730 {
731 if ( i.IsConstructor )
732 {
733 if ( i.IsStatic )
734 staticConstructor = new ILMethod ( i, this, appdomain, jitFlags );

747 lst.Add ( m );
748 }
749 }

801 public IMethod GetMethod ( string name, List<IType> param, IType [] genericArguments, IType returnType = null, bool declaredOnly = false )
802 {
803 if ( methods == null )
804 InitializeMethods ();
805 List<ILMethod> lst;
806 IMethod genericMethod = null;
807 if ( methods.TryGetValue ( name, out lst ) )
808 {
809 for ( var idx = 0; idx < lst.Count; idx++ )
810 {
811 var i = lst [ idx ];
812 int pCnt = param != null ? param.Count : 0;
813 if ( i.ParameterCount == pCnt )
814 {
815 bool match = true;
816 if ( genericArguments != null && i.GenericParameterCount == genericArguments.Length && genericMethod == null )
817 {
818 genericMethod = CheckGenericParams ( i, param, genericArguments, ref match );
819 }
820 else
821 {
822 match = CheckGenericArguments ( i, genericArguments );
823 if ( !match )
824 continue;
825 for ( int j = 0; j < pCnt; j++ )
826 {
827 if ( param [ j ] != i.Parameters [ j ] )
828 {
829 match = false;
830 break;
831 }
832 }
833 if ( match )
834 {
835 match = returnType == null || i.ReturnType == returnType;
836 }
837 if ( match )
838 return i;
839 }
840 }
841 }
842 }
843 if ( genericArguments != null && genericMethod != null )
844 {
845 var m = genericMethod.MakeGenericMethod ( genericArguments );
846 lst.Add ( ( ILMethod ) m );
847 return m;
848 }
849 if ( declaredOnly )
850 return null;
851 else
852 {
853 if ( BaseType != null )
854 return BaseType.GetMethod ( name, param, genericArguments, returnType, false );
855 else
856 return null;
857 }

905 ILMethod CheckGenericParams ( ILMethod i, List<IType> param, IType [] genericArguments, ref bool match )
906 {
907 ILMethod genericMethod = null;
908 if ( param != null )
909 {
910 for ( int j = 0; j < param.Count; j++ )
911 {
912 var p = i.Parameters [ j ];
913 if ( p.IsGenericParameter )
914 {
915 if ( IsGenericArgumentMatch ( p, param [ j ], genericArguments ) )
916 continue;
917 else
918 {
919 match = false;
920 break;
921 }
922 }
923 if ( p.IsByRef )
924 p = p.ElementType;
925 if ( p.IsArray )
926 p = p.ElementType;
927
928 var p2 = param [ j ];
929 if ( p2.IsByRef )
930 p2 = p2.ElementType;
931 if ( p2.IsArray )
932 p2 = p2.ElementType;
933 if ( p.IsGenericParameter )
934 {
935 if ( i.Parameters [ j ].IsByRef == param [ j ].IsByRef && i.Parameters [ j ].IsArray == param [ j ].IsArray && IsGenericArgumentMatch ( p, p2, genericArguments ) )
936 continue;
937 else
938 {
939 match = false;
940 break;
941 }
942 }
943 if ( p.HasGenericParameter )
944 {
945 if ( p.Name != p2.Name )
946 {
947 match = false;
948 break;
949 }
950 //TODO should match the generic parameters;
951 continue;
952 }
953
954
955 if ( p2 != p )
956 {
957 match = false;
958 break;
959 }
960 }
961 }
962 if ( match )
963 {
964 genericMethod = i;
965 }
966 return genericMethod;
