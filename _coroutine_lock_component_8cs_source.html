50 {
51 public override void Update(CoroutineLockComponent self)
52 {
53 // 检测超时的CoroutineLock
54 TimeoutCheck(self);
55
56 // 循环过程中会有对象继续加入队列
57 while(self.nextFrameRun.Count > 0)

64 private void TimeoutCheck(CoroutineLockComponent self)
65 {
66 // 超时的锁
67 if (self.timers.Count == 0)
68 {
69 return;
70 }
71
72 self.timeNow = TimeHelper.ClientFrameTime();
73
74 if (self.timeNow < self.minTime)
75 {
76 return;
77 }
78
79 self.timers.ForEachFunc(self.foreachFunc);
80
81 self.timerOutTimer.Clear();
82
83 while (self.timeOutIds.Count > 0)
84 {
85 long time = self.timeOutIds.Dequeue();
86 var list = self.timers[time];
87 for (int i = 0; i < list.Count; ++i)
88 {
89 CoroutineLockTimer coroutineLockTimer = list[i];
90 self.timerOutTimer.Enqueue(coroutineLockTimer);
91 }
92 self.timers.Remove(time);
93 }
94
95 while (self.timerOutTimer.Count > 0)
96 {
97 CoroutineLockTimer coroutineLockTimer = self.timerOutTimer.Dequeue();
98 if (coroutineLockTimer.CoroutineLockInstanceId != coroutineLockTimer.CoroutineLock.InstanceId)
99 {
100 continue;
101 }
102
103 CoroutineLock coroutineLock = coroutineLockTimer.CoroutineLock;
104 // 超时直接调用下一个锁
105 self.RunNextCoroutine(coroutineLock.coroutineLockType, coroutineLock.key, coroutineLock.level + 1);
106 coroutineLock.coroutineLockType = CoroutineLockType.None; // 上面调用了下一个, dispose不再调用
107 }
108 }

132 public static async ETTask<CoroutineLock> Wait(this CoroutineLockComponent self, int coroutineLockType, long key, int time = 60000)
133 {
134 CoroutineLockQueueType coroutineLockQueueType = self.list[coroutineLockType];
135
136 if (!coroutineLockQueueType.TryGetValue(key, out CoroutineLockQueue queue))
137 {
138 coroutineLockQueueType.Add(key, self.AddChildWithId<CoroutineLockQueue>(++self.idGenerator, true));
139 return self.CreateCoroutineLock(coroutineLockType, key, time, 1);
140 }
141 ETTask<CoroutineLock> tcs = ETTask<CoroutineLock>.Create(true);
142 queue.Add(tcs, time);
143 return await tcs;

147 {
148 CoroutineLock coroutineLock = self.AddChildWithId<CoroutineLock, int, long, int>(++self.idGenerator, coroutineLockType, key, level, true);

156 public static void Notify(this CoroutineLockComponent self, int coroutineLockType, long key, int level)
157 {
158 CoroutineLockQueueType coroutineLockQueueType = self.list[coroutineLockType];
159 if (!coroutineLockQueueType.TryGetValue(key, out CoroutineLockQueue queue))
160 {
161 return;
162 }
163
164 if (queue.Count == 0)
165 {
166 coroutineLockQueueType.Remove(key);
167 return;
168 }
169
170 CoroutineLockInfo coroutineLockInfo = queue.Dequeue();
171 coroutineLockInfo.Tcs.SetResult(self.CreateCoroutineLock(coroutineLockType, key, coroutineLockInfo.Time, level));
172 }
