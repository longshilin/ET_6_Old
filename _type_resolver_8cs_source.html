40 public MethodReference Resolve (MethodReference method)
41 {
42 var methodReference = method;
43 if (IsDummy ())
44 return methodReference;
45
46 var declaringType = Resolve (method.DeclaringType);
47
48 var genericInstanceMethod = method as GenericInstanceMethod;
49 if (genericInstanceMethod != null) {
50 methodReference = new MethodReference (method.Name, method.ReturnType, declaringType);
51
52 foreach (var p in method.Parameters)
53 methodReference.Parameters.Add (new ParameterDefinition (p.Name, p.Attributes, p.ParameterType));
54
55 foreach (var gp in genericInstanceMethod.ElementMethod.GenericParameters)
56 methodReference.GenericParameters.Add (new GenericParameter (gp.Name, methodReference));
57
58 methodReference.HasThis = method.HasThis;
59
60 var m = new GenericInstanceMethod (methodReference);
61 foreach (var ga in genericInstanceMethod.GenericArguments) {
62 m.GenericArguments.Add (Resolve (ga));
63 }
64
65 methodReference = m;
66 } else {
67 methodReference = new MethodReference (method.Name, method.ReturnType, declaringType);
68
69 foreach (var gp in method.GenericParameters)
70 methodReference.GenericParameters.Add (new GenericParameter (gp.Name, methodReference));
71
72 foreach (var p in method.Parameters)
73 methodReference.Parameters.Add (new ParameterDefinition (p.Name, p.Attributes, p.ParameterType));
74
75 methodReference.HasThis = method.HasThis;
76 }
77
78
79 return methodReference;
80 }

117 public TypeReference Resolve (TypeReference typeReference, bool includeTypeDefinitions)
118 {
119 if (IsDummy ())
120 return typeReference;
121
122 if (_typeDefinitionContext != null && _typeDefinitionContext.GenericArguments.Contains (typeReference))
123 return typeReference;
124 if (_methodDefinitionContext != null && _methodDefinitionContext.GenericArguments.Contains (typeReference))
125 return typeReference;
126
127 var genericParameter = typeReference as GenericParameter;
128 if (genericParameter != null) {
129 if (_typeDefinitionContext != null && _typeDefinitionContext.GenericArguments.Contains (genericParameter))
130 return genericParameter;
131 if (_methodDefinitionContext != null && _methodDefinitionContext.GenericArguments.Contains (genericParameter))
132 return genericParameter;
133 return ResolveGenericParameter (genericParameter);
134 }
135
136 var arrayType = typeReference as ArrayType;
137 if (arrayType != null)
138 return new ArrayType (Resolve (arrayType.ElementType), arrayType.Rank);
139
140 var pointerType = typeReference as PointerType;
141 if (pointerType != null)
142 return new PointerType (Resolve (pointerType.ElementType));
143
144 var byReferenceType = typeReference as ByReferenceType;
145 if (byReferenceType != null)
146 return new ByReferenceType (Resolve (byReferenceType.ElementType));
147
148 var pinnedType = typeReference as PinnedType;
149 if (pinnedType != null)
150 return new PinnedType (Resolve (pinnedType.ElementType));
151
152 var genericInstanceType = typeReference as GenericInstanceType;
153 if (genericInstanceType != null) {
154 var newGenericInstanceType = new GenericInstanceType (genericInstanceType.ElementType);
155 foreach (var genericArgument in genericInstanceType.GenericArguments)
156 newGenericInstanceType.GenericArguments.Add (Resolve (genericArgument));
157 return newGenericInstanceType;
158 }
159
160 var requiredModType = typeReference as RequiredModifierType;
161 if (requiredModType != null)
162 return Resolve (requiredModType.ElementType, includeTypeDefinitions);
163
164
165 if (includeTypeDefinitions) {
166 var typeDefinition = typeReference as TypeDefinition;
167 if (typeDefinition != null && typeDefinition.HasGenericParameters) {
168 var newGenericInstanceType = new GenericInstanceType (typeDefinition);
169 foreach (var gp in typeDefinition.GenericParameters)
170 newGenericInstanceType.GenericArguments.Add (Resolve (gp));
171 return newGenericInstanceType;
172 }
173 }
174
175 if (typeReference is TypeSpecification)
176 throw new NotSupportedException (string.Format ("The type {0} cannot be resolved correctly.", typeReference.FullName));
177
178 return typeReference;
179 }

186 private TypeReference ResolveGenericParameter (GenericParameter genericParameter)
187 {
188 if (genericParameter.Owner == null)
189 return HandleOwnerlessInvalidILCode (genericParameter);
190
191 var memberReference = genericParameter.Owner as MemberReference;
192 if (memberReference == null)
193 throw new NotSupportedException ();
194
195 return genericParameter.Type == GenericParameterType.Type
196 ? _typeDefinitionContext.GenericArguments[genericParameter.Position]
197 : (_methodDefinitionContext != null ? _methodDefinitionContext.GenericArguments[genericParameter.Position] : genericParameter);
198 }
199
200 private TypeReference HandleOwnerlessInvalidILCode (GenericParameter genericParameter)
201 {
202 // NOTE: If owner is null and we have a method parameter, then we'll assume that the method parameter
203 // is actually a type parameter, and we'll use the type parameter from the corresponding position. I think
204 // this assumption is valid, but if you're visiting this code then I might have been proven wrong.
205 if (genericParameter.Type == GenericParameterType.Method && (_typeDefinitionContext != null && genericParameter.Position < _typeDefinitionContext.GenericArguments.Count))
206 return _typeDefinitionContext.GenericArguments[genericParameter.Position];
207
208 // NOTE: Owner cannot be null, but sometimes the Mono compiler generates invalid IL and we
209 // end up in this situation.
210 // When we do, we assume that the runtime doesn't care about the resolved type of the GenericParameter,
211 // thus we return a reference to System.Object.
212 return genericParameter.Module.TypeSystem.Object;
213 }
