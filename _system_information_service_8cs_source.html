43 public void Add(InfoEntry info, string category = "Default")
44 {
45 IList<InfoEntry> list;
46
47 if (!_info.TryGetValue(category, out list))
48 {
49 list = new List<InfoEntry>();
50 _info.Add(category, list);
51 }
52
53 if (list.Any(p => p.Title == info.Title))
54 {
55 throw new ArgumentException("An InfoEntry object with the same title already exists in that category.", "info");
56 }
57
58 list.Add(info);
59 }

61 public Dictionary<string, Dictionary<string, object>> CreateReport(bool includePrivate = false)
62 {
63 var dict = new Dictionary<string, Dictionary<string, object>>();
64
65 foreach (var keyValuePair in _info)
66 {
67 var category = new Dictionary<string, object>();
68
69 foreach (var systemInfo in keyValuePair.Value)
70 {
71 if (systemInfo.IsPrivate && !includePrivate)
72 {
73 continue;
74 }
75
76 category.Add(systemInfo.Title, systemInfo.Value);
77 }
78
79 dict.Add(keyValuePair.Key, category);
80 }
81
82 return dict;
