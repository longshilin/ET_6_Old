201 private void PopulateStackTraceArea(ConsoleEntry entry)
202 {
203 if (entry == null)
204 {
205 SetCopyToClipboardButtonState(CopyToClipboardStates.Hidden);
206 StackTraceText.text = "";
207 }
208 else
209 {
210 if (SRDebug.CopyConsoleItemCallback != null)
211 {
212 SetCopyToClipboardButtonState(CopyToClipboardStates.Visible);
213 }
214
215 var text = entry.Message + Environment.NewLine +
216 (!string.IsNullOrEmpty(entry.StackTrace)
217 ? entry.StackTrace
218 : SRDebugStrings.Current.Console_NoStackTrace);
219
220 if (text.Length > MaxLength)
221 {
222 text = text.Substring(0, MaxLength);
223 text += "\n" + SRDebugStrings.Current.Console_MessageTruncated;
224 }
225
226 StackTraceText.text = text;
227 }
228
229 StackTraceScrollRect.normalizedPosition = new Vector2(0, 1);
230 _selectedItem = entry;
231 }

249 public enum CopyToClipboardStates

256 void SetCopyToClipboardButtonState(CopyToClipboardStates state)
257 {
258 StopAnimations();
259
260 switch (state)
261 {
262 case CopyToClipboardStates.Hidden:
263 CopyToClipboardContainer.SetActive(false);
264 CopyToClipboardButton.SetActive(false);
265 CopyToClipboardMessage.SetActive(false);
266 break;
267 case CopyToClipboardStates.Visible:
268 CopyToClipboardContainer.SetActive(true);
269 CopyToClipboardButton.SetActive(true);
270 CopyToClipboardMessage.SetActive(false);
271 break;
272 case CopyToClipboardStates.Activated:
273 CopyToClipboardMessageCanvasGroup.alpha = 1;
274 CopyToClipboardContainer.SetActive(true);
275 CopyToClipboardButton.SetActive(false);
276 CopyToClipboardMessage.SetActive(true);
277
278 _fadeButtonCoroutine = StartCoroutine(FadeCopyButton());
279 break;
280 default:
281 throw new ArgumentOutOfRangeException("state", state, null);
282 }
283 }
284
285 IEnumerator FadeCopyButton()
286 {
287 yield return new WaitForSecondsRealtime(2f);
288
289 float startTime = Time.realtimeSinceStartup;
290 float endTime = Time.realtimeSinceStartup + 1f;
291
292 while (Time.realtimeSinceStartup < endTime)
293 {
294 float currentAlpha = Mathf.InverseLerp(endTime, startTime, Time.realtimeSinceStartup);
295 CopyToClipboardMessageCanvasGroup.alpha = currentAlpha;
296 yield return new WaitForEndOfFrame();
297 }
298
299 CopyToClipboardMessageCanvasGroup.alpha = 0;
300 _fadeButtonCoroutine = null;
301 }
302
303 void StopAnimations()
304 {
305 if (_fadeButtonCoroutine != null)
306 {
307 StopCoroutine(_fadeButtonCoroutine);
308 _fadeButtonCoroutine = null;
309 CopyToClipboardMessageCanvasGroup.alpha = 0;
310 }
