94/* encode 8 bits unsigned int */
95static inline char *ikcp_encode8u(char *p, unsigned char c)
96{
97 *(unsigned char*)p++ = c;
98 return p;
99}

101/* decode 8 bits unsigned int */
102static inline const char *ikcp_decode8u(const char *p, unsigned char *c)
103{
104 *c = *(unsigned char*)p++;
105 return p;
106}
107
108/* encode 16 bits unsigned int (lsb) */
109static inline char *ikcp_encode16u(char *p, unsigned short w)
110{
111#if IWORDS_BIG_ENDIAN || IWORDS_MUST_ALIGN
112 *(unsigned char*)(p + 0) = (w & 255);
113 *(unsigned char*)(p + 1) = (w >> 8);
114#else
115 memcpy(p, &w, 2);
116#endif
117 p += 2;
118 return p;
119}
120
121/* decode 16 bits unsigned int (lsb) */
122static inline const char *ikcp_decode16u(const char *p, unsigned short *w)
123{
124#if IWORDS_BIG_ENDIAN || IWORDS_MUST_ALIGN
125 *w = *(const unsigned char*)(p + 1);
126 *w = *(const unsigned char*)(p + 0) + (*w << 8);
127#else
128 memcpy(w, p, 2);
129#endif
130 p += 2;
131 return p;
132}
133
134/* encode 32 bits unsigned int (lsb) */
135static inline char *ikcp_encode32u(char *p, IUINT32 l)
136{
137#if IWORDS_BIG_ENDIAN || IWORDS_MUST_ALIGN
138 *(unsigned char*)(p + 0) = (unsigned char)((l >> 0) & 0xff);
139 *(unsigned char*)(p + 1) = (unsigned char)((l >> 8) & 0xff);
140 *(unsigned char*)(p + 2) = (unsigned char)((l >> 16) & 0xff);
141 *(unsigned char*)(p + 3) = (unsigned char)((l >> 24) & 0xff);
142#else
143 memcpy(p, &l, 4);
144#endif
145 p += 4;
146 return p;
147}
148
149/* decode 32 bits unsigned int (lsb) */
150static inline const char *ikcp_decode32u(const char *p, IUINT32 *l)
151{
152#if IWORDS_BIG_ENDIAN || IWORDS_MUST_ALIGN
153 *l = *(const unsigned char*)(p + 3);
154 *l = *(const unsigned char*)(p + 2) + (*l << 8);
155 *l = *(const unsigned char*)(p + 1) + (*l << 8);
156 *l = *(const unsigned char*)(p + 0) + (*l << 8);
157#else
158 memcpy(l, p, 4);
159#endif
160 p += 4;
161 return p;
162}

165 return a <= b ? a : b;

169 return a >= b ? a : b;

173{
174 return _imin_(_imax_(lower, middle), upper);

334//---------------------------------------------------------------------
335// release a new kcpcb
336//---------------------------------------------------------------------
337void ikcp_release(ikcpcb *kcp)
338{
339 assert(kcp);
340 if (kcp) {
341 IKCPSEG *seg;
342 while (!iqueue_is_empty(&kcp->snd_buf)) {
343 seg = iqueue_entry(kcp->snd_buf.next, IKCPSEG, node);
344 iqueue_del(&seg->node);
345 ikcp_segment_delete(kcp, seg);
346 }
347 while (!iqueue_is_empty(&kcp->rcv_buf)) {
348 seg = iqueue_entry(kcp->rcv_buf.next, IKCPSEG, node);
349 iqueue_del(&seg->node);
350 ikcp_segment_delete(kcp, seg);
351 }
352 while (!iqueue_is_empty(&kcp->snd_queue)) {
353 seg = iqueue_entry(kcp->snd_queue.next, IKCPSEG, node);
354 iqueue_del(&seg->node);
355 ikcp_segment_delete(kcp, seg);
356 }
357 while (!iqueue_is_empty(&kcp->rcv_queue)) {
358 seg = iqueue_entry(kcp->rcv_queue.next, IKCPSEG, node);
359 iqueue_del(&seg->node);
360 ikcp_segment_delete(kcp, seg);
361 }
362 if (kcp->buffer) {
363 ikcp_free(kcp->buffer);
364 }
365 if (kcp->acklist) {
366 ikcp_free(kcp->acklist);
367 }
368
369 kcp->nrcv_buf = 0;
370 kcp->nsnd_buf = 0;
371 kcp->nrcv_que = 0;
372 kcp->nsnd_que = 0;
373 kcp->ackcount = 0;
374 kcp->buffer = NULL;
375 kcp->acklist = NULL;
376 ikcp_free(kcp);
377 }
378}

395//---------------------------------------------------------------------
396// user/upper level recv: returns size, returns below zero for EAGAIN
397//---------------------------------------------------------------------
398int ikcp_recv(ikcpcb *kcp, char *buffer, int index, int len)
399{
400 struct IQUEUEHEAD *p;
401 int ispeek = (len < 0)? 1 : 0;
402 int peeksize;
403 int recover = 0;
404 IKCPSEG *seg;
405 assert(kcp);
406 buffer += index;
407
408 if (iqueue_is_empty(&kcp->rcv_queue))
409 return -1;
410
411 if (len < 0) len = -len;
412
413 peeksize = ikcp_peeksize(kcp);
414
415 if (peeksize < 0)
416 return -2;
417
418 if (peeksize > len)
419 return -3;
420
421 if (kcp->nrcv_que >= kcp->rcv_wnd)
422 recover = 1;
423
424 // merge fragment
425 for (len = 0, p = kcp->rcv_queue.next; p != &kcp->rcv_queue; ) {
426 int fragment;
427 seg = iqueue_entry(p, IKCPSEG, node);
428 p = p->next;
429
430 if (buffer) {
431 memcpy(buffer, seg->data, seg->len);
432 buffer += seg->len;
433 }
434
435 len += seg->len;
436 fragment = seg->frg;
437
438 if (ikcp_canlog(kcp, IKCP_LOG_RECV)) {
439 ikcp_log(kcp, IKCP_LOG_RECV, "recv sn=%lu", (unsigned long)seg->sn);
440 }
441
442 if (ispeek == 0) {
443 iqueue_del(&seg->node);
444 ikcp_segment_delete(kcp, seg);
445 kcp->nrcv_que--;
446 }
447
448 if (fragment == 0)
449 break;
450 }
451
452 assert(len == peeksize);
453
454 // move available data from rcv_buf -> rcv_queue
455 while (! iqueue_is_empty(&kcp->rcv_buf)) {
456 seg = iqueue_entry(kcp->rcv_buf.next, IKCPSEG, node);
457 if (seg->sn == kcp->rcv_nxt && kcp->nrcv_que < kcp->rcv_wnd) {
458 iqueue_del(&seg->node);
459 kcp->nrcv_buf--;
460 iqueue_add_tail(&seg->node, &kcp->rcv_queue);
461 kcp->nrcv_que++;
462 kcp->rcv_nxt++;
463 } else {
464 break;
465 }
466 }
467
468 // fast recover
469 if (kcp->nrcv_que < kcp->rcv_wnd && recover) {
470 // ready to send back IKCP_CMD_WINS in ikcp_flush
471 // tell remote my window size
472 kcp->probe |= IKCP_ASK_TELL;
473 }
474
475 return len;
476}

479//---------------------------------------------------------------------
480// peek data size
481//---------------------------------------------------------------------
482int ikcp_peeksize(const ikcpcb *kcp)
483{
484 struct IQUEUEHEAD *p;
485 IKCPSEG *seg;
486 int length = 0;
487
488 assert(kcp);
489
490 if (iqueue_is_empty(&kcp->rcv_queue)) return -1;
491
492 seg = iqueue_entry(kcp->rcv_queue.next, IKCPSEG, node);
493 if (seg->frg == 0) return seg->len;
494
495 if (kcp->nrcv_que < seg->frg + 1) return -1;
496
497 for (p = kcp->rcv_queue.next; p != &kcp->rcv_queue; p = p->next) {
498 seg = iqueue_entry(p, IKCPSEG, node);
499 length += seg->len;
500 if (seg->frg == 0) break;
501 }
502
503 return length;
504}

507//---------------------------------------------------------------------
508// user/upper level send, returns below zero for error
509//---------------------------------------------------------------------
510int ikcp_send(ikcpcb *kcp, const char *buffer, int offset, int len)
511{
512 IKCPSEG *seg;
513 int count, i;
514
515 assert(kcp->mss > 0);
516 if (len < 0) return -1;
517
518 buffer += offset;
519
520 // append to previous segment in streaming mode (if possible)
521 if (kcp->stream != 0) {
522 if (!iqueue_is_empty(&kcp->snd_queue)) {
523 IKCPSEG *old = iqueue_entry(kcp->snd_queue.prev, IKCPSEG, node);
524 if (old->len < kcp->mss) {
525 int capacity = kcp->mss - old->len;
526 int extend = (len < capacity)? len : capacity;
527 seg = ikcp_segment_new(kcp, old->len + extend);
528 assert(seg);
529 if (seg == NULL) {
530 return -2;
531 }
532 iqueue_add_tail(&seg->node, &kcp->snd_queue);
533 memcpy(seg->data, old->data, old->len);
534 if (buffer) {
535 memcpy(seg->data + old->len, buffer, extend);
536 buffer += extend;
537 }
538 seg->len = old->len + extend;
539 seg->frg = 0;
540 len -= extend;
541 iqueue_del_init(&old->node);
542 ikcp_segment_delete(kcp, old);
543 }
544 }
545 if (len <= 0) {
546 return 0;
547 }
548 }
549
550 if (len <= (int)kcp->mss) count = 1;
551 else count = (len + kcp->mss - 1) / kcp->mss;
552
553 if (count >= (int)IKCP_WND_RCV) return -2;
554
555 if (count == 0) count = 1;
556
557 // fragment
558 for (i = 0; i < count; i++) {
559 int size = len > (int)kcp->mss ? (int)kcp->mss : len;
560 seg = ikcp_segment_new(kcp, size);
561 assert(seg);
562 if (seg == NULL) {
563 return -2;
564 }
565 if (buffer && len > 0) {
566 memcpy(seg->data, buffer, size);
567 }
568 seg->len = size;
569 seg->frg = (kcp->stream == 0)? (count - i - 1) : 0;
570 iqueue_init(&seg->node);
571 iqueue_add_tail(&seg->node, &kcp->snd_queue);
572 kcp->nsnd_que++;
573 if (buffer) {
574 buffer += size;
575 }
576 len -= size;
577 }
578
579 return 0;
580}

583//---------------------------------------------------------------------
584// parse ack
585//---------------------------------------------------------------------
586static void ikcp_update_ack(ikcpcb *kcp, IINT32 rtt)
587{
588 IINT32 rto = 0;
589 if (kcp->rx_srtt == 0) {
590 kcp->rx_srtt = rtt;
591 kcp->rx_rttval = rtt / 2;
592 } else {
593 long delta = rtt - kcp->rx_srtt;
594 if (delta < 0) delta = -delta;
595 kcp->rx_rttval = (3 * kcp->rx_rttval + delta) / 4;
596 kcp->rx_srtt = (7 * kcp->rx_srtt + rtt) / 8;
597 if (kcp->rx_srtt < 1) kcp->rx_srtt = 1;
598 }
599 rto = kcp->rx_srtt + _imax_(kcp->interval, 4 * kcp->rx_rttval);
600 kcp->rx_rto = _ibound_(kcp->rx_minrto, rto, IKCP_RTO_MAX);
601}

603static void ikcp_shrink_buf(ikcpcb *kcp)
604{
605 struct IQUEUEHEAD *p = kcp->snd_buf.next;
606 if (p != &kcp->snd_buf) {
607 IKCPSEG *seg = iqueue_entry(p, IKCPSEG, node);
608 kcp->snd_una = seg->sn;
609 } else {
610 kcp->snd_una = kcp->snd_nxt;
611 }

614static void ikcp_parse_ack(ikcpcb *kcp, IUINT32 sn)
615{
616 struct IQUEUEHEAD *p, *next;
617
618 if (_itimediff(sn, kcp->snd_una) < 0 || _itimediff(sn, kcp->snd_nxt) >= 0)
619 return;
620
621 for (p = kcp->snd_buf.next; p != &kcp->snd_buf; p = next) {
622 IKCPSEG *seg = iqueue_entry(p, IKCPSEG, node);
623 next = p->next;
624 if (sn == seg->sn) {
625 iqueue_del(p);
626 ikcp_segment_delete(kcp, seg);
627 kcp->nsnd_buf--;
628 break;
629 }
630 if (_itimediff(sn, seg->sn) < 0) {
631 break;
632 }
633 }
634}

636static void ikcp_parse_una(ikcpcb *kcp, IUINT32 una)
637{
638 struct IQUEUEHEAD *p, *next;
639 for (p = kcp->snd_buf.next; p != &kcp->snd_buf; p = next) {
640 IKCPSEG *seg = iqueue_entry(p, IKCPSEG, node);
641 next = p->next;
642 if (_itimediff(una, seg->sn) > 0) {
643 iqueue_del(p);
644 ikcp_segment_delete(kcp, seg);
645 kcp->nsnd_buf--;
646 } else {
647 break;
648 }
649 }
650}

652static void ikcp_parse_fastack(ikcpcb *kcp, IUINT32 sn, IUINT32 ts)
653{
654 struct IQUEUEHEAD *p, *next;
655
656 if (_itimediff(sn, kcp->snd_una) < 0 || _itimediff(sn, kcp->snd_nxt) >= 0)
657 return;
658
659 for (p = kcp->snd_buf.next; p != &kcp->snd_buf; p = next) {
660 IKCPSEG *seg = iqueue_entry(p, IKCPSEG, node);
661 next = p->next;
662 if (_itimediff(sn, seg->sn) < 0) {
663 break;
664 }
665 else if (sn != seg->sn) {
666 #ifndef IKCP_FASTACK_CONSERVE
667 seg->fastack++;
668 #else
669 if (_itimediff(ts, seg->ts) >= 0)
670 seg->fastack++;
671 #endif
672 }
673 }

677//---------------------------------------------------------------------
678// ack append
679//---------------------------------------------------------------------
680static void ikcp_ack_push(ikcpcb *kcp, IUINT32 sn, IUINT32 ts)
681{
682 size_t newsize = kcp->ackcount + 1;
683 IUINT32 *ptr;
684
685 if (newsize > kcp->ackblock) {
686 IUINT32 *acklist;
687 size_t newblock;
688
689 for (newblock = 8; newblock < newsize; newblock <<= 1);
690 acklist = (IUINT32*)ikcp_malloc(newblock * sizeof(IUINT32) * 2);
691
692 if (acklist == NULL) {
693 assert(acklist != NULL);
694 abort();
695 }
696
697 if (kcp->acklist != NULL) {
698 size_t x;
699 for (x = 0; x < kcp->ackcount; x++) {
700 acklist[x * 2 + 0] = kcp->acklist[x * 2 + 0];
701 acklist[x * 2 + 1] = kcp->acklist[x * 2 + 1];
702 }
703 ikcp_free(kcp->acklist);
704 }
705
706 kcp->acklist = acklist;
707 kcp->ackblock = newblock;
708 }
709
710 ptr = &kcp->acklist[kcp->ackcount * 2];
711 ptr[0] = sn;
712 ptr[1] = ts;
713 kcp->ackcount++;

723//---------------------------------------------------------------------
724// parse data
725//---------------------------------------------------------------------
726void ikcp_parse_data(ikcpcb *kcp, IKCPSEG *newseg)
727{
728 struct IQUEUEHEAD *p, *prev;
729 IUINT32 sn = newseg->sn;
730 int repeat = 0;
731
732 if (_itimediff(sn, kcp->rcv_nxt + kcp->rcv_wnd) >= 0 ||
733 _itimediff(sn, kcp->rcv_nxt) < 0) {
734 ikcp_segment_delete(kcp, newseg);
735 return;
736 }
737
738 for (p = kcp->rcv_buf.prev; p != &kcp->rcv_buf; p = prev) {
739 IKCPSEG *seg = iqueue_entry(p, IKCPSEG, node);
740 prev = p->prev;
741 if (seg->sn == sn) {
742 repeat = 1;
743 break;
744 }
745 if (_itimediff(sn, seg->sn) > 0) {
746 break;
747 }
748 }
749
750 if (repeat == 0) {
751 iqueue_init(&newseg->node);
752 iqueue_add(&newseg->node, p);
753 kcp->nrcv_buf++;
754 } else {
755 ikcp_segment_delete(kcp, newseg);
756 }
757
758#if 0
759 ikcp_qprint("rcvbuf", &kcp->rcv_buf);
760 printf("rcv_nxt=%lu\n", kcp->rcv_nxt);
761#endif
762
763 // move available data from rcv_buf -> rcv_queue
764 while (! iqueue_is_empty(&kcp->rcv_buf)) {
765 IKCPSEG *seg = iqueue_entry(kcp->rcv_buf.next, IKCPSEG, node);
766 if (seg->sn == kcp->rcv_nxt && kcp->nrcv_que < kcp->rcv_wnd) {
767 iqueue_del(&seg->node);
768 kcp->nrcv_buf--;
769 iqueue_add_tail(&seg->node, &kcp->rcv_queue);
770 kcp->nrcv_que++;
771 kcp->rcv_nxt++;
772 } else {
773 break;
774 }
775 }
776
777#if 0
778 ikcp_qprint("queue", &kcp->rcv_queue);
779 printf("rcv_nxt=%lu\n", kcp->rcv_nxt);
780#endif
781
782#if 1
783// printf("snd(buf=%d, queue=%d)\n", kcp->nsnd_buf, kcp->nsnd_que);
784// printf("rcv(buf=%d, queue=%d)\n", kcp->nrcv_buf, kcp->nrcv_que);
785#endif
786}

789//---------------------------------------------------------------------
790// input data
791//---------------------------------------------------------------------
792int ikcp_input(ikcpcb *kcp, const char *data, int offset, int size)
793{
794 IUINT32 prev_una = kcp->snd_una;
795 IUINT32 maxack = 0, latest_ts = 0;
796 int flag = 0;
797
798 if (data == NULL || (int)size < (int)IKCP_OVERHEAD) return -1;
799
800 data += offset;
801 while (1) {
802 IUINT32 ts, sn, len, una, conv;
803 IUINT16 wnd;
804 IUINT8 cmd, frg;
805 IKCPSEG *seg;
806
807 if (size < (int)IKCP_OVERHEAD) break;
808
809 data = ikcp_decode32u(data, &conv);
810 /*if (conv != kcp->conv) return -1;*/
811 data = ikcp_decode8u(data, &cmd);
812 data = ikcp_decode8u(data, &frg);
813 data = ikcp_decode16u(data, &wnd);
814 data = ikcp_decode32u(data, &ts);
815 data = ikcp_decode32u(data, &sn);
816 data = ikcp_decode32u(data, &una);
817 data = ikcp_decode32u(data, &len);
818 size -= IKCP_OVERHEAD;
819
820 if ((long)size < (long)len || (int)len < 0) return -2;
821
822 if (cmd != IKCP_CMD_PUSH && cmd != IKCP_CMD_ACK &&
823 cmd != IKCP_CMD_WASK && cmd != IKCP_CMD_WINS)
824 return -3;
825 kcp->rmt_wnd = wnd;
826 ikcp_parse_una(kcp, una);
827 ikcp_shrink_buf(kcp);
828 if (cmd == IKCP_CMD_ACK) {
829 if (_itimediff(kcp->current, ts) >= 0) {
830 ikcp_update_ack(kcp, _itimediff(kcp->current, ts));
831 }
832 ikcp_parse_ack(kcp, sn);
833 ikcp_shrink_buf(kcp);
834 if (flag == 0) {
835 flag = 1;
836 maxack = sn;
837 latest_ts = ts;
838 } else {
839 if (_itimediff(sn, maxack) > 0) {
840 #ifndef IKCP_FASTACK_CONSERVE
841 maxack = sn;
842 latest_ts = ts;
843 #else
844 if (_itimediff(ts, latest_ts) > 0) {
845 maxack = sn;
846 latest_ts = ts;
847 }
848 #endif
849 }
850 }
851 if (ikcp_canlog(kcp, IKCP_LOG_IN_ACK)) {
852 ikcp_log(kcp, IKCP_LOG_IN_ACK,
853 "input ack: sn=%lu rtt=%ld rto=%ld", (unsigned long)sn,
854 (long)_itimediff(kcp->current, ts),
855 (long)kcp->rx_rto);
856 }
857 }
858 else if (cmd == IKCP_CMD_PUSH) {
859 if (ikcp_canlog(kcp, IKCP_LOG_IN_DATA)) {
860 ikcp_log(kcp, IKCP_LOG_IN_DATA,
861 "input psh: sn=%lu ts=%lu", (unsigned long)sn, (unsigned long)ts);
862 }
863 if (_itimediff(sn, kcp->rcv_nxt + kcp->rcv_wnd) < 0) {
864 ikcp_ack_push(kcp, sn, ts);
865 if (_itimediff(sn, kcp->rcv_nxt) >= 0) {
866 seg = ikcp_segment_new(kcp, len);
867 seg->conv = conv;
868 seg->cmd = cmd;
869 seg->frg = frg;
870 seg->wnd = wnd;
871 seg->ts = ts;
872 seg->sn = sn;
873 seg->una = una;
874 seg->len = len;
875
876 if (len > 0) {
877 memcpy(seg->data, data, len);
878 }
879
880 ikcp_parse_data(kcp, seg);
881 }
882 }
883 }
884 else if (cmd == IKCP_CMD_WASK) {
885 // ready to send back IKCP_CMD_WINS in ikcp_flush
886 // tell remote my window size
887 kcp->probe |= IKCP_ASK_TELL;
888 if (ikcp_canlog(kcp, IKCP_LOG_IN_PROBE)) {
889 ikcp_log(kcp, IKCP_LOG_IN_PROBE, "input probe");
890 }
891 }
892 else if (cmd == IKCP_CMD_WINS) {
893 // do nothing
894 if (ikcp_canlog(kcp, IKCP_LOG_IN_WINS)) {
895 ikcp_log(kcp, IKCP_LOG_IN_WINS,
896 "input wins: %lu", (unsigned long)(wnd));
897 }
898 }
899 else {
900 return -3;
901 }
902 data += len;
903 size -= len;
904 }
905
906 if (flag != 0) {
907 ikcp_parse_fastack(kcp, maxack, latest_ts);
908 }
909
910 if (_itimediff(kcp->snd_una, prev_una) > 0) {
911 if (kcp->cwnd < kcp->rmt_wnd) {
912 IUINT32 mss = kcp->mss;
913 if (kcp->cwnd < kcp->ssthresh) {
914 kcp->cwnd++;
915 kcp->incr += mss;
916 } else {
917 if (kcp->incr < mss) kcp->incr = mss;
918 kcp->incr += (mss * mss) / kcp->incr + (mss / 16);
919 if ((kcp->cwnd + 1) * mss <= kcp->incr) {
920 #if 1
921 kcp->cwnd = (kcp->incr + mss - 1) / ((mss > 0)? mss : 1);
922 #else
923 kcp->cwnd++;
924 #endif
925 }
926 }
927 if (kcp->cwnd > kcp->rmt_wnd) {
928 kcp->cwnd = kcp->rmt_wnd;
929 kcp->incr = kcp->rmt_wnd * mss;
930 }
931 }
932 }
933
934 return 0;
935}

963//---------------------------------------------------------------------
964// ikcp_flush
965//---------------------------------------------------------------------
966void ikcp_flush(ikcpcb *kcp)
967{
968 IUINT32 current = kcp->current;
969 char *buffer = kcp->buffer;
970 char *ptr = buffer;
971 int count, size, i;
972 IUINT32 resent, cwnd;
973 IUINT32 rtomin;
974 struct IQUEUEHEAD *p;
975 int change = 0;
976 int lost = 0;
977 IKCPSEG seg;
978
979 // 'ikcp_update' haven't been called.
980 if (kcp->updated == 0) return;
981 seg.conv = kcp->conv;
982 seg.cmd = IKCP_CMD_ACK;
983 seg.frg = 0;
984 seg.wnd = ikcp_wnd_unused(kcp);
985 seg.una = kcp->rcv_nxt;
986 seg.len = 0;
987 seg.sn = 0;
988 seg.ts = 0;
989 // flush acknowledges
990 count = kcp->ackcount;
991 for (i = 0; i < count; i++) {
992 size = (int)(ptr - buffer);
993 if (size + (int)IKCP_OVERHEAD > (int)kcp->mtu) {
994 ikcp_output(kcp, buffer, size);
995 ptr = buffer;
996 }
997 ikcp_ack_get(kcp, i, &seg.sn, &seg.ts);
998 ptr = ikcp_encode_seg(ptr, &seg);
999 }
1000
1001 kcp->ackcount = 0;
1002 // probe window size (if remote window size equals zero)
1003 if (kcp->rmt_wnd == 0) {
1004 if (kcp->probe_wait == 0) {
1005 kcp->probe_wait = IKCP_PROBE_INIT;
1006 kcp->ts_probe = kcp->current + kcp->probe_wait;
1007 }
1008 else {
1009 if (_itimediff(kcp->current, kcp->ts_probe) >= 0) {
1010 if (kcp->probe_wait < IKCP_PROBE_INIT)
1011 kcp->probe_wait = IKCP_PROBE_INIT;
1012 kcp->probe_wait += kcp->probe_wait / 2;
1013 if (kcp->probe_wait > IKCP_PROBE_LIMIT)
1014 kcp->probe_wait = IKCP_PROBE_LIMIT;
1015 kcp->ts_probe = kcp->current + kcp->probe_wait;
1016 kcp->probe |= IKCP_ASK_SEND;
1017 }
1018 }
1019 } else {
1020 kcp->ts_probe = 0;
1021 kcp->probe_wait = 0;
1022 }
1023 // flush window probing commands
1024 if (kcp->probe & IKCP_ASK_SEND) {
1025 seg.cmd = IKCP_CMD_WASK;
1026 size = (int)(ptr - buffer);
1027 if (size + (int)IKCP_OVERHEAD > (int)kcp->mtu) {
1028 ikcp_output(kcp, buffer, size);
1029 ptr = buffer;
1030 }
1031 ptr = ikcp_encode_seg(ptr, &seg);
1032 }
1033
1034 // flush window probing commands
1035 if (kcp->probe & IKCP_ASK_TELL) {
1036 seg.cmd = IKCP_CMD_WINS;
1037 size = (int)(ptr - buffer);
1038 if (size + (int)IKCP_OVERHEAD > (int)kcp->mtu) {
1039 ikcp_output(kcp, buffer, size);
1040 ptr = buffer;
1041 }
1042 ptr = ikcp_encode_seg(ptr, &seg);
1043 }
1044 kcp->probe = 0;
1045
1046 // calculate window size
1047 cwnd = _imin_(kcp->snd_wnd, kcp->rmt_wnd);
1048 if (kcp->nocwnd == 0) cwnd = _imin_(kcp->cwnd, cwnd);
1049 // move data from snd_queue to snd_buf
1050 while (_itimediff(kcp->snd_nxt, kcp->snd_una + cwnd) < 0) {
1051 IKCPSEG *newseg;
1052 if (iqueue_is_empty(&kcp->snd_queue)) break;
1053
1054 newseg = iqueue_entry(kcp->snd_queue.next, IKCPSEG, node);
1055
1056 iqueue_del(&newseg->node);
1057 iqueue_add_tail(&newseg->node, &kcp->snd_buf);
1058 kcp->nsnd_que--;
1059 kcp->nsnd_buf++;
1060
1061 newseg->conv = kcp->conv;
1062 newseg->cmd = IKCP_CMD_PUSH;
1063 newseg->wnd = seg.wnd;
1064 newseg->ts = current;
1065 newseg->sn = kcp->snd_nxt++;
1066 newseg->una = kcp->rcv_nxt;
1067 newseg->resendts = current;
1068 newseg->rto = kcp->rx_rto;
1069 newseg->fastack = 0;
1070 newseg->xmit = 0;
1071 }
1072 // calculate resent
1073 resent = (kcp->fastresend > 0)? (IUINT32)kcp->fastresend : 0xffffffff;
1074 rtomin = (kcp->nodelay == 0)? (kcp->rx_rto >> 3) : 0;
1075 // flush data segments
1076 for (p = kcp->snd_buf.next; p != &kcp->snd_buf; p = p->next) {
1077 IKCPSEG *segment = iqueue_entry(p, IKCPSEG, node);
1078 int needsend = 0;
1079 if (segment->xmit == 0) {
1080 needsend = 1;
1081 segment->xmit++;
1082 segment->rto = kcp->rx_rto;
1083 segment->resendts = current + segment->rto + rtomin;
1084 }
1085 else if (_itimediff(current, segment->resendts) >= 0) {
1086 needsend = 1;
1087 segment->xmit++;
1088 kcp->xmit++;
1089 if (kcp->nodelay == 0) {
1090 segment->rto += _imax_(segment->rto, (IUINT32)kcp->rx_rto);
1091 } else {
1092 IINT32 step = (kcp->nodelay < 2)?
1093 ((IINT32)(segment->rto)) : kcp->rx_rto;
1094 segment->rto += step / 2;
1095 }
1096 segment->resendts = current + segment->rto;
1097 lost = 1;
1098 }
1099 else if (segment->fastack >= resent) {
1100 if ((int)segment->xmit <= kcp->fastlimit ||
1101 kcp->fastlimit <= 0) {
1102 needsend = 1;
1103 segment->xmit++;
1104 segment->fastack = 0;
1105 segment->resendts = current + segment->rto;
1106 change++;
1107 }
1108 }
1109
1110 if (needsend) {
1111 int need;
1112 segment->ts = current;
1113 segment->wnd = seg.wnd;
1114 segment->una = kcp->rcv_nxt;
1115
1116 size = (int)(ptr - buffer);
1117 need = IKCP_OVERHEAD + segment->len;
1118
1119 if (size + need > (int)kcp->mtu) {
1120 ikcp_output(kcp, buffer, size);
1121 ptr = buffer;
1122 }
1123
1124 ptr = ikcp_encode_seg(ptr, segment);
1125
1126 if (segment->len > 0) {
1127 memcpy(ptr, segment->data, segment->len);
1128 ptr += segment->len;
1129 }
1130
1131 if (segment->xmit >= kcp->dead_link) {
1132 kcp->state = (IUINT32)-1;
1133 }
1134 }
1135 }
1136 // flash remain segments
1137 size = (int)(ptr - buffer);
1138 if (size > 0) {
1139 ikcp_output(kcp, buffer, size);
1140 }
1141 // update ssthresh
1142 if (change) {
1143 IUINT32 inflight = kcp->snd_nxt - kcp->snd_una;
1144 kcp->ssthresh = inflight / 2;
1145 if (kcp->ssthresh < IKCP_THRESH_MIN)
1146 kcp->ssthresh = IKCP_THRESH_MIN;
1147 kcp->cwnd = kcp->ssthresh + resent;
1148 kcp->incr = kcp->cwnd * kcp->mss;
1149 }
1150 if (lost) {
1151 kcp->ssthresh = cwnd / 2;
1152 if (kcp->ssthresh < IKCP_THRESH_MIN)
1153 kcp->ssthresh = IKCP_THRESH_MIN;
1154 kcp->cwnd = 1;
1155 kcp->incr = kcp->mss;
1156 }
1157 if (kcp->cwnd < 1) {
1158 kcp->cwnd = 1;
1159 kcp->incr = kcp->mss;
1160 }
1161}

1164//---------------------------------------------------------------------
1165// update state (call it repeatedly, every 10ms-100ms), or you can ask
1166// ikcp_check when to call it again (without ikcp_input/_send calling).
1167// 'current' - current timestamp in millisec.
1168//---------------------------------------------------------------------
1169void ikcp_update(ikcpcb *kcp, IUINT32 current)
1170{
1171 IINT32 slap;
1172
1173 kcp->current = current;
1174
1175 if (kcp->updated == 0) {
1176 kcp->updated = 1;
1177 kcp->ts_flush = kcp->current;
1178 }
1179
1180 slap = _itimediff(kcp->current, kcp->ts_flush);
1181
1182 if (slap >= 10000 || slap < -10000) {
1183 kcp->ts_flush = kcp->current;
1184 slap = 0;
1185 }
1186 if (slap >= 0) {
1187 kcp->ts_flush += kcp->interval;
1188 if (_itimediff(kcp->current, kcp->ts_flush) >= 0) {
1189 kcp->ts_flush = kcp->current + kcp->interval;
1190 }
1191 ikcp_flush(kcp);
1192 }
1193}

1196//---------------------------------------------------------------------
1197// Determine when should you invoke ikcp_update:
1198// returns when you should invoke ikcp_update in millisec, if there
1199// is no ikcp_input/_send calling. you can call ikcp_update in that
1200// time, instead of call update repeatly.
1201// Important to reduce unnacessary ikcp_update invoking. use it to
1202// schedule ikcp_update (eg. implementing an epoll-like mechanism,
1203// or optimize ikcp_update when handling massive kcp connections)
1204//---------------------------------------------------------------------
1205IUINT32 ikcp_check(const ikcpcb *kcp, IUINT32 current)
1206{
1207 IUINT32 ts_flush = kcp->ts_flush;
1208 IINT32 tm_flush = 0x7fffffff;
1209 IINT32 tm_packet = 0x7fffffff;
1210 IUINT32 minimal = 0;
1211 struct IQUEUEHEAD *p;
1212
1213 if (kcp->updated == 0) {
1214 return current;
1215 }
1216
1217 if (_itimediff(current, ts_flush) >= 10000 ||
1218 _itimediff(current, ts_flush) < -10000) {
1219 ts_flush = current;
1220 }
1221
1222 if (_itimediff(current, ts_flush) >= 0) {
1223 return current;
1224 }
1225
1226 tm_flush = _itimediff(ts_flush, current);
1227
1228 for (p = kcp->snd_buf.next; p != &kcp->snd_buf; p = p->next) {
1229 const IKCPSEG *seg = iqueue_entry(p, const IKCPSEG, node);
1230 IINT32 diff = _itimediff(seg->resendts, current);
1231 if (diff <= 0) {
1232 return current;
1233 }
1234 if (diff < tm_packet) tm_packet = diff;
1235 }
1236
1237 minimal = (IUINT32)(tm_packet < tm_flush ? tm_packet : tm_flush);
1238 if (minimal >= kcp->interval) minimal = kcp->interval;
1239
1240 return current + minimal;
1241}
