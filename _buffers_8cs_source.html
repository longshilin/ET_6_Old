117 public void WriteTableHeap ()
118 {
119 WriteUInt32 (0); // Reserved
120 WriteByte (GetTableHeapVersion ()); // MajorVersion
121 WriteByte (0); // MinorVersion
122 WriteByte (GetHeapSizes ()); // HeapSizes
123 WriteByte (10); // Reserved2
124 WriteUInt64 (GetValid ()); // Valid
125 WriteUInt64 (0xc416003301fa00); // Sorted
126
127 WriteRowCount ();
128 WriteTables ();
129 }

131 void WriteRowCount ()
132 {
133 for (int i = 0; i < tables.Length; i++) {
134 var table = tables [i];
135 if (table == null || table.Length == 0)
136 continue;
137
138 WriteUInt32 ((uint) table.Length);
139 }

142 void WriteTables ()
143 {
144 for (int i = 0; i < tables.Length; i++) {
145 var table = tables [i];
146 if (table == null || table.Length == 0)
147 continue;

153 ulong GetValid ()
154 {
155 ulong valid = 0;
156
157 for (int i = 0; i < tables.Length; i++) {
158 var table = tables [i];
159 if (table == null || table.Length == 0)
160 continue;
161
162 table.Sort ();

220 public void FixupData (RVA data_rva)
221 {
222 var table = GetTable<FieldRVATable> (Table.FieldRVA);
223 if (table.length == 0)
224 return;
225
226 var field_idx_size = GetTable<FieldTable> (Table.Field).IsLarge ? 4 : 2;
227 var previous = this.position;
228
229 base.position = table.position;
230 for (int i = 0; i < table.length; i++) {
231 var rva = ReadUInt32 ();
232 base.position -= 4;
233 WriteUInt32 (rva + data_rva);
234 base.position += field_idx_size;
235 }
236
237 base.position = previous;
238 }

342 public uint [] WriteStrings ()
343 {
344 var sorted = SortStrings (strings);
345 strings = null;
346
347 // Add 1 for empty string whose index and offset are both 0
348 var string_offsets = new uint [sorted.Count + 1];
349 string_offsets [0] = 0;
350
351 // Find strings that can be folded
352 var previous = string.Empty;
353 foreach (var entry in sorted) {
354 var @string = entry.Key;
355 var index = entry.Value;
356 var position = base.position;
357
358 if (previous.EndsWith (@string, StringComparison.Ordinal) && !IsLowSurrogateChar (entry.Key [0])) {
359 // Map over the tail of prev string. Watch for null-terminator of prev string.
360 string_offsets [index] = (uint) (position - (Encoding.UTF8.GetByteCount (entry.Key) + 1));
361 } else {
362 string_offsets [index] = (uint) position;
363 WriteString (@string);
364 }
365
366 previous = entry.Key;
367 }
368
369 return string_offsets;
370 }

372 static List<KeyValuePair<string, uint>> SortStrings (Dictionary<string, uint> strings)
373 {
374 var sorted = new List<KeyValuePair<string, uint>> (strings);
375 sorted.Sort (new SuffixSort ());
376 return sorted;
377 }

379 static bool IsLowSurrogateChar (int c)
380 {
381 return unchecked((uint)(c - 0xDC00)) <= 0xDFFF - 0xDC00;
382 }

384 protected virtual void WriteString (string @string)
385 {
386 WriteBytes (Encoding.UTF8.GetBytes (@string));
387 WriteByte (0);
388 }
389
390 // Sorts strings such that a string is followed immediately by all strings
391 // that are a suffix of it.
392 private class SuffixSort : IComparer<KeyValuePair<string, uint>> {
393
394 public int Compare(KeyValuePair<string, uint> xPair, KeyValuePair<string, uint> yPair)
395 {
396 var x = xPair.Key;
397 var y = yPair.Key;
398
399 for (int i = x.Length - 1, j = y.Length - 1; i >= 0 & j >= 0; i--, j--) {
400 if (x [i] < y [j]) {
401 return -1;
402 }
403
404 if (x [i] > y [j]) {
405 return +1;
406 }
407 }
408
409 return y.Length.CompareTo (x.Length);
410 }
411 }

461 protected override void WriteString (string @string)
462 {

468 var @char = @string [i];
469 WriteUInt16 (@char);
470
471 if (special == 1)

474 if (@char < 0x20 || @char > 0x7e) {
475 if (@char > 0x7e
476 || (@char >= 0x01 && @char <= 0x08)
477 || (@char >= 0x0e && @char <= 0x1f)
478 || @char == 0x27
479 || @char == 0x2d) {
480
481 special = 1;
482 }
483 }
484 }
