350 private void Export(string path)
351 {
352 StringBuilder stringBuilder = new StringBuilder();
353 for (int i = 0; i < todoList.list.Count; i++)
354 {
355 ToDoElement element = todoList.list[i];
356 string line = "";
357 if (element.completed) line += "[Complete]";
358 line += element.text.Replace("\n", @"\n");
359 stringBuilder.AppendLine(line);
360 }
361
362 File.WriteAllText(path, stringBuilder.ToString());
363 }

365 private void Import(string path)
366 {
367 string[] lines = File.ReadAllLines(path);
368
369 for (int i = 0; i < lines.Length; i++)
370 {
371 string line = lines[i];
372
373 bool complete = false;
374 string text = line;
375 if (line.StartsWith("[Complete]"))
376 {
377 text = line.Substring(10);
378 complete = true;
379 }
380
381 text = text.Replace(@"\n", "\n");
382
383 ToDoElement element = new ToDoElement() { text = text, completed = complete };
384 todoList.list.Add(element);
385 }
386 }
