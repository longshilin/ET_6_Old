66 public override object Read(object untyped, ProtoReader source)
67 {
68 TDictionary typed = AppendToCollection ? ((TDictionary)untyped) : null;
69 if (typed == null) typed = (TDictionary)Activator.CreateInstance(concreteType);
70
71 do
72 {
73 var key = DefaultKey;
74 var value = DefaultValue;
75 SubItemToken token = ProtoReader.StartSubItem(source);
76 int field;
77 while ((field = source.ReadFieldHeader()) > 0)
78 {
79 switch (field)
80 {
81 case 1:
82 key = (TKey)keyTail.Read(null, source);
83 break;
84 case 2:
85 value = (TValue)Tail.Read(Tail.RequiresOldValue ? (object)value : null, source);
86 break;
87 default:
88 source.SkipField();
89 break;
90 }
91 }
92
93 ProtoReader.EndSubItem(token, source);
94 typed[key] = value;
95 } while (source.TryReadFieldHeader(fieldNumber));
96
97 return typed;
98 }

175 protected override void EmitRead(CompilerContext ctx, Local valueFrom)
176 {
177 using (Compiler.Local list = AppendToCollection ? ctx.GetLocalWithValue(ExpectedType, valueFrom)
178 : new Compiler.Local(ctx, typeof(TDictionary)))

188 }

195 ctx.StoreValue(list);
196 ctx.MarkLabel(notNull);
197 }
198

200 ctx.MarkLabel(redoFromStart);
201
202 // key = default(TKey); value = default(TValue);
203 if (typeof(TKey) == typeof(string))
204 {
205 ctx.LoadValue("");
206 ctx.StoreValue(key);
207 }
208 else
209 {
210 ctx.InitLocal(typeof(TKey), key);
211 }
212 if (typeof(TValue) == typeof(string))
213 {
214 ctx.LoadValue("");
215 ctx.StoreValue(value);
216 }
217 else
218 {
219 ctx.InitLocal(typeof(TValue), @value);
220 }
221

223 ctx.LoadReaderWriter();

225 ctx.StoreValue(token);

236

238 ctx.MarkLabel(@default);
239 ctx.LoadReaderWriter();
