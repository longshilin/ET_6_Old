182 public void UpdatePosition( bool immediately )
183 {
184 Vector2 canvasSize = debugManager.canvasTR.rect.size;
185
186 float canvasWidth = canvasSize.x;
187 float canvasHeight = canvasSize.y;
188
189 // normalizedPosition allows us to glue the popup to a specific edge of the screen. It becomes useful when
190 // the popup is at the right edge and we switch from portrait screen orientation to landscape screen orientation.
191 // Without normalizedPosition, popup could jump to bottom or top edges instead of staying at the right edge
192 Vector2 pos = immediately ? new Vector2( normalizedPosition.x * canvasWidth, normalizedPosition.y * canvasHeight ) : popupTransform.anchoredPosition;
193
194 // Find distances to all four edges
195 float distToLeft = canvasWidth * 0.5f + pos.x;
196 float distToRight = canvasWidth - distToLeft;
197
198 float distToBottom = canvasHeight * 0.5f + pos.y;
199 float distToTop = canvasHeight - distToBottom;
200
201 float horDistance = Mathf.Min( distToLeft, distToRight );
202 float vertDistance = Mathf.Min( distToBottom, distToTop );
203
204 // Find the nearest edge's coordinates
205 if( horDistance < vertDistance )
206 {
207 if( distToLeft < distToRight )
208 pos = new Vector2( canvasWidth * -0.5f + halfSize.x, pos.y );
209 else
210 pos = new Vector2( canvasWidth * 0.5f - halfSize.x, pos.y );
211
212 pos.y = Mathf.Clamp( pos.y, canvasHeight * -0.5f + halfSize.y, canvasHeight * 0.5f - halfSize.y );
213 }
214 else
215 {
216 if( distToBottom < distToTop )
217 pos = new Vector2( pos.x, canvasHeight * -0.5f + halfSize.y );
218 else
219 pos = new Vector2( pos.x, canvasHeight * 0.5f - halfSize.y );
220
221 pos.x = Mathf.Clamp( pos.x, canvasWidth * -0.5f + halfSize.x, canvasWidth * 0.5f - halfSize.x );
222 }
223
224 normalizedPosition.Set( pos.x / canvasWidth, pos.y / canvasHeight );
225
226 // If another smooth movement animation is in progress, cancel it
227 if( moveToPosCoroutine != null )
228 {
229 StopCoroutine( moveToPosCoroutine );
230 moveToPosCoroutine = null;
231 }
232
233 if( immediately )
234 popupTransform.anchoredPosition = pos;
235 else
236 {
237 // Smoothly translate the popup to the specified position
238 moveToPosCoroutine = MoveToPosAnimation( pos );
239 StartCoroutine( moveToPosCoroutine );
240 }
241 }
