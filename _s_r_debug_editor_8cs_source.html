60 internal static IEnumerable<IntegrityIssue> QuickIntegrityCheck()
61 {
62 int enabledCount = 0;
63 int disabledCount = 0;
64
65 foreach (ResourceDirectory directory in GetResourcePaths())
66 {
67 if (directory.IsEnabled && directory.IsDisabled)
68 {
69 yield return new SomeResourcesAreEnabledAndDisabledIntegrityIssue();
70 yield break;
71 }
72
73 if (directory.IsEnabled) enabledCount++;
74 if (directory.IsDisabled) disabledCount++;
75 }
76
77 if (enabledCount > 0 && disabledCount > 0)
78 {
79#if DISABLE_SRDEBUGGER
80 yield return new SomeResourcesEnabledIntegrityIssue();
81#else
82 yield return new SomeResourcesDisabledIntegrityIssue();
83#endif
84 yield break; // Don't do any further resource-related checks.
85 }
86
87 if (!IsEnabled && enabledCount > 0)
88 {
89 yield return new ScriptsDisabledButResourcesEnabled();
90 }
91
92 if (IsEnabled && disabledCount > 0)
93 {
94 yield return new ScriptsEnabledButResourcesDisabled();
95 }
96 }

204 private new const string Title = "Duplicate SRDebugger resource directories";

206 private new const string Description =
207 "Some SRDebugger resource directories exist in both an enabled and disabled state. \n\n" +

209

211 {
212

214
215 protected override IEnumerable<Fix> CreateFixes()
216 {
217 if (!IsEnabled)
218 {
219 var deletePaths = GetResourcePaths().Where(p => p.IsDisabled && p.IsEnabled).ToList();
220 string paths = " - " + string.Join("\n - ", deletePaths
221 .SelectMany(p => new string[] { p.DisabledPath, p.DisabledPathBackupMetaFile }).ToArray());
222
223 yield return new DelegateFix(
224 "Keep enabled resources, disable SRDebugger",
225 "If you have just installed a new version of SRDebugger, this will keep the most up-to-date resources from the imported package. SRDebugger will be disabled after the old resources are deleted. \n\n The following paths will be deleted: \n\n" + paths,
226 () =>
227 {
228 foreach (ResourceDirectory rd in GetResourcePaths())
229 {
230 if (rd.IsEnabled && rd.IsDisabled)
231 {
232 Debug.Log("[SRDebugger] Delete Path: " + rd.DisabledPath);
233 Directory.Delete(rd.DisabledPath, true);
