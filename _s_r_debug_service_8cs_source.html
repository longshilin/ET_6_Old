283 private void PromptEntryCode()
284 {
285 SRServiceManager.GetService<IPinEntryService>()
286 .ShowPinEntry(Settings.Instance.EntryCode, SRDebugStrings.Current.PinEntryPrompt,
287 entered =>
288 {
289 if (entered)
290 {
291 if (!Settings.Instance.RequireEntryCodeEveryTime)
292 {
293 _hasAuthorised = true;
294 }
295
296 if (_queuedTab.HasValue)
297 {
298 var t = _queuedTab.Value;
299
300 _queuedTab = null;
301 ShowDebugPanel(t, false);
302 }
303 else
304 {
305 ShowDebugPanel(false);
306 }
307 }
308
309 _queuedTab = null;
310 });

313 public RectTransform EnableWorldSpaceMode()
314 {
315 if (_worldSpaceTransform != null)
316 {
317 return _worldSpaceTransform;
318 }
319
320 if (Settings.Instance.UseDebugCamera)
321 {
322 throw new InvalidOperationException("UseDebugCamera cannot be enabled at the same time as EnableWorldSpaceMode.");
323 }
324
325 _debugPanelService.IsVisible = true;
326
327 var root = ((DebugPanelServiceImpl) _debugPanelService).RootObject;
328 root.Canvas.gameObject.RemoveComponentIfExists<SRRetinaScaler>();
329 root.Canvas.gameObject.RemoveComponentIfExists<CanvasScaler>();
330 root.Canvas.renderMode = RenderMode.WorldSpace;
331
332 var rectTransform = root.Canvas.GetComponent<RectTransform>();
333 rectTransform.sizeDelta = new Vector2(1024, 768);
334 rectTransform.position = Vector3.zero;
