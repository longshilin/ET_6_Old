573 void InitializeFields()
574 {
575 fieldMapping = new Dictionary<string, int>();
576 fieldInfoCache = new Dictionary<int, FieldInfo>();
577
578 var fields = clrType.GetFields(BindingFlags.Public | BindingFlags.Instance | BindingFlags.NonPublic | BindingFlags.Static);
579 int idx = 0;
580 bool hasValueTypeBinder = ValueTypeBinder != null;
581 if (hasValueTypeBinder)
582 {
583 fieldIdxMapping = new Dictionary<int, int>();
584 }
585 if (hasValueTypeBinder || isEnum)
586 {
587 orderedFieldTypes = new IType[fields.Length];
588 }
589 foreach (var i in fields)
590 {
591 int hashCode = i.GetHashCode();
592
593 if (i.IsPublic || i.IsFamily || hasValueTypeBinder)
594 {
595 fieldMapping[i.Name] = hashCode;
596 fieldInfoCache[hashCode] = i;
597 }
598 if ((hasValueTypeBinder || isEnum) && !i.IsStatic)
599 {
600 orderedFieldTypes[idx] = appdomain.GetType(i.FieldType);
601 if (hasValueTypeBinder)
602 fieldIdxMapping[hashCode] = idx++;
603 else
604 idx++;
605 }
606
607 CLRFieldGetterDelegate getter;
608 if (AppDomain.FieldGetterMap.TryGetValue(i, out getter))
609 {
610 if (fieldGetterCache == null) fieldGetterCache = new Dictionary<int, CLRFieldGetterDelegate>();
611 fieldGetterCache[hashCode] = getter;
612 }
613
614 CLRFieldSetterDelegate setter;
615 if (AppDomain.FieldSetterMap.TryGetValue(i, out setter))
616 {
617 if (fieldSetterCache == null) fieldSetterCache = new Dictionary<int, CLRFieldSetterDelegate>();
618 fieldSetterCache[hashCode] = setter;
619 }
620
621 KeyValuePair<CLRFieldBindingDelegate, CLRFieldBindingDelegate> binding;
622 if(AppDomain.FieldBindingMap.TryGetValue(i, out binding))
623 {
624 if (fieldBindingCache == null) fieldBindingCache = new Dictionary<int, KeyValuePair<CLRFieldBindingDelegate, CLRFieldBindingDelegate>>();
625 fieldBindingCache[hashCode] = binding;
626 }
627 }
628 if (orderedFieldTypes != null)
629 {
630 Array.Resize(ref orderedFieldTypes, idx);
631 }

705 {

709 {

714 }
715 throw new NotSupportedException();

718 {

721 if (q.IsArray)

723 return MatchGenericParameters(args, type.GetElementType(), q.GetElementType(), genericArguments);
