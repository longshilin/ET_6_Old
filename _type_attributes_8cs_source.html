15 [Flags]
16 public enum TypeAttributes : uint {
17 // Visibility attributes
18 VisibilityMask = 0x00000007, // Use this mask to retrieve visibility information
19 NotPublic = 0x00000000, // Class has no public scope
20 Public = 0x00000001, // Class has public scope
21 NestedPublic = 0x00000002, // Class is nested with public visibility
22 NestedPrivate = 0x00000003, // Class is nested with private visibility
23 NestedFamily = 0x00000004, // Class is nested with family visibility
24 NestedAssembly = 0x00000005, // Class is nested with assembly visibility
25 NestedFamANDAssem = 0x00000006, // Class is nested with family and assembly visibility
26 NestedFamORAssem = 0x00000007, // Class is nested with family or assembly visibility
27
28 // Class layout attributes
29 LayoutMask = 0x00000018, // Use this mask to retrieve class layout information
30 AutoLayout = 0x00000000, // Class fields are auto-laid out
31 SequentialLayout = 0x00000008, // Class fields are laid out sequentially
32 ExplicitLayout = 0x00000010, // Layout is supplied explicitly
33
34 // Class semantics attributes
35 ClassSemanticMask = 0x00000020, // Use this mask to retrieve class semantics information
36 Class = 0x00000000, // Type is a class
37 Interface = 0x00000020, // Type is an interface
38
39 // Special semantics in addition to class semantics
40 Abstract = 0x00000080, // Class is abstract
41 Sealed = 0x00000100, // Class cannot be extended
42 SpecialName = 0x00000400, // Class name is special
43
44 // Implementation attributes
45 Import = 0x00001000, // Class/Interface is imported
46 Serializable = 0x00002000, // Class is serializable
47 WindowsRuntime = 0x00004000, // Windows Runtime type
48
49 // String formatting attributes
50 StringFormatMask = 0x00030000, // Use this mask to retrieve string information for native interop
51 AnsiClass = 0x00000000, // LPSTR is interpreted as ANSI
52 UnicodeClass = 0x00010000, // LPSTR is interpreted as Unicode
53 AutoClass = 0x00020000, // LPSTR is interpreted automatically
54
55 // Class initialization attributes
56 BeforeFieldInit = 0x00100000, // Initialize the class before first static field access
57
58 // Additional flags
59 RTSpecialName = 0x00000800, // CLI provides 'special' behavior, depending upon the name of the Type
60 HasSecurity = 0x00040000, // Type has security associate with it
61 Forwarder = 0x00200000, // Exported type is a type forwarder
62 }
