156 private static object ParseDefaultValue(Type type, object value)
157 {
158 if(true)
159 {
160 Type tmp = Helpers.GetUnderlyingType(type);
161 if (tmp != null) type = tmp;
162 }
163 switch (Helpers.GetTypeCode(type))
164 {
165 case ProtoTypeCode.Boolean:
166 case ProtoTypeCode.Byte:
167 case ProtoTypeCode.Char: // char.Parse missing on CF/phone7
168 case ProtoTypeCode.DateTime:
169 case ProtoTypeCode.Decimal:
170 case ProtoTypeCode.Double:
171 case ProtoTypeCode.Int16:
172 case ProtoTypeCode.Int32:
173 case ProtoTypeCode.Int64:
174 case ProtoTypeCode.SByte:
175 case ProtoTypeCode.Single:
176 case ProtoTypeCode.String:
177 case ProtoTypeCode.UInt16:
178 case ProtoTypeCode.UInt32:
179 case ProtoTypeCode.UInt64:
180 case ProtoTypeCode.TimeSpan:
181 case ProtoTypeCode.Uri:
182 case ProtoTypeCode.Guid:
183 {
184 value = value + "";
185 }
186 break;
187 }
188 if (value is string)
189 {
190 string s = (string)value;
191 if (Helpers.IsEnum(type)) return Helpers.ParseEnum(type, s);
192
193 switch (Helpers.GetTypeCode(type))
194 {
195 case ProtoTypeCode.Boolean: return bool.Parse(s);
196 case ProtoTypeCode.Byte: return byte.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);
197 case ProtoTypeCode.Char: // char.Parse missing on CF/phone7
198 if (s.Length == 1) return s[0];
199 throw new FormatException("Single character expected: \"" + s + "\"");
200 case ProtoTypeCode.DateTime: return DateTime.Parse(s, CultureInfo.InvariantCulture);
201 case ProtoTypeCode.Decimal: return decimal.Parse(s, NumberStyles.Any, CultureInfo.InvariantCulture);
202 case ProtoTypeCode.Double: return double.Parse(s, NumberStyles.Any, CultureInfo.InvariantCulture);
203 case ProtoTypeCode.Int16: return short.Parse(s, NumberStyles.Any, CultureInfo.InvariantCulture);
204 case ProtoTypeCode.Int32: return int.Parse(s, NumberStyles.Any, CultureInfo.InvariantCulture);
205 case ProtoTypeCode.Int64: return long.Parse(s, NumberStyles.Any, CultureInfo.InvariantCulture);
206 case ProtoTypeCode.SByte: return sbyte.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);
207 case ProtoTypeCode.Single: return float.Parse(s, NumberStyles.Any, CultureInfo.InvariantCulture);
208 case ProtoTypeCode.String: return s;
209 case ProtoTypeCode.UInt16: return ushort.Parse(s, NumberStyles.Any, CultureInfo.InvariantCulture);
210 case ProtoTypeCode.UInt32: return uint.Parse(s, NumberStyles.Any, CultureInfo.InvariantCulture);
211 case ProtoTypeCode.UInt64: return ulong.Parse(s, NumberStyles.Any, CultureInfo.InvariantCulture);
212 case ProtoTypeCode.TimeSpan: return TimeSpan.Parse(s);
213 case ProtoTypeCode.Uri: return s; // Uri is decorated as string
214 case ProtoTypeCode.Guid: return new Guid(s);
215 }
216 }
217#if FEAT_IKVM
218 if (Helpers.IsEnum(type)) return value; // return the underlying type instead
219 System.Type convertType = null;
220 switch(Helpers.GetTypeCode(type))
221 {
222 case ProtoTypeCode.SByte: convertType = typeof(sbyte); break;
223 case ProtoTypeCode.Int16: convertType = typeof(short); break;
224 case ProtoTypeCode.Int32: convertType = typeof(int); break;
225 case ProtoTypeCode.Int64: convertType = typeof(long); break;
226 case ProtoTypeCode.Byte: convertType = typeof(byte); break;
227 case ProtoTypeCode.UInt16: convertType = typeof(ushort); break;
228 case ProtoTypeCode.UInt32: convertType = typeof(uint); break;
229 case ProtoTypeCode.UInt64: convertType = typeof(ulong); break;
230 case ProtoTypeCode.Single: convertType = typeof(float); break;
231 case ProtoTypeCode.Double: convertType = typeof(double); break;
232 case ProtoTypeCode.Decimal: convertType = typeof(decimal); break;
233 }
234 if(convertType != null) return Convert.ChangeType(value, convertType, CultureInfo.InvariantCulture);
235 throw new ArgumentException("Unable to process default value: " + value + ", " + type.FullName);
236#else
237 if (Helpers.IsEnum(type))
238 {
239 if (value is ILRuntime.Mono.Cecil.CustomAttributeArgument)
240 {
241 var v = (ILRuntime.Mono.Cecil.CustomAttributeArgument)value;
242 return v.Value;
243 }
244
245 return Enum.ToObject(type, value);
246 }
247 return Convert.ChangeType(value, type, CultureInfo.InvariantCulture);
248#endif

388 {
389 if (this.getSpecified != getSpecified || this.setSpecified != setSpecified)
390 {
391 if (getSpecified != null)
392 {
393 if (getSpecified.ReturnType != model.MapType(typeof(bool))
394 || getSpecified.IsStatic
395 || getSpecified.GetParameters().Length != 0)
396 {
397 throw new ArgumentException("Invalid pattern for checking member-specified", "getSpecified");
398 }
399 }
400 if (setSpecified != null)
401 {
402 ParameterInfo[] args;
403 if (setSpecified.ReturnType != model.MapType(typeof(void))
404 || setSpecified.IsStatic
405 || (args = setSpecified.GetParameters()).Length != 1
406 || args[0].ParameterType != model.MapType(typeof(bool)))
407 {
408 throw new ArgumentException("Invalid pattern for setting member-specified", "setSpecified");
409 }
410 }
411
412 ThrowIfFrozen();
413 this.getSpecified = getSpecified;
414 this.setSpecified = setSpecified;

421 internal bool ResolveMapTypes(out Type dictionaryType, out Type keyType, out Type valueType)
422 {
423 dictionaryType = keyType = valueType = null;
424
425#if WINRT || COREFX
426 var info = memberType.GetTypeInfo();
427#else
428 var info = memberType;
429#endif
430 MethodInfo b, a, ar, f;
431 if(ImmutableCollectionDecorator.IdentifyImmutable(model, MemberType, out b, out a, out ar, out f))
432 {
433 return false;
434 }
435 if (info.IsInterface && info.IsGenericType && info.GetGenericTypeDefinition() == typeof(IDictionary<,>))
436 {
437 var typeArgs = memberType.GetGenericArguments();
438
439 if (IsValidMapKeyType(typeArgs[0]))
440 {
441 keyType = typeArgs[0];
442 valueType = typeArgs[1];
443 dictionaryType = memberType;
444 }
445 return false;
446 }
447
448 foreach (var iType in memberType.GetInterfaces())
449 {
450#if WINRT || COREFX
451 info = iType.GetTypeInfo();
452#else
453 info = iType;
454#endif
455 if (info.IsGenericType && info.GetGenericTypeDefinition() == typeof(IDictionary<,>))
456 {
457 if (dictionaryType != null) throw new InvalidOperationException("Multiple dictionary interfaces implemented by type: " + memberType.FullName);
458 var typeArgs = iType.GetGenericArguments();
459
460 if (IsValidMapKeyType(typeArgs[0]))
461 {
462 keyType = typeArgs[0];
463 valueType = typeArgs[1];
464 dictionaryType = memberType;
465 }
466 }
467 }
468 if (dictionaryType == null) return false;
469
470 // (note we checked the key type already)
471 // not a map if value is repeated
472 Type itemType = null, defaultType = null;
473 model.ResolveListTypes(valueType, ref itemType, ref defaultType);
474 if (itemType != null) return false;
475
476 return dictionaryType != null;
477 }

479 static bool IsValidMapKeyType(Type type)
480 {
481 if (type == null) return false;
482 switch(Helpers.GetTypeCode(type))
483 {
484 case ProtoTypeCode.Boolean:
485 case ProtoTypeCode.Byte:
486 case ProtoTypeCode.Char:
487 case ProtoTypeCode.Int16:
488 case ProtoTypeCode.Int32:
489 case ProtoTypeCode.Int64:
490 case ProtoTypeCode.String:
491
492 case ProtoTypeCode.SByte:
493 case ProtoTypeCode.UInt16:
494 case ProtoTypeCode.UInt32:
495 case ProtoTypeCode.UInt64:
496 return true;
497 }
498 return false;

500 private IProtoSerializer BuildSerializer()
501 {
502 int opaqueToken = 0;
503 try
504 {
505 model.TakeLock(ref opaqueToken);// check nobody is still adding this type
506 var member = backingMember ?? originalMember;
507 IProtoSerializer ser;
508 if (IsMap)
509 {
510 Type dictionaryType,keyType,valueType;
511 ResolveMapTypes(out dictionaryType, out keyType, out valueType);
512
513 if (dictionaryType == null)
514 {
515 throw new InvalidOperationException("Unable to resolve map type for type: " + memberType.FullName);
516 }
517 var concreteType = defaultType;
518 if(concreteType == null && Helpers.IsClass(memberType))
519 {
520 concreteType = memberType;
521 }
522 WireType keyWireType;
523 var keySer = TryGetCoreSerializer(model, MapKeyFormat, keyType, out keyWireType, false, false, false, false);
524 if(!AsReference)
525 {
526 AsReference = MetaType.GetAsReferenceDefault(model, valueType);
527 }
528 WireType valueWireType;
529 var valueSer = TryGetCoreSerializer(model, MapValueFormat, valueType, out valueWireType, AsReference, DynamicType, false, true);
530
531 var ctors = typeof(MapDecorator<,,>).MakeGenericType(new Type[] { dictionaryType, keyType, valueType }).GetConstructors(
532 BindingFlags.NonPublic | BindingFlags.Public | BindingFlags.Instance);
533 if (ctors.Length != 1) throw new InvalidOperationException("Unable to resolve MapDecorator constructor");
534 ser = (IProtoSerializer)ctors[0].Invoke(new object[] {model, concreteType, keySer, valueSer, fieldNumber,
535 DataFormat == DataFormat.Group ? WireType.StartGroup : WireType.String, keyWireType, valueWireType, OverwriteList });
536 }
537 else
538 {
539 WireType wireType;
540 Type finalType = itemType == null ? memberType : itemType;
541 ser = TryGetCoreSerializer(model, dataFormat, finalType, out wireType, AsReference, DynamicType, OverwriteList, true);
542 if (ser == null)
543 {
544 throw new InvalidOperationException("No serializer defined for type: " + finalType.FullName);
545 }
546
547 // apply tags
548 if (itemType != null && SupportNull)
549 {
550 if (IsPacked)
551 {
552 throw new NotSupportedException("Packed encodings cannot support null values");
553 }
554 ser = new TagDecorator(NullDecorator.Tag, wireType, IsStrict, ser);
555 ser = new NullDecorator(model, ser);
556 ser = new TagDecorator(fieldNumber, WireType.StartGroup, false, ser);
557 }
558 else
559 {
560 ser = new TagDecorator(fieldNumber, wireType, IsStrict, ser);
561 }
562 // apply lists if appropriate
563 if (itemType != null)
564 {
565#if NO_GENERICS
566 Type underlyingItemType = itemType;
567#else
568 Type underlyingItemType = SupportNull ? itemType : Helpers.GetUnderlyingType(itemType) ?? itemType;
569#endif
570 Helpers.DebugAssert(underlyingItemType == ser.ExpectedType
571 || (ser.ExpectedType == model.MapType(typeof(object)) && !Helpers.IsValueType(underlyingItemType))
572 , "Wrong type in the tail; expected {0}, received {1}", ser.ExpectedType, underlyingItemType);
573 if (memberType.IsArray)
574 {
575 ser = new ArrayDecorator(model, ser, fieldNumber, IsPacked, wireType, memberType, OverwriteList, SupportNull);
576 }
577 else
578 {
579 ser = ListDecorator.Create(model, memberType, defaultType, ser, fieldNumber, IsPacked, wireType, member != null && PropertyDecorator.CanWrite(model, member), OverwriteList, SupportNull);
580 }
581 }
582 else if (defaultValue != null && !IsRequired && getSpecified == null)
583 { // note: "ShouldSerialize*" / "*Specified" / etc ^^^^ take precedence over defaultValue,
584 // as does "IsRequired"
585 ser = new DefaultValueDecorator(model, defaultValue, ser);
586 }
587 if (memberType == model.MapType(typeof(Uri)))
588 {
589 ser = new UriDecorator(model, ser);
590 }
591#if PORTABLE
592 else if(memberType.FullName == typeof(Uri).FullName)
593 {
594 // In PCLs, the Uri type may not match (WinRT uses Internal/Uri, .Net uses System/Uri)
595 ser = new ReflectedUriDecorator(memberType, model, ser);
596 }
597#endif
598 }
599 if (member != null)
600 {
601 PropertyInfo prop = member as PropertyInfo;
602 if (prop != null)
603 {
604 ser = new PropertyDecorator(model, parentType, (PropertyInfo)member, ser);
605 }
606 else
607 {
608 FieldInfo fld = member as FieldInfo;
609 if (fld != null)
610 {
611 ser = new FieldDecorator(parentType, (FieldInfo)member, ser);
612 }
613 else
614 {
615 throw new InvalidOperationException();
616 }
617 }
618 if (getSpecified != null || setSpecified != null)
619 {
620 ser = new MemberSpecifiedDecorator(getSpecified, setSpecified, ser);
621 }
622 }
623 return ser;
624 }
625 finally
626 {
627 model.ReleaseLock(opaqueToken);
628 }
629 }

654 internal static IProtoSerializer TryGetCoreSerializer(RuntimeTypeModel model, DataFormat dataFormat, Type type, out WireType defaultWireType,
655 bool asReference, bool dynamicType, bool overwriteList, bool allowComplexTypes)
656 {
657#if !NO_GENERICS
658 {
659 Type tmp = Helpers.GetUnderlyingType(type);
660 if (tmp != null) type = tmp;
661 }
662#endif
663 if (Helpers.IsEnum(type))
664 {
665 if (allowComplexTypes && model != null)
666 {
667 // need to do this before checking the typecode; an int enum will report Int32 etc
668 defaultWireType = WireType.Variant;
669 return new EnumSerializer(type, model.GetEnumMap(type));
670 }
671 else
672 { // enum is fine for adding as a meta-type
673 defaultWireType = WireType.None;
674 return null;
675 }
676 }
677 ProtoTypeCode code = Helpers.GetTypeCode(type);
678 switch (code)
679 {
680 case ProtoTypeCode.Int32:
681 defaultWireType = GetIntWireType(dataFormat, 32);
682 return new Int32Serializer(model);
683 case ProtoTypeCode.UInt32:
684 defaultWireType = GetIntWireType(dataFormat, 32);
685 return new UInt32Serializer(model);
686 case ProtoTypeCode.Int64:
687 defaultWireType = GetIntWireType(dataFormat, 64);
688 return new Int64Serializer(model);
689 case ProtoTypeCode.UInt64:
690 defaultWireType = GetIntWireType(dataFormat, 64);
691 return new UInt64Serializer(model);
692 case ProtoTypeCode.String:
693 defaultWireType = WireType.String;
694 if (asReference)
695 {
696 return new NetObjectSerializer(model, model.MapType(typeof(string)), 0, BclHelpers.NetObjectOptions.AsReference);
697 }
698 return new StringSerializer(model);
699 case ProtoTypeCode.Single:
700 defaultWireType = WireType.Fixed32;
701 return new SingleSerializer(model);
702 case ProtoTypeCode.Double:
703 defaultWireType = WireType.Fixed64;
704 return new DoubleSerializer(model);
705 case ProtoTypeCode.Boolean:
706 defaultWireType = WireType.Variant;
707 return new BooleanSerializer(model);
708 case ProtoTypeCode.DateTime:
709 defaultWireType = GetDateTimeWireType(dataFormat);
710 return new DateTimeSerializer(dataFormat, model);
711 case ProtoTypeCode.Decimal:
712 defaultWireType = WireType.String;
713 return new DecimalSerializer(model);
714 case ProtoTypeCode.Byte:
715 defaultWireType = GetIntWireType(dataFormat, 32);
716 return new ByteSerializer(model);
717 case ProtoTypeCode.SByte:
718 defaultWireType = GetIntWireType(dataFormat, 32);
719 return new SByteSerializer(model);
720 case ProtoTypeCode.Char:
721 defaultWireType = WireType.Variant;
722 return new CharSerializer(model);
723 case ProtoTypeCode.Int16:
724 defaultWireType = GetIntWireType(dataFormat, 32);
725 return new Int16Serializer(model);
726 case ProtoTypeCode.UInt16:
727 defaultWireType = GetIntWireType(dataFormat, 32);
728 return new UInt16Serializer(model);
729 case ProtoTypeCode.TimeSpan:
730 defaultWireType = GetDateTimeWireType(dataFormat);
731 return new TimeSpanSerializer(dataFormat, model);
732 case ProtoTypeCode.Guid:
733 defaultWireType = dataFormat == DataFormat.Group ? WireType.StartGroup : WireType.String;
734 return new GuidSerializer(model);
735 case ProtoTypeCode.Uri:
736 defaultWireType = WireType.String;
737 return new StringSerializer(model);
738 case ProtoTypeCode.ByteArray:
739 defaultWireType = WireType.String;
740 return new BlobSerializer(model, overwriteList);
741 case ProtoTypeCode.Type:
742 defaultWireType = WireType.String;
743 return new SystemTypeSerializer(model);
744 }
745 IProtoSerializer parseable = model.AllowParseableTypes ? ParseableSerializer.TryCreate(type, model) : null;
746 if (parseable != null)
747 {
748 defaultWireType = WireType.String;
749 return parseable;
750 }
751 if (allowComplexTypes && model != null)
752 {
753 int key = model.GetKey(type, false, true);
754 MetaType meta = null;
755 if (key >= 0)
756 {
757 meta = model[type];
758 if(dataFormat == DataFormat.Default && meta.IsGroup)
759 {
760 dataFormat = DataFormat.Group;
761 }
762 }
763
764 if (asReference || dynamicType)
765 {
766 BclHelpers.NetObjectOptions options = BclHelpers.NetObjectOptions.None;
767 if (asReference) options |= BclHelpers.NetObjectOptions.AsReference;
768 if (dynamicType) options |= BclHelpers.NetObjectOptions.DynamicType;
769 if (meta != null)
770 { // exists
771 if (asReference && Helpers.IsValueType(type))
772 {
773 string message = "AsReference cannot be used with value-types";
774
775 if (type.Name == "KeyValuePair`2")
776 {
777 message += "; please see http://stackoverflow.com/q/14436606/";
778 }
779 else
780 {
781 message += ": " + type.FullName;
782 }
783 throw new InvalidOperationException(message);
784 }
785
786 if (asReference && meta.IsAutoTuple) options |= BclHelpers.NetObjectOptions.LateSet;
787 if (meta.UseConstructor) options |= BclHelpers.NetObjectOptions.UseConstructor;
788 }
789 defaultWireType = dataFormat == DataFormat.Group ? WireType.StartGroup : WireType.String;
790 return new NetObjectSerializer(model, type, key, options);
791 }
792 if (key >= 0)
793 {
794 defaultWireType = dataFormat == DataFormat.Group ? WireType.StartGroup : WireType.String;
795 return new SubItemSerializer(type, key, meta, true);
796 }
797 }
798 defaultWireType = WireType.None;
799 return null;
800 }
