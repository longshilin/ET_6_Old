271 protected void Populate()
272 {
273 var sortedOptions = new Dictionary<string, List<OptionDefinition>>();
274
275 foreach (var option in Service.Options.Options)
276 {
277 // Find a properly list for that category, or create a new one
278 List<OptionDefinition> memberList;
279
280 if (!sortedOptions.TryGetValue(option.Category, out memberList))
281 {
282 memberList = new List<OptionDefinition>();
283 sortedOptions.Add(option.Category, memberList);
284 }
285
286 memberList.Add(option);
287 }
288
289 var hasCreated = false;
290
291 foreach (var kv in sortedOptions)
292 {
293 if (kv.Value.Count == 0)
294 {
295 continue;
296 }
297
298 hasCreated = true;
299 CreateCategory(kv.Key, kv.Value);
300 }
301
302 if (hasCreated)
303 {
304 NoOptionsNotice.SetActive(false);
305 }

308 protected void CreateCategory(string title, List<OptionDefinition> options)
309 {
310 options.Sort((d1, d2) => d1.SortPriority.CompareTo(d2.SortPriority));
311
312 var groupInstance = SRInstantiate.Instantiate(CategoryGroupPrefab);
313 var categoryInstance = new CategoryInstance(groupInstance);
314
315 _categories.Add(categoryInstance);
316
317 groupInstance.CachedTransform.SetParent(ContentContainer, false);
318 groupInstance.Header.text = title;
319 groupInstance.SelectionModeEnabled = false;
320
321 categoryInstance.CategoryGroup.SelectionToggle.onValueChanged.AddListener(
322 b => OnCategorySelectionToggle(categoryInstance, b));
323
324 foreach (var option in options)
325 {
326 var control = OptionControlFactory.CreateControl(option, title);
327
328 if (control == null)
329 {
330 Debug.LogError("[SRDebugger.OptionsTab] Failed to create option control for {0}".Fmt(option.Name));
331 continue;
332 }
333
334 categoryInstance.Options.Add(control);
335 control.CachedTransform.SetParent(groupInstance.Container, false);
336 control.IsSelected = Service.PinnedUI.HasPinned(option);
337 control.SelectionModeEnabled = false;
338 control.SelectionModeToggle.onValueChanged.AddListener(OnOptionSelectionToggle);
339
340 _options.Add(option, control);
341 _controls.Add(control);
342 }
