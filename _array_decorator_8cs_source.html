117 private void EmitWriteArrayLoop(Compiler.CompilerContext ctx, Compiler.Local i, Compiler.Local arr)
118 {
119 // i = 0
120 ctx.LoadValue(0);
121 ctx.StoreValue(i);
122
123 // range test is last (to minimise branches)
124 Compiler.CodeLabel loopTest = ctx.DefineLabel(), processItem = ctx.DefineLabel();
125 ctx.Branch(loopTest, false);
126 ctx.MarkLabel(processItem);
127
128 // {...}
129 ctx.LoadArrayValue(arr, i);
130 if (SupportNull)
131 {
132 Tail.EmitWrite(ctx, null);
133 }
134 else
135 {
136 ctx.WriteNullCheckedTail(itemType, Tail, null);
137 }
138
139 // i++
140 ctx.LoadValue(i);
141 ctx.LoadValue(1);
142 ctx.Add();
143 ctx.StoreValue(i);
144
145 // i < arr.Length
146 ctx.MarkLabel(loopTest);
147 ctx.LoadValue(i);
148 ctx.LoadLength(arr, false);
149 ctx.BranchIfLess(processItem, false);
150 }

156 internal static bool CanUsePackedPrefix(WireType packedWireType, Type itemType)
157 {
158 // needs to be a suitably simple type *and* be definitely not nullable
159 switch(packedWireType)
160 {
161 case WireType.Fixed32:
162 case WireType.Fixed64:
163 break;
164 default:
165 return false; // nope
166 }
167 if (!Helpers.IsValueType(itemType)) return false;
168 return Helpers.GetUnderlyingType(itemType) == null;
169 }

178 public override void Write(object value, ProtoWriter dest)
179 {
180 IList arr = (IList)value;
181 int len = arr.Count;
182 SubItemToken token;
183 bool writePacked = (options & OPTIONS_WritePacked) != 0;
184 bool fixedLengthPacked = writePacked && CanUsePackedPrefix();
185
186 if (writePacked)
187 {
188 ProtoWriter.WriteFieldHeader(fieldNumber, WireType.String, dest);
189
190 if (fixedLengthPacked)
191 {
192 ProtoWriter.WritePackedPrefix(arr.Count, packedWireType, dest);
193 token = new SubItemToken(); // default
194 }
195 else
196 {
197 token = ProtoWriter.StartSubItem(value, dest);
198 }
199 ProtoWriter.SetPackedField(fieldNumber, dest);
200 }
201 else
202 {
203 token = new SubItemToken(); // default
204 }
205 bool checkForNull = !SupportNull;
206 for (int i = 0; i < len; i++)
207 {
208 object obj = arr[i];
209 if (checkForNull && obj == null) { throw new NullReferenceException(); }
210 Tail.Write(obj, dest);
211 }
212 if (writePacked)
213 {
214 if (fixedLengthPacked)
215 {
216 ProtoWriter.ClearPackedField(fieldNumber, dest);
217 }
218 else
219 {
220 ProtoWriter.EndSubItem(token, dest);
221 }
222 }

224 public override object Read(object value, ProtoReader source)
225 {
226 int field = source.FieldNumber;
227 BasicList list = new BasicList();
228 if (packedWireType != WireType.None && source.WireType == WireType.String)
229 {
230 SubItemToken token = ProtoReader.StartSubItem(source);
231 while (ProtoReader.HasSubValue(packedWireType, source))
232 {
233 list.Add(Tail.Read(null, source));
234 }
235 ProtoReader.EndSubItem(token, source);
236 }
237 else
238 {
239 do
240 {
241 list.Add(Tail.Read(null, source));
242 } while (source.TryReadFieldHeader(field));
243 }
244 int oldLen = AppendToCollection ? ((value == null ? 0 : ((Array)value).Length)) : 0;
245 Array result = Array.CreateInstance(itemType, oldLen + list.Count);
246 if (oldLen != 0) ((Array)value).CopyTo(result, 0);
247 list.CopyTo(result, oldLen);
248 return result;
249 }
