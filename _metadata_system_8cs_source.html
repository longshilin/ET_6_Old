97 public static void TryProcessPrimitiveTypeReference (TypeReference type)
98 {
99 if (type.Namespace != "System")
100 return;
101
102 var scope = type.scope;
103 if (scope == null || scope.MetadataScopeType != MetadataScopeType.AssemblyNameReference)
104 return;
105
106 Row<ElementType, bool> primitive_data;
107 if (!TryGetPrimitiveData (type, out primitive_data))
108 return;
109
110 type.etype = primitive_data.Col1;
111 type.IsValueType = primitive_data.Col2;
112 }
113
114 public static bool TryGetPrimitiveElementType (TypeDefinition type, out ElementType etype)
115 {
116 etype = ElementType.None;
117
118 if (type.Namespace != "System")
119 return false;
120
121 Row<ElementType, bool> primitive_data;
122 if (TryGetPrimitiveData (type, out primitive_data)) {
123 etype = primitive_data.Col1;
124 return true;
125 }
126
127 return false;
128 }

413 static TypeDefinition BinaryRangeSearch (TypeDefinition [] types, uint rid, bool field)
414 {
415 int min = 0;
416 int max = types.Length - 1;
417 while (min <= max) {
418 int mid = min + ((max - min) / 2);
419 var type = types [mid];
420 var range = field ? type.fields_range : type.methods_range;
421
422 if (rid < range.Start)
423 max = mid - 1;
424 else if (rid >= range.Start + range.Length)
425 min = mid + 1;
426 else
427 return type;
428 }
429
430 return null;
431 }
