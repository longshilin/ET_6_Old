148 public static float SpringLerp(float strength, float deltaTime)
149 {
150 var ms = Mathf.RoundToInt(deltaTime*1000f);
151 var step = 0.001f*strength;
152
153 var from = 0f;
154 var to = 1f;
155
156 for (var i = 0; i < ms; i++)
157 {
158 from = Mathf.Lerp(from, to, step);
159 }
160
161 return from;
162 }
