716 public SequencePoint GetSequencePoint (Instruction instruction)
717 {
718 if (!HasSequencePoints)
719 return null;
720
721 for (int i = 0; i < sequence_points.Count; i++)
722 if (sequence_points [i].Offset == instruction.Offset)
723 return sequence_points [i];
724
725 return null;
726 }
727
728 public IDictionary<Instruction, SequencePoint> GetSequencePointMapping ()
729 {
730 var instruction_mapping = new Dictionary<Instruction, SequencePoint> ();
731 if (!HasSequencePoints || !method.HasBody)
732 return instruction_mapping;
733
734 var offset_mapping = new Dictionary<int, SequencePoint> (sequence_points.Count);
735
736 for (int i = 0; i < sequence_points.Count; i++) {
737 if (!offset_mapping.ContainsKey (sequence_points [i].Offset))
738 offset_mapping.Add (sequence_points [i].Offset, sequence_points [i]);
739 }
740
741 var instructions = method.Body.Instructions;
742
743 for (int i = 0; i < instructions.Count; i++) {
744 SequencePoint sequence_point;
745 if (offset_mapping.TryGetValue (instructions [i].Offset, out sequence_point))
746 instruction_mapping.Add (instructions [i], sequence_point);
747 }
748
749 return instruction_mapping;
750 }

752 public IEnumerable<ScopeDebugInformation> GetScopes ()
753 {
754 if (scope == null)
755 return Empty<ScopeDebugInformation>.Array;
756
757 return GetScopes (new[] { scope });

760 static IEnumerable<ScopeDebugInformation> GetScopes (IList<ScopeDebugInformation> scopes)
761 {
762 for (int i = 0; i < scopes.Count; i++) {
763 var scope = scopes [i];
764
765 yield return scope;
766
767 if (!scope.HasScopes)
768 continue;
769
770 foreach (var sub_scope in GetScopes (scope.Scopes))
771 yield return sub_scope;
772 }

775 public bool TryGetName (VariableDefinition variable, out string name)
776 {
777 name = null;
778
779 var has_name = false;
780 var unique_name = "";
781
782 foreach (var scope in GetScopes ()) {
783 string slot_name;
784 if (!scope.TryGetName (variable, out slot_name))
785 continue;
786
787 if (!has_name) {
788 has_name = true;
789 unique_name = slot_name;
790 continue;
791 }
792
793 if (unique_name != slot_name)
794 return false;
795 }
796
797 name = unique_name;
798 return has_name;
799 }

866 public ISymbolReader GetSymbolReader (ModuleDefinition module, string fileName)
867 {
868 if (module.Image.HasDebugTables ())
869 return null;
870
871 if (module.HasDebugHeader) {
872 var header = module.GetDebugHeader ();
873 var entry = header.GetEmbeddedPortablePdbEntry ();
874 if (entry != null)
875 return new EmbeddedPortablePdbReaderProvider ().GetSymbolReader (module, fileName);
876 }
877
878 var pdb_file_name = Mixin.GetPdbFileName (fileName);
879
880 if (File.Exists (pdb_file_name)) {
881 if (Mixin.IsPortablePdb (Mixin.GetPdbFileName (fileName)))
882 return new PortablePdbReaderProvider ().GetSymbolReader (module, fileName);
883
884 try {
885 return SymbolProvider.GetReaderProvider (SymbolKind.NativePdb).GetSymbolReader (module, fileName);
886 } catch (Exception) {
887 // We might not include support for native pdbs.
888 }
889 }
890
891 var mdb_file_name = Mixin.GetMdbFileName (fileName);
892 if (File.Exists (mdb_file_name)) {
893 try {
894 return SymbolProvider.GetReaderProvider (SymbolKind.Mdb).GetSymbolReader (module, fileName);
895 } catch (Exception) {
896 // We might not include support for mdbs.
897 }
898 }
899
900 if (throw_if_no_symbol)
901 throw new SymbolsNotFoundException (string.Format ("No symbol found for file: {0}", fileName));
902
903 return null;
904 }

906 public ISymbolReader GetSymbolReader (ModuleDefinition module, Stream symbolStream)
907 {
908 if (module.Image.HasDebugTables ())
909 return null;
910
911 if (module.HasDebugHeader) {
912 var header = module.GetDebugHeader ();
913 var entry = header.GetEmbeddedPortablePdbEntry ();
914 if (entry != null)
915 return new EmbeddedPortablePdbReaderProvider ().GetSymbolReader (module, "");
916 }
917
918 Mixin.CheckStream (symbolStream);
919 Mixin.CheckReadSeek (symbolStream);
920
921 var position = symbolStream.Position;
922
923 const int portablePdbHeader = 0x424a5342;
924
925 var reader = new BinaryStreamReader (symbolStream);
926 var intHeader = reader.ReadInt32 ();
927 symbolStream.Position = position;
928
929 if (intHeader == portablePdbHeader) {
930 return new PortablePdbReaderProvider ().GetSymbolReader (module, symbolStream);
931 }
932
933 const string nativePdbHeader = "Microsoft C/C++ MSF 7.00";
934
935 var bytesHeader = reader.ReadBytes (nativePdbHeader.Length);
936 symbolStream.Position = position;
937 var isNativePdb = true;
938
939 for (var i = 0; i < bytesHeader.Length; i++) {
940 if (bytesHeader [i] != (byte) nativePdbHeader [i]) {
941 isNativePdb = false;
942 break;
943 }
944 }
945
946 if (isNativePdb) {
947 try {
948 return SymbolProvider.GetReaderProvider (SymbolKind.NativePdb).GetSymbolReader (module, symbolStream);
949 } catch (Exception) {
950 // We might not include support for native pdbs.
951 }
952 }
953
954 const long mdbHeader = 0x45e82623fd7fa614;
955
956 var longHeader = reader.ReadInt64 ();
957 symbolStream.Position = position;
958
959 if (longHeader == mdbHeader) {
960 try {
961 return SymbolProvider.GetReaderProvider (SymbolKind.Mdb).GetSymbolReader (module, symbolStream);
962 } catch (Exception) {
963 // We might not include support for mdbs.
964 }
965 }
966
967 if (throw_if_no_symbol)
968 throw new SymbolsNotFoundException (string.Format ("No symbols found in stream"));
969
970 return null;
971 }
972 }
