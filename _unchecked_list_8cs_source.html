383 // Ensures that the capacity of this list is at least the given minimum
384 // value. If the current capacity of the list is less than min, the
385 // capacity is increased to twice the current capacity or to min,
386 // whichever is larger.
387 private void EnsureCapacity(int min)
388 {
389 if (_items.Length < min)
390 {
391 int newCapacity = _items.Length == 0 ? _defaultCapacity : _items.Length * 2;
392 // Allow the list to grow to maximum possible capacity (~2G elements) before encountering overflow.
393 // Note that this check works even when _items.Length overflowed thanks to the (uint) cast
394 if ((uint) newCapacity > Int32.MaxValue) newCapacity = Int32.MaxValue;
395 if (newCapacity < min) newCapacity = min;
396 Capacity = newCapacity;
397 }
398 }

605 // Inserts the elements of the given collection at a given index. If
606 // required, the capacity of the list is increased to twice the previous
607 // capacity or the new size, whichever is larger. Ranges may be added
608 // to the end of the list by setting index to the UncheckedList's size.
609 //
610 public void InsertRange(int index, IEnumerable<T> collection)
611 {
612 ICollection<T> c = collection as ICollection<T>;
613 if (c != null)
614 { // if collection is ICollection<T>
615 int count = c.Count;
616 if (count > 0)
617 {
618 EnsureCapacity(_size + count);
619 if (index < _size)
620 {
621 Array.Copy(_items, index, _items, index + count, _size - index);
622 }
623
624 // If we're inserting a UncheckedList into itself, we want to be able to deal with that.
625 if (this == c)
626 {
627 // Copy first part of _items to insert location
628 Array.Copy(_items, 0, _items, index, index);
629 // Copy last part of _items back to inserted location
630 Array.Copy(_items, index + count, _items, index * 2, _size - index);
631 }
632 else
633 {
634 c.CopyTo(_items, index);
635 }
636 _size += count;
637 }
638 }
639 else if (index < _size)
640 {
641 // We're inserting a lazy enumerable. Call Insert on each of the constituent items.
642 using (IEnumerator<T> en = collection.GetEnumerator())
643 {
644 while (en.MoveNext())
645 {
646 Insert(index++, en.Current);
647 }
648 }
649 }
650 else
651 {
652 // We're adding a lazy enumerable because the index is at the end of this list.
653 AddEnumerable(collection);
654 }
655 _version++;
656 }

734 // This method removes all items which matches the predicate.
735 // The complexity is O(n).
736 public int RemoveAll(Predicate<T> match)
737 {
738 int freeIndex = 0; // the first free slot in items array
739
740 // Find the first item which needs to be removed.
741 while (freeIndex < _size && !match(_items[freeIndex])) freeIndex++;
742 if (freeIndex >= _size) return 0;
743
744 int current = freeIndex + 1;
745 while (current < _size)
746 {
747 // Find the first item which needs to be kept.
748 while (current < _size && match(_items[current])) current++;
749
750 if (current < _size)
751 {
752 // copy item to the free slot.
753 _items[freeIndex++] = _items[current++];
754 }
755 }
756
757 if (!typeof(T).IsValueType)
758 {
759 Array.Clear(_items, freeIndex, _size - freeIndex); // Clear the elements so that the gc can reclaim the references.
760 }
761
762 int result = _size - freeIndex;
763 _size = freeIndex;
764 _version++;
765 return result;
766 }
