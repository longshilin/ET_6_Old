296 private string AddFileImpl(string path, string projectPath, PBXSourceTree tree, bool isFolderReference)
297 {
298 path = PBXPath.FixSlashes(path);
299 projectPath = PBXPath.FixSlashes(projectPath);
300
301 if (!isFolderReference && Path.GetExtension(path) != Path.GetExtension(projectPath))
302 throw new Exception("Project and real path extensions do not match");
303
304 string guid = FindFileGuidByProjectPath(projectPath);
305 if (guid == null)
306 guid = FindFileGuidByRealPath(path);
307 if (guid == null)
308 {
309 PBXFileReferenceData fileRef;
310 if (isFolderReference)
311 fileRef = PBXFileReferenceData.CreateFromFolderReference(path, PBXPath.GetFilename(projectPath), tree);
312 else
313 fileRef = PBXFileReferenceData.CreateFromFile(path, PBXPath.GetFilename(projectPath), tree);
314 PBXGroupData parent = CreateSourceGroup(PBXPath.GetDirectory(projectPath));
315 parent.children.AddGUID(fileRef.guid);
316 FileRefsAdd(path, projectPath, parent, fileRef);
317 guid = fileRef.guid;
318 }
319
320 return guid;
321 }

621 // Allow user to add a Capability
622 public bool AddCapability(string targetGuid, PBXCapabilityType capability, string entitlementsFilePath = null, bool addOptionalFramework = false)
623 {
624 // If the capability requires entitlements then you have to provide the name of it or we don't add the capability.
625 if (capability.requiresEntitlements && entitlementsFilePath == "")
626 {
627 throw new Exception("Couldn't add the Xcode Capability " + capability.id + " to the PBXProject file because this capability requires an entitlement file.");
628 }
629
630 var p = project.project;
631
632 // If an entitlement with a different name was added for another capability
633 // we don't add this capacity.
634 if (p.entitlementsFile != null && entitlementsFilePath != null && p.entitlementsFile != entitlementsFilePath)
635 {
636 if (p.capabilities.Count > 0)
637 {
638 //throw new WarningException("Attention, it seems that you have multiple entitlements file. Only one will be added the Project : " + p.entitlementsFile);
639 throw new Exception("Attention, it seems that you have multiple entitlements file. Only one will be added the Project : " + p.entitlementsFile);
640 }
641
642 return false;
643 }
644
645 // Add the capability only if it doesn't already exist.
646 if (p.capabilities.Contains(new PBXCapabilityType.TargetCapabilityPair(targetGuid, capability)))
647 {
648 //throw new WarningException("This capability has already been added. Method ignored");
649 }
650
651 p.capabilities.Add(new PBXCapabilityType.TargetCapabilityPair(targetGuid, capability));
652
653 // Add the required framework.
654 if (capability.framework != "" && !capability.optionalFramework ||
655 (capability.framework != "" && capability.optionalFramework && addOptionalFramework))
656 {
657 AddFrameworkToProject(targetGuid, capability.framework, false);
658 }
659
660 // Finally add the entitlement code signing if it wasn't added before.
661 if (entitlementsFilePath != null && p.entitlementsFile == null)
662 {
663 p.entitlementsFile = entitlementsFilePath;
664 AddFileImpl(entitlementsFilePath, entitlementsFilePath, PBXSourceTree.Source, false);
665 SetBuildProperty(targetGuid, "CODE_SIGN_ENTITLEMENTS", PBXPath.FixSlashes(entitlementsFilePath));
666 }
667
668 return true;
669 }

891 private PBXGroupData CreateSourceGroup(string sourceGroup)
892 {
893 sourceGroup = PBXPath.FixSlashes(sourceGroup);
894
895 if (sourceGroup == null || sourceGroup == "")
896 return GroupsGetMainGroup();
897
898 PBXGroupData gr = GroupsGetByProjectPath(sourceGroup);
899 if (gr != null)
900 return gr;
901
902 // the group does not exist -- create new
903 gr = GroupsGetMainGroup();
904
905 var elements = PBXPath.Split(sourceGroup);
906 string projectPath = null;
907 foreach (string pathEl in elements)
908 {
909 if (projectPath == null)
910 projectPath = pathEl;
911 else
912 projectPath += "/" + pathEl;
913
914 PBXGroupData child = GetPBXGroupChildByName(gr, pathEl);
915 if (child != null)
916 gr = child;
917 else
918 {
919 PBXGroupData newGroup = PBXGroupData.Create(pathEl, pathEl, PBXSourceTree.Group);
920 gr.children.AddGUID(newGroup.guid);
921 GroupsAdd(projectPath, gr, newGroup);
922 gr = newGroup;
923 }
924 }
925
926 return gr;

963 internal void AddExternalLibraryDependency(string targetGuid, string filename, string remoteFileGuid, string projectPath,
964 string remoteInfo)
965 {
966 PBXNativeTargetData target = nativeTargets[targetGuid];
967 filename = PBXPath.FixSlashes(filename);
968 projectPath = PBXPath.FixSlashes(projectPath);
969
970 // find the products group to put the new library in
971 string projectGuid = FindFileGuidByRealPath(projectPath);
972 if (projectGuid == null)
973 throw new Exception("No such project");
974
975 string productsGroupGuid = null;
976 foreach (var proj in project.project.projectReferences)
977 {
978 if (proj.projectRef == projectGuid)
979 {
980 productsGroupGuid = proj.group;
981 break;
982 }
983 }
984
985 if (productsGroupGuid == null)
986 throw new Exception("Malformed project: no project in project references");
987
988 PBXGroupData productGroup = GroupsGet(productsGroupGuid);
989
990 // verify file extension
991 string ext = Path.GetExtension(filename);
992 if (!FileTypeUtils.IsBuildableFile(ext))
993 throw new Exception("Wrong file extension");
994
995 // create ContainerItemProxy object
996 var container = PBXContainerItemProxyData.Create(projectGuid, "2", remoteFileGuid, remoteInfo);
997 containerItems.AddEntry(container);
998
999 // create a reference and build file for the library
1000 string typeName = FileTypeUtils.GetTypeName(ext);
1001
1002 var libRef = PBXReferenceProxyData.Create(filename, typeName, container.guid, "BUILT_PRODUCTS_DIR");
1003 references.AddEntry(libRef);
1004 PBXBuildFileData libBuildFile = PBXBuildFileData.CreateFromFile(libRef.guid, false, null);
1005 BuildFilesAdd(targetGuid, libBuildFile);
1006 BuildSectionAny(target, ext, false).files.AddGUID(libBuildFile.guid);
1007
1008 // add to products folder
1009 productGroup.children.AddGUID(libRef.guid);
