113 public string GetStackTrace(ILIntepreter intepreper)
114 {
115 StringBuilder sb = new StringBuilder();
116 ILRuntime.CLR.Method.ILMethod m;
117 StackFrame[] frames = intepreper.Stack.Frames.ToArray();
118 Mono.Cecil.Cil.Instruction ins = null;
119 RegisterVMSymbol vmSymbol;
120 if (frames[0].Address != null)
121 {
122 if (frames[0].IsRegister)
123 {
124 frames[0].Method.RegisterVMSymbols.TryGetValue(frames[0].Address.Value, out vmSymbol);
125 ins = vmSymbol.Instruction;
126 sb.AppendLine(string.Format("{0}(JIT_{1:0000}:{2})", ins, frames[0].Address.Value, frames[0].Method.BodyRegister[frames[0].Address.Value]));
127 }
128 else
129 {
130 ins = frames[0].Method.Definition.Body.Instructions[frames[0].Address.Value];
131 sb.AppendLine(ins.ToString());
132 }
133 }
134 for (int i = 0; i < frames.Length; i++)
135 {
136 var f = frames[i];
137 m = f.Method;
138 string document = "";
139 if (f.IsRegister)
140 {
141 if (f.Address != null)
142 {
143 if (f.Method.RegisterVMSymbols.TryGetValue(f.Address.Value, out vmSymbol))
144 {
145 RegisterVMSymbolLink link = null;
146 do
147 {
148 if (link != null)
149 vmSymbol = link.Value;
150 ins = vmSymbol.Instruction;
151 var md = vmSymbol.Method.Definition;
152 document = GetInstructionDocument(ins, md);
153 sb.AppendFormat("at {0} {1}\r\n", vmSymbol.Method, document);
154 link = vmSymbol.ParentSymbol;
155 }
156 while (link != null);
157 }
158 else
159 sb.AppendFormat("at {0} {1}\r\n", m, document);
160 }
161 else
162 sb.AppendFormat("at {0} {1}\r\n", m, document);
163 }
164 else
165 {
166 if (f.Address != null)
167 {
168 ins = m.Definition.Body.Instructions[f.Address.Value];
169 var md = m.Definition;
170
171 document = GetInstructionDocument(ins, md);
172 }
173 sb.AppendFormat("at {0} {1}\r\n", m, document);
174 }
175 }
176 return sb.ToString();
177 }

179 public unsafe string GetThisInfo(ILIntepreter intepreter)
180 {
181 var topFrame = intepreter.Stack.Frames.Peek();
182 var arg = Minus(topFrame.LocalVarPointer, topFrame.Method.ParameterCount);
183 if (topFrame.Method.HasThis)
184 arg--;
185 if (arg->ObjectType == ObjectTypes.StackObjectReference)
186 {
187 var addr = *(long*)&arg->Value;
188 arg = (StackObject*)addr;
189 }
190 ILTypeInstance instance = arg->ObjectType != ObjectTypes.Null ? intepreter.Stack.ManagedStack[arg->Value] as ILTypeInstance : null;
191 if (instance == null)
192 return "null";
193 var fields = instance.Type.TypeDefinition.Fields;
194 int idx = 0;
195 StringBuilder sb = new StringBuilder();
196 for (int i = 0; i < fields.Count; i++)
197 {
198 try
199 {
200 var f = fields[i];
201 if (f.IsStatic)
202 continue;
203 var field = instance.Fields[idx];
204 var v = StackObject.ToObject(&field, intepreter.AppDomain, instance.ManagedObjects);
205 if (v == null)
206 v = "null";
207 string name = f.Name;
208 sb.AppendFormat("{0} {1} = {2}", f.FieldType.Name, name, v);
209 if ((idx % 3 == 0 && idx != 0) || idx == instance.Fields.Length - 1)
210 sb.AppendLine();
211 else
212 sb.Append(", ");
213 idx++;
214 }
215 catch
216 {
217
218 }
219 }
220 return sb.ToString();
221 }

348 unsafe StackObject* ResolveCurrentFrameBasePointer(ILIntepreter intp,ILMethod method = null, int ip = -1)
349 {
350 StackObject* basePointer = intp.Stack.Frames.Peek().BasePointer;
351 if (method == null)
352 method = intp.Stack.Frames.Peek().Method;
353 if (ip < 0)
354 ip = intp.Stack.Frames.Peek().Address.Value;
355 if (intp.Stack.Frames.Peek().IsRegister)
356 {
357 basePointer = intp.Stack.Frames.Peek().LocalVarPointer;
358 RegisterVMSymbol vmSymbol;
359 if (method.RegisterVMSymbols.TryGetValue(ip, out vmSymbol))
360 {
361 var paramCnt = method.HasThis ? method.ParameterCount + 1 : method.ParameterCount;
362 var frameBase = basePointer - paramCnt;
363 int registerCnt = vmSymbol.Method.StackRegisterCount + vmSymbol.Method.LocalVariableCount;
364 if (method.HasThis)
365 frameBase--;
366 var curParamCnt = vmSymbol.Method.HasThis ? vmSymbol.Method.ParameterCount + 1 : vmSymbol.Method.ParameterCount;
367
368 if (vmSymbol.ParentSymbol != null)
369 {
370 basePointer = frameBase + vmSymbol.ParentSymbol.BaseRegisterIndex;
371 }
372 else
373 {
374 registerCnt -= vmSymbol.Method.StackRegisterCount;
375 basePointer = frameBase ;
376 }
377 basePointer = basePointer + curParamCnt + registerCnt;
378 }
379 }
380 return basePointer;
381 }

383 unsafe internal void CheckShouldBreak(ILMethod method, ILIntepreter intp, int ip)
384 {
385 if (server != null && server.IsAttached)
386 {
387 RegisterVMSymbol vmSymbol;
388 Mono.Cecil.Cil.Instruction ins = null;
389 Mono.Cecil.MethodDefinition md = null;
390 ILMethod m = method;
391 if (intp.Stack.Frames.Peek().IsRegister)
392 {
393 if (!method.IsRegisterVMSymbolFixed)
394 method.FixRegisterVMSymbol();
395 if(method.RegisterVMSymbols.TryGetValue(ip, out vmSymbol))
396 {
397 ins = vmSymbol.Instruction;
398 m = vmSymbol.Method;
399 md = vmSymbol.Method.Definition;
400
401 }
402 }
403 else
404 {
405 md = method.Definition;
406 ins = md.Body.Instructions[ip];
407 }
408 StackObject* basePointer = ResolveCurrentFrameBasePointer(intp, method, ip);
409
410 int methodHash = m.GetHashCode();
411 BreakpointInfo[] lst = null;
412
413 lock (activeBreakpoints)
414 {
415 LinkedList<BreakpointInfo> bps;
416 if (activeBreakpoints.TryGetValue(methodHash, out bps))
417 lst = bps.ToArray();
418 }
419 if (ins == null)
420 return;
421 if (lst != null)
422 {
423 var sp = md.DebugInformation.GetSequencePoint(ins);
424 if (sp != null)
425 {
426 foreach (var i in lst)
427 {
428 if ((i.StartLine + 1) == sp.StartLine)
429 {
430 DoBreak(intp, i.BreakpointHashCode, false);
431 return;
432 }
433 }
434 }
435 }
436
437 if (intp.CurrentStepType != StepTypes.None)
438 {
439 var sp = md.DebugInformation.GetSequencePoint(ins);
440 if (sp != null && IsSequenceValid(sp))
441 {
442 switch (intp.CurrentStepType)
443 {
444 case StepTypes.Into:
445 DoBreak(intp, 0, true);
446 break;
447 case StepTypes.Over:
448 if (basePointer <= intp.LastStepFrameBase && ip != intp.LastStepInstructionIndex)
449 {
450 DoBreak(intp, 0, true);
451 }
452 break;
453 case StepTypes.Out:
454 {
455 if (intp.Stack.Frames.Count > 0 && basePointer < intp.LastStepFrameBase)
456 {
457 DoBreak(intp, 0, true);
458 }
459 }
460 break;
461 }
462 }
463 }
464 }

472 void DoBreak(ILIntepreter intp, int bpHash, bool isStep)
473 {
474 var arr = AppDomain.Intepreters.ToArray();
475 KeyValuePair<int, StackFrameInfo[]>[] frames = new KeyValuePair<int, StackFrameInfo[]>[arr.Length];
476 frames[0] = new KeyValuePair<int, StackFrameInfo[]>(intp.GetHashCode(), GetStackFrameInfo(intp));
477 int idx = 1;
478 foreach (var j in arr)
479 {
480 if (j.Value != intp)
481 {
482 j.Value.ShouldBreak = true;
483 try
484 {
485 frames[idx++] = new KeyValuePair<int, Debugger.StackFrameInfo[]>(j.Value.GetHashCode(), GetStackFrameInfo(j.Value));
486 }
487 catch
488 {
489 frames[idx++] = new KeyValuePair<int, Debugger.StackFrameInfo[]>(j.Value.GetHashCode(), new StackFrameInfo[0]);
490 }
491 }
492 }
493 if (!isStep)
494 server.SendSCBreakpointHit(intp.GetHashCode(), bpHash, frames);
495 else
496 server.SendSCStepComplete(intp.GetHashCode(), frames);
497 //Breakpoint hit
498 intp.Break();
499 }

501 unsafe void InitializeStackFrameInfo(ILIntepreter intp, StackFrame f, List<StackFrameInfo> frameInfos)
502 {
503 Mono.Cecil.Cil.Instruction ins = null;
504 var m = f.Method;
505 int argCnt = m.HasThis ? m.ParameterCount + 1 : m.ParameterCount;
506 StackObject* frameBasePointer = Minus(f.LocalVarPointer, argCnt);
507 if (f.Address != null)
508 {
509 if (f.IsRegister)
510 {
511 RegisterVMSymbol vmSymbol;
512 if(m.RegisterVMSymbols.TryGetValue(f.Address.Value, out vmSymbol))
513 {
514 RegisterVMSymbolLink link = null;
515 StackObject* basePointer;
516 do
517 {
518 if (link != null)
519 {
520 vmSymbol = link.Value;
521 }
522 ins = vmSymbol.Instruction;
523 m = vmSymbol.Method;
524 if(vmSymbol.ParentSymbol!=null)
525 {
526 basePointer = Add(frameBasePointer, vmSymbol.ParentSymbol.BaseRegisterIndex);
527 }
528 else
529 {
530 basePointer = frameBasePointer;
531 }
532 var info = CreateStackFrameInfo(m, ins);
533 AddStackFrameInfoVariables(intp, info, m, basePointer);
534 frameInfos.Add(info);
535 link = vmSymbol.ParentSymbol;
536 }
537 while (link != null);
538 }
539 else
540 {
541 var info = CreateStackFrameInfo(m, null);
542 AddStackFrameInfoVariables(intp, info, m, frameBasePointer);
543 frameInfos.Add(info);
544 }
545 }
546 else
547 {
548 ins = m.Definition.Body.Instructions[f.Address.Value];
549 var info = CreateStackFrameInfo(m, ins);
550 AddStackFrameInfoVariables(intp, info, m, frameBasePointer);
551 frameInfos.Add(info);
552 }
553 }
554 else
555 {
556 var info = CreateStackFrameInfo(m, null);
557 AddStackFrameInfoVariables(intp, info, m, frameBasePointer);
558 frameInfos.Add(info);
559 }
560 }

582 unsafe void AddStackFrameInfoVariables(ILIntepreter intp, StackFrameInfo info, ILMethod m, StackObject* basePointer)
583 {
584 int argumentCount = m.ParameterCount;
585 if (m.HasThis)
586 argumentCount++;
587 info.LocalVariables = new VariableInfo[argumentCount + m.LocalVariableCount];
588 for (int i = 0; i < argumentCount; i++)
589 {
590 int argIdx = m.HasThis ? i - 1 : i;
591 var arg = basePointer;
592 string name = null;
593 object v = null;
594 string typeName = null;
595 var val = Add(arg, i);
596 v = StackObject.ToObject(val, intp.AppDomain, intp.Stack.ManagedStack);
597 if (argIdx >= 0)
598 {
599 var lv = m.Definition.Parameters[argIdx];
600 name = string.IsNullOrEmpty(lv.Name) ? "arg" + lv.Index : lv.Name;
601 typeName = lv.ParameterType.FullName;
602 }
603 else
604 {
605 name = "this";
606 typeName = m.DeclearingType.FullName;
607 }
608
609 VariableInfo vinfo = VariableInfo.FromObject(v);
610 vinfo.Address = (long)val;
611 vinfo.Name = name;
612 vinfo.TypeName = typeName;
613 vinfo.Expandable = GetValueExpandable(val, intp.Stack.ManagedStack);
614
615 info.LocalVariables[i] = vinfo;
616 }
617 for (int i = argumentCount; i < info.LocalVariables.Length; i++)
618 {
619 var locIdx = i - argumentCount;
620 var lv = m.Definition.Body.Variables[locIdx];
621 var val = Add(basePointer, argumentCount + locIdx);
622 var v = StackObject.ToObject(val, intp.AppDomain, intp.Stack.ManagedStack);
623 var type = intp.AppDomain.GetType(lv.VariableType, m.DeclearingType, m);
624 string vName = null;
625 m.Definition.DebugInformation.TryGetName(lv, out vName);
626 string name = string.IsNullOrEmpty(vName) ? "v" + lv.Index : vName;
627 VariableInfo vinfo = VariableInfo.FromObject(v);
628 vinfo.Address = (long)val;
629 vinfo.Name = name;
630 vinfo.TypeName = lv.VariableType.FullName;
631 vinfo.Expandable = GetValueExpandable(val, intp.Stack.ManagedStack);
632 info.LocalVariables[i] = vinfo;
633 }

649 {

652 {

659 }
660 return null;
661 }
662#endif
663 object obj;

665 if (obj != null)
666 {

669 return EnumArray((Array)obj, intepreter);

674 }

692 else

694 }
695 else

697 }

1281 internal unsafe void DumpStack(StackObject* esp, RuntimeStack stack)
1282 {
1283 var start = stack.StackBase;
1284 var end = esp + 10;
1285 var frames = stack.Frames;
1286 var mStack = stack.ManagedStack;
1287 var valuePointerEnd = stack.ValueTypeStackPointer;
1288 StringBuilder final = new StringBuilder();
1289 HashSet<long> leakVObj = new HashSet<long>();
1290 for (var i = stack.ValueTypeStackBase; i > stack.ValueTypeStackPointer;)
1291 {
1292 leakVObj.Add((long)i);
1293 i = Minus(i, i->ValueLow + 1);
1294 }
1295 for (var i = start; i <= end; i++)
1296 {
1297 StringBuilder sb = new StringBuilder();
1298 ILMethod localMethod = null, baseMethod = null;
1299 bool isLocal = false;
1300 bool isBase = false;
1301 int localIdx = 0;
1302 if (i == esp)
1303 sb.Append("->");
1304 foreach (var j in frames)
1305 {
1306 if (i >= j.LocalVarPointer && i < j.BasePointer)
1307 {
1308 isLocal = true;
1309 localIdx = (int)(i - j.LocalVarPointer);
1310 localMethod = j.Method;
1311 }
1312 else if (i == j.BasePointer)
1313 {
1314 isBase = true;
1315 baseMethod = j.Method;
1316 }
1317 }
1318 sb.Append(string.Format("(0x{0:X8}) Type:{1} ", (long)i, i->ObjectType));
1319 try
1320 {
1321 GetStackObjectText(sb, i, mStack, valuePointerEnd);
1322 }
1323 catch
1324 {
1325 sb.Append(" Cannot Fetch Object Info");
1326 }
1327 if (i < esp)
1328 {
1329 if (i->ObjectType == ObjectTypes.ValueTypeObjectReference)
1330 VisitValueTypeReference(ILIntepreter.ResolveReference(i), leakVObj);
1331 }
1332 if (isLocal)
1333 {
1334 sb.Append(string.Format("|Loc:{0}", localIdx));
1335 if (localIdx == 0)
1336 {
1337 sb.Append(" Method:");
1338 sb.Append(localMethod.ToString());
1339 }
1340 }
1341 if (isBase)
1342 {
1343 sb.Append("|Base");
1344 sb.Append(" Method:");
1345 sb.Append(baseMethod.ToString());
1346 }
1347
1348 final.AppendLine(sb.ToString());
1349 }
1350
1351 for (var i = stack.ValueTypeStackBase; i > stack.ValueTypeStackPointer;)
1352 {
1353 var vt = domain.GetTypeByIndex(i->Value);
1354 var cnt = i->ValueLow;
1355 bool leak = leakVObj.Contains((long)i);
1356 final.AppendLine("----------------------------------------------");
1357 final.AppendLine(string.Format("{2}(0x{0:X8}){1}", (long)i, vt, leak ? "*" : ""));
1358 for (int j = 0; j < cnt; j++)
1359 {
1360 StringBuilder sb = new StringBuilder();
1361 var ptr = Minus(i, j + 1);
1362 sb.Append(string.Format("(0x{0:X8}) Type:{1} ", (long)ptr, ptr->ObjectType));

1366 i = Minus(i, i->ValueLow + 1);
1367 }
1368 final.AppendLine("Managed Objects:");
1369 for (int i = 0; i < mStack.Count; i++)
1370 {
1371 final.AppendLine(string.Format("({0}){1}", i, mStack[i]));

1373#if !UNITY_5 && !UNITY_2017_1_OR_NEWER && !UNITY_4
