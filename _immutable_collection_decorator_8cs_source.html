52 internal static bool IdentifyImmutable(TypeModel model, Type declaredType, out MethodInfo builderFactory, out MethodInfo add, out MethodInfo addRange, out MethodInfo finish)
53 {
54 builderFactory = add = addRange = finish = null;
55 if (model == null || declaredType == null) return false;
56#if WINRT || COREFX
57 TypeInfo declaredTypeInfo = declaredType.GetTypeInfo();
58#else
59 Type declaredTypeInfo = declaredType;
60#endif
61
62 // try to detect immutable collections; firstly, they are all generic, and all implement IReadOnlyCollection<T> for some T
63 if(!declaredTypeInfo.IsGenericType) return false;
64
65#if WINRT || COREFX
66 Type[] typeArgs = declaredTypeInfo.GenericTypeArguments, effectiveType;
67#else
68 Type[] typeArgs = declaredTypeInfo.GetGenericArguments(), effectiveType;
69#endif
70 switch (typeArgs.Length)
71 {
72 case 1:
73 effectiveType = typeArgs;
74 break; // fine
75 case 2:
76 Type kvp = model.MapType(typeof(System.Collections.Generic.KeyValuePair<,>));
77 if (kvp == null) return false;
78 kvp = kvp.MakeGenericType(typeArgs);
79 effectiveType = new Type[] { kvp };
80 break;
81 default:
82 return false; // no clue!
83 }
84
85 if (ResolveIReadOnlyCollection(declaredType, null) == null) return false; // no IReadOnlyCollection<T> found
86
87 // and we want to use the builder API, so for generic Foo<T> or IFoo<T> we want to use Foo.CreateBuilder<T>
88 string name = declaredType.Name;
89 int i = name.IndexOf('`');
90 if (i <= 0) return false;
91 name = declaredTypeInfo.IsInterface ? name.Substring(1, i - 1) : name.Substring(0, i);
92
93 Type outerType = model.GetType(declaredType.Namespace + "." + name, declaredTypeInfo.Assembly);
94 // I hate special-cases...
95 if (outerType == null && name == "ImmutableSet")
96 {
97 outerType = model.GetType(declaredType.Namespace + ".ImmutableHashSet", declaredTypeInfo.Assembly);
98 }
99 if (outerType == null) return false;
100
101#if WINRT
102 foreach (MethodInfo method in outerType.GetTypeInfo().DeclaredMethods)
103#else
104 foreach (MethodInfo method in outerType.GetMethods())
105#endif
106 {
107 if (!method.IsStatic || method.Name != "CreateBuilder" || !method.IsGenericMethodDefinition || method.GetParameters().Length != 0
108 || method.GetGenericArguments().Length != typeArgs.Length) continue;
109
110 builderFactory = method.MakeGenericMethod(typeArgs);
111 break;
112 }
113 Type voidType = model.MapType(typeof(void));
114 if (builderFactory == null || builderFactory.ReturnType == null || builderFactory.ReturnType == voidType) return false;
115
116
117 add = Helpers.GetInstanceMethod(builderFactory.ReturnType, "Add", effectiveType);
118 if (add == null) return false;
119
120 finish = Helpers.GetInstanceMethod(builderFactory.ReturnType, "ToImmutable", Helpers.EmptyTypes);
121 if (finish == null || finish.ReturnType == null || finish.ReturnType == voidType) return false;
122
123 if (!(finish.ReturnType == declaredType || Helpers.IsAssignableFrom(declaredType, finish.ReturnType))) return false;
124
125 addRange = Helpers.GetInstanceMethod(builderFactory.ReturnType, "AddRange", new Type[] { declaredType });
126 if (addRange == null)
127 {
128 Type enumerable = model.MapType(typeof(System.Collections.Generic.IEnumerable<>), false);
129 if (enumerable != null)
130 {
131 addRange = Helpers.GetInstanceMethod(builderFactory.ReturnType, "AddRange", new Type[] { enumerable.MakeGenericType(effectiveType) });
132 }
133 }
134
135 return true;
136 }

149 public override object Read(object value, ProtoReader source)
150 {
151 object builderInstance = builderFactory.Invoke(null, null);
152 int field = source.FieldNumber;
153 object[] args = new object[1];
154 if (AppendToCollection && value != null && ((ICollection)value).Count != 0)
155 {
156 if(addRange !=null)
157 {
158 args[0] = value;
159 addRange.Invoke(builderInstance, args);
160 }
161 else
162 {
163 foreach(object item in (ICollection)value)
164 {
165 args[0] = item;
166 add.Invoke(builderInstance, args);
167 }
168 }
169 }
170
171 if (packedWireType != WireType.None && source.WireType == WireType.String)
172 {
173 SubItemToken token = ProtoReader.StartSubItem(source);
174 while (ProtoReader.HasSubValue(packedWireType, source))
175 {
176 args[0] = Tail.Read(null, source);
177 add.Invoke(builderInstance, args);
178 }
179 ProtoReader.EndSubItem(token, source);
180 }
181 else
182 {
183 do
184 {
185 args[0] = Tail.Read(null, source);
186 add.Invoke(builderInstance, args);
187 } while (source.TryReadFieldHeader(field));
188 }
189
190 return finish.Invoke(builderInstance, null);
191 }

195 protected override void EmitRead(Compiler.CompilerContext ctx, Compiler.Local valueFrom)
196 {
197 using (Compiler.Local oldList = AppendToCollection ? ctx.GetLocalWithValue(ExpectedType, valueFrom) : null)
198 using(Compiler.Local builder = new Compiler.Local(ctx, builderFactory.ReturnType))
199 {
200 ctx.EmitCall(builderFactory);
201 ctx.StoreValue(builder);
202
203 if(AppendToCollection)
204 {
205 Compiler.CodeLabel done = ctx.DefineLabel();
206 if(!Helpers.IsValueType(ExpectedType))
207 {
208 ctx.LoadValue(oldList);
209 ctx.BranchIfFalse(done, false); // old value null; nothing to add
210 }
211#if COREFX
212 TypeInfo typeInfo = ExpectedType.GetTypeInfo();
213#else
214 Type typeInfo = ExpectedType;
215#endif
216 PropertyInfo prop = Helpers.GetProperty(typeInfo, "Length", false);
217 if(prop == null) prop = Helpers.GetProperty(typeInfo, "Count", false);
218#if !NO_GENERICS
219 if (prop == null) prop = Helpers.GetProperty(ResolveIReadOnlyCollection(ExpectedType, Tail.ExpectedType), "Count", false);
220#endif
221 ctx.LoadAddress(oldList, oldList.Type);
222 ctx.EmitCall(Helpers.GetGetMethod(prop, false, false));
223 ctx.BranchIfFalse(done, false); // old list is empty; nothing to add
224
225 Type voidType = ctx.MapType(typeof(void));
226 if(addRange != null)
227 {
228 ctx.LoadValue(builder);
229 ctx.LoadValue(oldList);
230 ctx.EmitCall(addRange);
231 if (addRange.ReturnType != null && add.ReturnType != voidType) ctx.DiscardValue();
232 }
233 else
234 {
235 // loop and call Add repeatedly
236 MethodInfo moveNext, current, getEnumerator = GetEnumeratorInfo(ctx.Model, out moveNext, out current);
237 Helpers.DebugAssert(moveNext != null);
238 Helpers.DebugAssert(current != null);
239 Helpers.DebugAssert(getEnumerator != null);
240
241 Type enumeratorType = getEnumerator.ReturnType;
242 using (Compiler.Local iter = new Compiler.Local(ctx, enumeratorType))
243 {
244 ctx.LoadAddress(oldList, ExpectedType);
245 ctx.EmitCall(getEnumerator);
246 ctx.StoreValue(iter);
247 using (ctx.Using(iter))
248 {
249 Compiler.CodeLabel body = ctx.DefineLabel(), next = ctx.DefineLabel();
250 ctx.Branch(next, false);
251
252 ctx.MarkLabel(body);
253 ctx.LoadAddress(builder, builder.Type);
254 ctx.LoadAddress(iter, enumeratorType);
255 ctx.EmitCall(current);
256 ctx.EmitCall(add);
257 if (add.ReturnType != null && add.ReturnType != voidType) ctx.DiscardValue();
258
259 ctx.MarkLabel(@next);
260 ctx.LoadAddress(iter, enumeratorType);
261 ctx.EmitCall(moveNext);
262 ctx.BranchIfTrue(body, false);
263 }
264 }
265 }
266
267
268 ctx.MarkLabel(done);
269 }
270
271 EmitReadList(ctx, builder, Tail, add, packedWireType, false);
272
273 ctx.LoadAddress(builder, builder.Type);
274 ctx.EmitCall(finish);
275 if(ExpectedType != finish.ReturnType)
276 {
277 ctx.Cast(ExpectedType);
278 }
279 }
280 }
