16 class MergeSort<T> {
17 private readonly T [] elements;
18 private readonly T [] buffer;
19 private readonly IComparer<T> comparer;
20
21 private MergeSort (T [] elements, IComparer<T> comparer)
22 {
23 this.elements = elements;
24 this.buffer = new T [elements.Length];
25 Array.Copy (this.elements, this.buffer, elements.Length);
26 this.comparer = comparer;
27 }
28
29 public static void Sort (T [] source, IComparer<T> comparer)
30 {
31 Sort (source, 0, source.Length, comparer);
32 }
33
34 public static void Sort (T [] source, int start, int length, IComparer<T> comparer)
35 {
36 new MergeSort<T> (source, comparer).Sort (start, length);
37 }
38
39 private void Sort (int start, int length)
40 {
41 TopDownSplitMerge (this.buffer, this.elements, start, length);
42 }
43
44 private void TopDownSplitMerge (T [] a, T [] b, int start, int end)
45 {
46 if (end - start < 2)
47 return;
48
49 int middle = (end + start) / 2;
50 TopDownSplitMerge (b, a, start, middle);
51 TopDownSplitMerge (b, a, middle, end);
52 TopDownMerge (a, b, start, middle, end);
53 }
54
55 private void TopDownMerge (T [] a, T [] b, int start, int middle, int end)
56 {
57 for (int i = start, j = middle, k = start; k < end; k++) {
58 if (i < middle && (j >= end || comparer.Compare (a [i], a [j]) <= 0)) {
59 b [k] = a [i++];
60 } else {
61 b [k] = a [j++];
62 }
63 }
64 }
65 }
