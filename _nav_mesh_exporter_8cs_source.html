110 private static void InputVertices(Vector3[] vertices)
111 {
112 for (int i = 0, n = vertices.Length - 1; i <= n; i++)
113 {
114 var point = vertices[i];
115 var x = (float) Math.Round(point.x, 2);
116 var y = (float) Math.Round(point.y, 2);
117 var z = (float) Math.Round(point.z, 2);
118 if (!pointVertDict.ContainsKey(x))
119 {
120 pointVertDict.Add(x, new Dictionary<float, Vert>());
121 }
122
123 Vert vert;
124 if (pointVertDict[x].ContainsKey(z))
125 {
126 vert = pointVertDict[x][z];
127 }
128 else
129 {
130 vert = new Vert();
131 vert.x = x;
132 vert.y = y;
133 vert.z = z;
134 pointVertDict[x][z] = vert;
135 }
136
137 indexVertDict.Add(i, vert);
138 }
139 }

141 private static void InputTriangles(int[] indices, int[] areas)
142 {
143 Face face = null;
144 var faceIndices = new HashSet<int>();
145 for (int i = 0, n = areas.Length; i < n; i++)
146 {
147 var triangleIndexList = new int[3];
148 var triangleVertList = new Vert[3];
149 for (var j = 0; j < 3; j++)
150 {
151 triangleIndexList[j] = indices[i * 3 + j];
152 triangleVertList[j] = indexVertDict[triangleIndexList[j]];
153 }
154
155 var vert0 = triangleVertList[0];
156 var vert1 = triangleVertList[1];
157 var vert2 = triangleVertList[2];
158 if (vert0 == vert1 || vert1 == vert2 || vert2 == vert0)
159 {
160 continue;
161 }
162
163 var newFace = true;
164 var area = areas[i] >= 3? areas[i] - 2 : 0;
165 if (face != null && face.area == area)
166 {
167 for (var j = 0; j < 3; j++)
168 {
169 if (faceIndices.Contains(triangleIndexList[j]))
170 {
171 newFace = false;
172 break;
173 }
174 }
175 }
176
177 if (newFace)
178 {
179 if (face != null)
180 {
181 InitFace(face);
182 faceIndices.Clear();
183 }
184
185 face = new Face();
186 face.area = area;
187 }
188
189 double x1 = vert1.x - vert0.x;
190 double y1 = vert1.y - vert0.y;
191 double z1 = vert1.z - vert0.z;
192 double x2 = vert2.x - vert0.x;
193 double y2 = vert2.y - vert0.y;
194 double z2 = vert2.z - vert0.z;
195 double normalA = y1 * z2 - z1 * y2;
196 double normalB = z1 * x2 - x1 * z2;
197 double normalC = x1 * y2 - y1 * x2;
198 if (normalB < -0.000001 || 0.000001 < normalB)
199 {
200 var normalD = normalA + normalB + normalC;
201 if (normalD > face.normalD)
202 {
203 face.normalA = normalA;
204 face.normalB = normalB;
205 face.normalC = normalC;
206 face.normalD = normalD;
207 }
208 }
209
210 for (var j = 0; j < 3; j++)
211 {
212 if (!faceIndices.Contains(triangleIndexList[j]))
213 {
214 faceIndices.Add(triangleIndexList[j]);
215 face.verts.Add(triangleVertList[j]);
216 }
217 }
218 }
219
220 if (face != null)
221 {
222 InitFace(face);
223 }
224
225 foreach (var pair in pairList)
226 {
227 var firstFace = pair.firstEdgeFace;
228 var secondFace = pair.secondEdgeFace;
229 var firstDistance = GetDistance(firstFace.centerX - pair.centerX, firstFace.centerZ - pair.centerZ);
230 var secondDistance = GetDistance(secondFace.centerX - pair.centerX, secondFace.centerZ - pair.centerZ);
231 pair.distance = firstDistance + secondDistance;
232 }
233 }
234
235 private static float GetDistance(float deltaX, float deltaZ)
236 {
237 return (float) Math.Round(Math.Sqrt((double) deltaX * (double) deltaX + (double) deltaZ * (double) deltaZ), 2);
238 }

240 private static void InitFace(Face face)
241 {
242 face.centerX = 0;
243 face.centerZ = 0;
244 var vertCount = face.verts.Count;
245 foreach (var vert in face.verts)
246 {
247 face.centerX += vert.x;
248 face.centerZ += vert.z;
249 if (!vertFaceDict.ContainsKey(vert))
250 {
251 vertFaceDict.Add(vert, face);
252 vertList.Add(vert);
253 }
254 }
255
256 face.centerX /= vertCount;
257 face.centerZ /= vertCount;
258 if (face.normalB != 0)
259 {
260 face.normalX = (float) Math.Round(face.normalA / face.normalB, 6);
261 face.normalZ = (float) Math.Round(face.normalC / face.normalB, 6);
262 }
263
264 for (int i = 0, n = vertCount - 1; i <= n; i++)
265 {
266 var firstVert = face.verts[i];
267 var secondVert = face.verts[i == n? 0 : i + 1];
268 if (!vertPairDict.ContainsKey(firstVert))
269 {
270 vertPairDict.Add(firstVert, new Dictionary<Vert, Pair>());
271 }
272
273 if (!vertPairDict.ContainsKey(secondVert))
274 {
275 vertPairDict.Add(secondVert, new Dictionary<Vert, Pair>());
276 }
277
278 if (!vertPairDict[secondVert].ContainsKey(firstVert))
279 {
280 var pair = new Pair();
281 pair.firstEdgeFace = face;
282 pair.firstEdgeIndex = i;
283 vertPairDict[firstVert][secondVert] = pair;
284 }
285 else
286 {
287 var pair = vertPairDict[secondVert][firstVert];
288 pair.centerX = (firstVert.x + secondVert.x) / 2;
289 pair.centerZ = (firstVert.z + secondVert.z) / 2;
290 pair.secondEdgeFace = face;
291 pair.secondEdgeIndex = i;
292 pairList.Add(pair);
293 }
294 }
295
296 faceList.Add(face);
297 }

299 private static void IndexVertsAndFaces()
300 {
301 var minX = float.MaxValue;
302 var maxX = float.MinValue;
303 var minZ = float.MaxValue;
304 var maxZ = float.MinValue;
305 foreach (var vert in vertList)
306 {
307 if (minX > vert.x)
308 {
309 minX = vert.x;
310 }
311
312 if (maxX < vert.x)
313 {
314 maxX = vert.x;
315 }
316
317 if (minZ > vert.z)
318 {
319 minZ = vert.z;
320 }
321
322 if (maxZ < vert.x)
323 {
324 maxZ = vert.x;
325 }
326 }
327
328 var hilbertX = 65535f / (maxX - minX);
329 var hilbertZ = 65535f / (maxZ - minZ);
330 foreach (var face in faceList)
331 {
332 var X = (uint) Math.Round((face.centerX - minX) * hilbertX);
333 var Z = (uint) Math.Round((face.centerZ - minZ) * hilbertZ);
334 var a = X ^ Z;
335 var b = 0xFFFF ^ a;
336 var c = 0xFFFF ^ (X | Z);
337 var d = X & (Z ^ 0xFFFF);
338 var A = a | (b >> 1);
339 var B = (a >> 1) ^ a;
340 var C = ((c >> 1) ^ (b & (d >> 1))) ^ c;
341 var D = ((a & (c >> 1)) ^ (d >> 1)) ^ d;
342 a = A;
343 b = B;
344 c = C;
345 d = D;
346 A = (a & (a >> 2)) ^ (b & (b >> 2));
347 B = (a & (b >> 2)) ^ (b & ((a ^ b) >> 2));
348 C ^= (a & (c >> 2)) ^ (b & (d >> 2));
349 D ^= (b & (c >> 2)) ^ ((a ^ b) & (d >> 2));
350 a = A;
351 b = B;
352 c = C;
353 d = D;
354 A = (a & (a >> 4)) ^ (b & (b >> 4));
355 B = (a & (b >> 4)) ^ (b & ((a ^ b) >> 4));
356 C ^= (a & (c >> 4)) ^ (b & (d >> 4));
357 D ^= (b & (c >> 4)) ^ ((a ^ b) & (d >> 4));
358 a = A;
359 b = B;
360 c = C;
361 d = D;
362 C ^= (a & (c >> 8)) ^ (b & (d >> 8));
363 D ^= (b & (c >> 8)) ^ ((a ^ b) & (d >> 8));
364 C ^= C >> 1;
365 D ^= D >> 1;
366 c = X ^ Z;
367 d = D | (0xFFFF ^ (c | C));
368 c = (c | (c << 8)) & 0x00FF00FF;
369 c = (c | (c << 4)) & 0x0F0F0F0F;
370 c = (c | (c << 2)) & 0x33333333;
371 c = (c | (c << 1)) & 0x55555555;
372 d = (d | (d << 8)) & 0x00FF00FF;
373 d = (d | (d << 4)) & 0x0F0F0F0F;
374 d = (d | (d << 2)) & 0x33333333;
375 d = (d | (d << 1)) & 0x55555555;
376 face.sortValue = (d << 1) | c;
377 }
378
379 faceList.Sort(SortComparison);
380 for (int i = 0, n = vertList.Count; i < n; i++)
381 {
382 vertList[i].id = i;
383 }
384
385 for (int i = 0, n = faceList.Count; i < n; i++)
386 {
387 faceList[i].id = i;
388 }
389 }
390
391 private static int SortComparison(Face a, Face b)
392 {
393 return a.sortValue.CompareTo(b.sortValue);
394 }

396 private static void WriteFile()
397 {
398 if (!System.IO.Directory.Exists(outputClientFolder))
399 {
400 System.IO.Directory.CreateDirectory(outputClientFolder);
401 }
402
403 var path = outputClientFolder + SceneManager.GetActiveScene().name + ".bytes";
404 var writer = new BinaryWriter(new FileStream(path, FileMode.Create));
405 writer.Write('N');
406 writer.Write('a');
407 writer.Write('v');
408 writer.Write('M');
409 writer.Write('e');
410 writer.Write('s');
411 writer.Write('h');
412 writer.Write(VERSION);
413 writer.Write(vertList.Count);
414 foreach (var vert in vertList)
415 {
416 writer.Write(vert.x);
417 writer.Write(vert.y);
418 writer.Write(vert.z);
419 }
420
421 writer.Write(faceList.Count);
422 foreach (var face in faceList)
423 {
424 writer.Write(face.area);
425 writer.Write(face.normalX);
426 writer.Write(face.normalZ);
427 writer.Write(face.verts.Count);
428 foreach (var vert in face.verts)
429 {
430 writer.Write(vert.id);
431 }
432 }
433
434 writer.Write(pairList.Count);
435 foreach (var pair in pairList)
436 {
437 writer.Write(pair.distance);
438 writer.Write(pair.firstEdgeFace.id);
439 writer.Write(pair.firstEdgeIndex);
440 writer.Write(pair.secondEdgeFace.id);
441 writer.Write(pair.secondEdgeIndex);
442 }
443
444 writer.Flush();
445 writer.Close();
446 AssetDatabase.Refresh();
447 }

600 public static string MeshToString(MeshFilter mf, Dictionary<string, ObjMaterial> materialList)
601 {
602 Mesh m = mf.sharedMesh;
603 Material[] mats = mf.GetComponent<Renderer>().sharedMaterials;
604 StringBuilder sb = new StringBuilder();
605 sb.Append("g ").Append(mf.name).Append("\n");
606 // foreach(Vector3 v in m.vertices) {
607 // sb.Append(string.Format("v {0} {1} {2}\n",v.x,v.y,v.z));
608 // }
609 foreach (Vector3 lv in m.vertices)
610 {
611 Vector3 wv = mf.transform.TransformPoint(lv);
612 //This is sort of ugly - inverting x-component since we're in
613 //a different coordinate system than "everyone" is "used to".
614 sb.Append(string.Format("v {0} {1} {2}\n", -wv.x, wv.y, wv.z));
615 }
616
617 sb.Append("\n");
618
619 // foreach(Vector3 v in m.normals) {
620 // sb.Append(string.Format("vn {0} {1} {2}\n",v.x,v.y,v.z));
621 // }
622 foreach (Vector3 lv in m.normals)
623 {
624 Vector3 wv = mf.transform.TransformDirection(lv);
625 sb.Append(string.Format("vn {0} {1} {2}\n", -wv.x, wv.y, wv.z));
626 }
627
628 sb.Append("\n");
629
630 foreach (Vector3 v in m.uv)
631 {
632 sb.Append(string.Format("vt {0} {1}\n", v.x, v.y));
633 }
634
635 int countMat = m.subMeshCount;
636 if (mats == null)
637 {
638 Debug.LogWarning($"NavMeshExporter MeshToString Error - 没有找到材质");
639 return sb.ToString();
640 }
641 else if (mats.Length < countMat)
642 {
643 Debug.LogWarning($"NavMeshExporter MeshToString Error - 共享材质数量小于该物体的子物体数量 - {mats.Length} / {countMat}");
644 countMat = mats.Length;
645 }
646
647 for (int material = 0; material < countMat; material++)
648 {
649 string nameMat = "null";
650 Texture mainTexture = null;
651 if (mats[material] != null)
652 {
653 nameMat = mats[material].name;
654 mainTexture = mats[material].mainTexture;
655 }
656
657 sb.Append("\n");
658 sb.Append("usemtl ").Append(nameMat).Append("\n");
659 sb.Append("usemap ").Append(nameMat).Append("\n");
660
661 //See if this material is already in the materiallist.
662 try
663 {
664 ObjMaterial objMaterial = new ObjMaterial();
665 objMaterial.name = nameMat;
666 if (mainTexture)
667 objMaterial.textureName = AssetDatabase.GetAssetPath(mainTexture);
668 else
669 objMaterial.textureName = null;
670 materialList.Add(objMaterial.name, objMaterial);
671 }
672 catch (ArgumentException)
673 {
674 //Already in the dictionary
675 }
676
677 // int[] triangles = m.GetTriangles(material);
678 // for (int i=0;i<triangles.Length;i+=3) {
679 // sb.Append(string.Format("f {0}/{0}/{0} {1}/{1}/{1} {2}/{2}/{2}\n",
680 // triangles[i]+1, triangles[i+1]+1, triangles[i+2]+1));
681 // }
682 int[] triangles = m.GetTriangles(material);
683 for (int i = 0; i < triangles.Length; i += 3)
684 {
685 //Because we inverted the x-component, we also needed to alter the triangle winding.
686 sb.Append(string.Format("f {1}/{1}/{1} {0}/{0}/{0} {2}/{2}/{2}\n",
687 triangles[i] + 1 + vertexOffset, triangles[i + 1] + 1 + normalOffset, triangles[i + 2] + 1 + uvOffset));
688 }
689 }
690
691 vertexOffset += m.vertices.Length;
692 normalOffset += m.normals.Length;
693 uvOffset += m.uv.Length;
694 return sb.ToString();
695 }
