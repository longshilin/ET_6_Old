202 static ElementType ConvertGenericParameterType (GenericParameterType type)
203 {
204 switch (type) {
205 case GenericParameterType.Type:
206 return ElementType.Var;
207 case GenericParameterType.Method:
208 return ElementType.MVar;
209 }
210
211 throw new ArgumentOutOfRangeException ();
