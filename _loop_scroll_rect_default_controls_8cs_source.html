39 private static void SetPositionVisibleinSceneView(RectTransform canvasRTransform, RectTransform itemTransform)
40 {
41 // Find the best scene view
42 SceneView sceneView = SceneView.lastActiveSceneView;
43 if (sceneView == null && SceneView.sceneViews.Count > 0)
44 sceneView = SceneView.sceneViews[0] as SceneView;
45
46 // Couldn't find a SceneView. Don't set position.
47 if (sceneView == null || sceneView.camera == null)
48 return;
49
50 // Create world space Plane from canvas position.
51 Vector2 localPlanePosition;
52 Camera camera = sceneView.camera;
53 Vector3 position = Vector3.zero;
54 if (RectTransformUtility.ScreenPointToLocalPointInRectangle(canvasRTransform, new Vector2(camera.pixelWidth / 2, camera.pixelHeight / 2), camera, out localPlanePosition))
55 {
56 // Adjust for canvas pivot
57 localPlanePosition.x = localPlanePosition.x + canvasRTransform.sizeDelta.x * canvasRTransform.pivot.x;
58 localPlanePosition.y = localPlanePosition.y + canvasRTransform.sizeDelta.y * canvasRTransform.pivot.y;
59
60 localPlanePosition.x = Mathf.Clamp(localPlanePosition.x, 0, canvasRTransform.sizeDelta.x);
61 localPlanePosition.y = Mathf.Clamp(localPlanePosition.y, 0, canvasRTransform.sizeDelta.y);
62
63 // Adjust for anchoring
64 position.x = localPlanePosition.x - canvasRTransform.sizeDelta.x * itemTransform.anchorMin.x;
65 position.y = localPlanePosition.y - canvasRTransform.sizeDelta.y * itemTransform.anchorMin.y;
66
67 Vector3 minLocalPosition;
68 minLocalPosition.x = canvasRTransform.sizeDelta.x * (0 - canvasRTransform.pivot.x) + itemTransform.sizeDelta.x * itemTransform.pivot.x;
69 minLocalPosition.y = canvasRTransform.sizeDelta.y * (0 - canvasRTransform.pivot.y) + itemTransform.sizeDelta.y * itemTransform.pivot.y;
70
71 Vector3 maxLocalPosition;
72 maxLocalPosition.x = canvasRTransform.sizeDelta.x * (1 - canvasRTransform.pivot.x) - itemTransform.sizeDelta.x * itemTransform.pivot.x;
73 maxLocalPosition.y = canvasRTransform.sizeDelta.y * (1 - canvasRTransform.pivot.y) - itemTransform.sizeDelta.y * itemTransform.pivot.y;
74
75 position.x = Mathf.Clamp(position.x, minLocalPosition.x, maxLocalPosition.x);
76 position.y = Mathf.Clamp(position.y, minLocalPosition.y, maxLocalPosition.y);
77 }
78
79 itemTransform.anchoredPosition = position;
80 itemTransform.localRotation = Quaternion.identity;
81 itemTransform.localScale = Vector3.one;
82 }

84 private static void PlaceUIElementRoot(GameObject element, MenuCommand menuCommand)
85 {
86 GameObject parent = menuCommand.context as GameObject;
87 if (parent == null || parent.GetComponentInParent<Canvas>() == null)
88 {
89 parent = GetOrCreateCanvasGameObject();
90 }
91
92 string uniqueName = GameObjectUtility.GetUniqueNameForSibling(parent.transform, element.name);
93 element.name = uniqueName;
94 Undo.RegisterCreatedObjectUndo(element, "Create " + element.name);
95 Undo.SetTransformParent(element.transform, parent.transform, "Parent " + element.name);
96 GameObjectUtility.SetParentAndAlign(element, parent);
97 if (parent != menuCommand.context) // not a context click, so center in sceneview
98 SetPositionVisibleinSceneView(parent.GetComponent<RectTransform>(), element.GetComponent<RectTransform>());
99
100 Selection.activeGameObject = element;
101 }

119 // Helper function that returns a Canvas GameObject; preferably a parent of the selection, or other existing Canvas.
120 static public GameObject GetOrCreateCanvasGameObject()
121 {
122 GameObject selectedGo = Selection.activeGameObject;
123
124 // Try to find a gameobject that is the selected GO or one if its parents.
125 Canvas canvas = (selectedGo != null) ? selectedGo.GetComponentInParent<Canvas>() : null;
126 if (canvas != null && canvas.gameObject.activeInHierarchy)
127 return canvas.gameObject;
128
129 // No canvas in selection or its parents? Then use just any canvas..
130 canvas = Object.FindObjectOfType(typeof(Canvas)) as Canvas;
131 if (canvas != null && canvas.gameObject.activeInHierarchy)
132 return canvas.gameObject;
133
134 // No canvas in the scene at all? Then create a new one.
135 return LoopScrollRectMenuOptions.CreateNewUI();
136 }
