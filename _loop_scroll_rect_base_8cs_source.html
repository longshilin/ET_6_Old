402 public void SrollToCellWithinTime(int index, float time)
403 {
404 if (totalCount >= 0 && (index < 0 || index >= totalCount))
405 {
406 Debug.LogErrorFormat("invalid index {0}", index);
407 return;
408 }
409 StopAllCoroutines();
410 if (time <= 0)

412 RefillCells(index);
413 return;
414 }
415 float dist = 0;
416 float offset = 0;
417 int currentFirst = reverseDirection ? GetLastItem(out offset) : GetFirstItem(out offset);
418
419 int TargetLine = (index / contentConstraintCount);
420 int CurrentLine = (currentFirst / contentConstraintCount);
421
422 if (TargetLine == CurrentLine)

424 dist = offset;

426 else
427 {
428 if (sizeHelper != null)
429 {

434 {

437 dist -= offset;

443 IEnumerator ScrollToCellCoroutine(int index, float speed)
444 {
445 bool needMoving = true;
446 while (needMoving)
447 {
448 yield return null;
449 if (!m_Dragging)
450 {
451 float move = 0;
452 if (index < itemTypeStart)
453 {
454 move = -Time.deltaTime * speed;
455 }
456 else if (index >= itemTypeEnd)
457 {
458 move = Time.deltaTime * speed;
459 }
460 else
461 {
462 m_ViewBounds = new Bounds(viewRect.rect.center, viewRect.rect.size);
463 var m_ItemBounds = GetBounds4Item(index);
464 var offset = 0.0f;
465 if (direction == LoopScrollRectDirection.Vertical)
466 offset = reverseDirection ? (m_ViewBounds.min.y - m_ItemBounds.min.y) : (m_ViewBounds.max.y - m_ItemBounds.max.y);
467 else
468 offset = reverseDirection ? (m_ItemBounds.max.x - m_ViewBounds.max.x) : (m_ItemBounds.min.x - m_ViewBounds.min.x);
469 // check if we cannot move on
470 if (totalCount >= 0)
471 {
472 if (offset > 0 && itemTypeEnd == totalCount && !reverseDirection)
473 {
474 m_ItemBounds = GetBounds4Item(totalCount - 1);
475 // reach bottom
476 if ((direction == LoopScrollRectDirection.Vertical && m_ItemBounds.min.y > m_ViewBounds.min.y) ||
477 (direction == LoopScrollRectDirection.Horizontal && m_ItemBounds.max.x < m_ViewBounds.max.x))
478 {
479 needMoving = false;
480 break;
481 }
482 }
483 else if (offset < 0 && itemTypeStart == 0 && reverseDirection)
484 {
485 m_ItemBounds = GetBounds4Item(0);
486 if ((direction == LoopScrollRectDirection.Vertical && m_ItemBounds.max.y < m_ViewBounds.max.y) ||
487 (direction == LoopScrollRectDirection.Horizontal && m_ItemBounds.min.x > m_ViewBounds.min.x))
488 {
489 needMoving = false;
490 break;
491 }
492 }
493 }
494
495 float maxMove = Time.deltaTime * speed;
496 if (Mathf.Abs(offset) < maxMove)
497 {
498 needMoving = false;
499 move = offset;
500 }
501 else
502 move = Mathf.Sign(offset) * maxMove;
503 }
504 if (move != 0)
505 {
506 Vector2 offset = GetVector(move);
507 content.anchoredPosition += offset;
508 m_PrevPosition += offset;
509 m_ContentStartPosition += offset;
510 UpdateBounds(true);
511 }
512 }
513 }
514 StopMovement();
515 UpdatePrevData();

544 public void RefillCellsFromEnd(int endItem = 0, bool alignStart = false)
545 {
546 if (!Application.isPlaying)
547 return;
548
549 itemTypeEnd = reverseDirection ? endItem : totalCount - endItem;
550 itemTypeStart = itemTypeEnd;
551
552 if (totalCount >= 0 && itemTypeStart % contentConstraintCount != 0)
553 {
554 itemTypeStart = (itemTypeStart / contentConstraintCount) * contentConstraintCount;
555 }
556
557 ReturnToTempPool(!reverseDirection, m_Content.childCount);
558
559 float sizeToFill = GetAbsDimension(viewRect.rect.size), sizeFilled = 0;
560
561 bool first = true;
562 while (sizeToFill > sizeFilled)
563 {
564 float size = reverseDirection ? NewItemAtEnd(!first) : NewItemAtStart(!first);
565 if (size < 0)
566 break;
567 first = false;
568 sizeFilled += size;
569 }
570
571 // refill from start in case not full yet
572 while (sizeToFill > sizeFilled)
573 {
574 float size = reverseDirection ? NewItemAtStart(!first) : NewItemAtEnd(!first);
575 if (size < 0)
576 break;
577 first = false;
578 sizeFilled += size;
579 }
580
581 Vector2 pos = m_Content.anchoredPosition;
582 float dist = alignStart ? 0 : Mathf.Max(0, sizeFilled - sizeToFill);
583 if (reverseDirection)
584 dist = -dist;
585 if (direction == LoopScrollRectDirection.Vertical)
586 pos.y = dist;
587 else
588 pos.x = -dist;
589 m_Content.anchoredPosition = pos;
590 m_ContentStartPosition = pos;
591
592 ClearTempPool();
593 // force build bounds here so scrollbar can access newest bounds
594 LayoutRebuilder.ForceRebuildLayoutImmediate(content);
595 m_ContentBounds = GetBounds();
596 UpdateScrollbars(Vector2.zero);
597 StopMovement();
598 UpdatePrevData();

601 public void RefillCells(int startItem = 0, bool fillViewRect = false, float contentOffset = 0)
602 {
603 if (!Application.isPlaying)
604 return;
605
606 itemTypeStart = reverseDirection ? totalCount - startItem : startItem;
607 if (totalCount >= 0 && itemTypeStart % contentConstraintCount != 0)
608 {
609 itemTypeStart = (itemTypeStart / contentConstraintCount) * contentConstraintCount;
610 }
611 itemTypeEnd = itemTypeStart;
612
613 // Don't `Canvas.ForceUpdateCanvases();` here, or it will new/delete cells to change itemTypeStart/End
614 ReturnToTempPool(reverseDirection, m_Content.childCount);
615
616 float sizeToFill = GetAbsDimension(viewRect.rect.size) + Mathf.Abs(contentOffset);
617 float sizeFilled = 0;
618 // m_ViewBounds may be not ready when RefillCells on Start
619
620 float itemSize = 0;
621
622 bool first = true;
623 while (sizeToFill > sizeFilled)
624 {
625 float size = reverseDirection ? NewItemAtStart(!first) : NewItemAtEnd(!first);
626 if (size < 0)
627 break;
628 first = false;
629 itemSize = size;
630 sizeFilled += size;
631 }
632
633 // refill from start in case not full yet
634 while (sizeToFill > sizeFilled)
635 {
636 float size = reverseDirection ? NewItemAtEnd(!first) : NewItemAtStart(!first);
637 if (size < 0)
638 break;
639 first = false;
640 sizeFilled += size;
641 }
642
643 if (fillViewRect && itemSize > 0 && sizeFilled < sizeToFill)
644 {
645 int itemsToAddCount = (int)((sizeToFill - sizeFilled) / itemSize); //calculate how many items can be added above the offset, so it still is visible in the view
646 int newOffset = startItem - itemsToAddCount;
647 if (newOffset < 0) newOffset = 0;
648 if (newOffset != startItem) RefillCells(newOffset); //refill again, with the new offset value, and now with fillViewRect disabled.
649 }
650
651 Vector2 pos = m_Content.anchoredPosition;
652 if (direction == LoopScrollRectDirection.Vertical)
653 pos.y = -contentOffset;
654 else
655 pos.x = contentOffset;
656 m_Content.anchoredPosition = pos;
657 m_ContentStartPosition = pos;
658
659 ClearTempPool();
660 // force build bounds here so scrollbar can access newest bounds
661 LayoutRebuilder.ForceRebuildLayoutImmediate(content);
662 m_ContentBounds = GetBounds();
663 UpdateScrollbars(Vector2.zero);
664 StopMovement();
665 UpdatePrevData();
666 }

669 {

672 return -1;
673 }

676 {

678 RectTransform newItem = GetFromTempPool(itemTypeStart);
679 newItem.SetSiblingIndex(deletedItemTypeStart);

688 m_PrevPosition += offset;
689 m_ContentStartPosition += offset;

908 delta.y *= -1;

910 {
911 if (Mathf.Abs(delta.x) > Mathf.Abs(delta.y))
912 delta.y = delta.x;
913 delta.x = 0;
914 }
915 if (horizontal && !vertical)
916 {
917 if (Mathf.Abs(delta.y) > Mathf.Abs(delta.x))
918 delta.x = delta.y;
919 delta.y = 0;

923 position += delta * m_ScrollSensitivity;
924 if (m_MovementType == MovementType.Clamped)
925 position += CalculateOffset(position - m_Content.anchoredPosition);

927 SetContentAnchoredPosition(position);
928 UpdateBounds();

1010 protected virtual void LateUpdate()
1011 {
1012 if (!m_Content)
1013 return;
1014
1015 EnsureLayoutHasRebuilt();
1016 UpdateScrollbarVisibility();
1017 UpdateBounds();
1018 float deltaTime = Time.unscaledDeltaTime;
1019 Vector2 offset = CalculateOffset(Vector2.zero);
1020 if (!m_Dragging && (offset != Vector2.zero || m_Velocity != Vector2.zero))
1021 {
1022 Vector2 position = m_Content.anchoredPosition;
1023 for (int axis = 0; axis < 2; axis++)
1024 {
1025 // Apply spring physics if movement is elastic and content has an offset from the view.
1026 if (m_MovementType == MovementType.Elastic && offset[axis] != 0)
1027 {
1028 float speed = m_Velocity[axis];
1029 position[axis] = Mathf.SmoothDamp(m_Content.anchoredPosition[axis], m_Content.anchoredPosition[axis] + offset[axis], ref speed, m_Elasticity, Mathf.Infinity, deltaTime);
1030 m_Velocity[axis] = speed;
1031 }
1032 // Else move content according to velocity with deceleration applied.
1033 else if (m_Inertia)
1034 {
1035 m_Velocity[axis] *= Mathf.Pow(m_DecelerationRate, deltaTime);
1036 if (Mathf.Abs(m_Velocity[axis]) < 1)
1037 m_Velocity[axis] = 0;
1038 position[axis] += m_Velocity[axis] * deltaTime;
1039 }
1040 // If we have neither elaticity or friction, there shouldn't be any velocity.
1041 else
1042 {
1043 m_Velocity[axis] = 0;
1044 }
1045 }
1046
1047 if (m_Velocity != Vector2.zero)
1048 {
1049 if (m_MovementType == MovementType.Clamped)
1050 {
1051 offset = CalculateOffset(position - m_Content.anchoredPosition);
1052 position += offset;
1053 }
1054
1055 SetContentAnchoredPosition(position);
1056 }
1057 }
1058
1059 if (m_Dragging && m_Inertia)
1060 {
1061 Vector3 newVelocity = (m_Content.anchoredPosition - m_PrevPosition) / deltaTime;
1062 m_Velocity = Vector3.Lerp(m_Velocity, newVelocity, deltaTime * 10);
1063 }
1064
1065 if (m_ViewBounds != m_PrevViewBounds || m_ContentBounds != m_PrevContentBounds || m_Content.anchoredPosition != m_PrevPosition)
1066 {
1067 UpdateScrollbars(offset);
1068 m_OnValueChanged.Invoke(normalizedPosition);
1069 UpdatePrevData();
1070 }
1071 }

1217 {

1220 return;

1249 m_Velocity[axis] = 0;
1250 UpdateBounds(true);

1399 private void UpdateBounds(bool updateItems = false)
1400 {
1401 m_ViewBounds = new Bounds(viewRect.rect.center, viewRect.rect.size);
1402 m_ContentBounds = GetBounds();
1403
1404 if (m_Content == null)
1405 return;
1406
1407 // ============LoopScrollRect============
1408 // Don't do this in Rebuild
1409 if (Application.isPlaying && updateItems && UpdateItems(m_ViewBounds, m_ContentBounds))
1410 {
1411 Canvas.ForceUpdateCanvases();
1412 m_ContentBounds = GetBounds();
1413 }
1414 // ============LoopScrollRect============
1415
1416 // Make sure content bounds are at least as large as view by adding padding if not.
1417 // One might think at first that if the content is smaller than the view, scrolling should be allowed.
1418 // However, that's not how scroll views normally work.
1419 // Scrolling is *only* possible when content is *larger* than view.
1420 // We use the pivot of the content rect to decide in which directions the content bounds should be expanded.
1421 // E.g. if pivot is at top, bounds are expanded downwards.
1422 // This also works nicely when ContentSizeFitter is used on the content.
1423 Vector3 contentSize = m_ContentBounds.size;
1424 Vector3 contentPos = m_ContentBounds.center;
1425 Vector3 excess = m_ViewBounds.size - contentSize;
1426 if (excess.x > 0)
1427 {
1428 contentPos.x -= excess.x * (m_Content.pivot.x - 0.5f);
1429 contentSize.x = m_ViewBounds.size.x;
1430 }
1431 if (excess.y > 0)
1432 {
1433 contentPos.y -= excess.y * (m_Content.pivot.y - 0.5f);
1434 contentSize.y = m_ViewBounds.size.y;
1435 }
1436
1437 m_ContentBounds.size = contentSize;
1438 m_ContentBounds.center = contentPos;
1439 }

1442 private Bounds GetBounds()
1443 {
1444 if (m_Content == null)
1445 return new Bounds();

1448 var vMax = new Vector3(float.MinValue, float.MinValue, float.MinValue);
1449

1451 m_Content.GetWorldCorners(m_Corners);

1453 {

1456 vMax = Vector3.Max(v, vMax);
1457 }
1458
1459 var bounds = new Bounds(vMin, Vector3.zero);
1460 bounds.Encapsulate(vMax);
1461 return bounds;
