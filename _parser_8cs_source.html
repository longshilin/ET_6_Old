47 public Parser(TokenList tokens)
48 {
49 this.tokens = tokens;
50 currPos = SkipComments(0);
51 }
52
53 int SkipComments(int pos)
54 {
55 while (pos < tokens.Count && tokens[pos].type == TokenType.Comment)
56 {
57 pos++;
58 }
59 return pos;
60 }
61
62 // returns new position
63 int IncInternal(int pos)
64 {
65 if (pos >= tokens.Count)
66 return pos;
67 pos++;
68
69 return SkipComments(pos);
70 }
71
72 // Increments current pointer if not past the end, returns previous pos
73 int Inc()
74 {
75 int prev = currPos;
76 currPos = IncInternal(currPos);
77 return prev;
78 }
79
80 // Returns the token type of the current token
81 TokenType Tok()
82 {
83 if (currPos >= tokens.Count)
84 return TokenType.EOF;
85 return tokens[currPos].type;
86 }

88 void SkipIf(TokenType type)
89 {
90 if (Tok() == type)
91 Inc();
92 }

95 {
96 return "Invalid PBX project (parsing line " + tokens[currPos].line + ")";
97 }

99 public IdentifierAST ParseIdentifier()
100 {
101 if (Tok() != TokenType.String && Tok() != TokenType.QuotedString)
102 throw new Exception(GetErrorMsg());

105 return ast;
106 }

108 public TreeAST ParseTree()
109 {
110 if (Tok() != TokenType.LBrace)
111 throw new Exception(GetErrorMsg());
112 Inc();
113
114 var ast = new TreeAST();
115 while (Tok() != TokenType.RBrace && Tok() != TokenType.EOF)
116 {
117 ast.values.Add(ParseKeyValue());
118 }
119 SkipIf(TokenType.RBrace);
120 return ast;
121 }
122
123 public ArrayAST ParseList()
124 {
125 if (Tok() != TokenType.LParen)
126 throw new Exception(GetErrorMsg());
127 Inc();
128
129 var ast = new ArrayAST();
130 while (Tok() != TokenType.RParen && Tok() != TokenType.EOF)
131 {
132 ast.values.Add(ParseValue());
133 SkipIf(TokenType.Comma);
134 }
135 SkipIf(TokenType.RParen);
136 return ast;
137 }
138
139 // throws on error
140 public KeyValueAST ParseKeyValue()
141 {
142 var ast = new KeyValueAST();
143 ast.key = ParseIdentifier();
144
145 if (Tok() != TokenType.Eq)
146 throw new Exception(GetErrorMsg());
147 Inc(); // skip '='
148
149 ast.value = ParseValue();
150 SkipIf(TokenType.Semicolon);
151
152 return ast;
153 }
154
155 // throws on error
156 public ValueAST ParseValue()
157 {
158 if (Tok() == TokenType.String || Tok() == TokenType.QuotedString)
159 return ParseIdentifier();
160 else if (Tok() == TokenType.LBrace)
161 return ParseTree();
162 else if (Tok() == TokenType.LParen)
163 return ParseList();
164 throw new Exception(GetErrorMsg());
165 }
