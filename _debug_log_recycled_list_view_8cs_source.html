101 private void OnLogItemClickedInternal( int itemIndex, DebugLogItem referenceItem = null )
102 {
103 if( indexOfSelectedLogEntry != itemIndex )
104 {
105 DeselectSelectedLogItem();
106
107 if( !referenceItem )
108 {
109 if( currentTopIndex == -1 )
110 UpdateItemsInTheList( false ); // Try to generate some DebugLogItems, we need one DebugLogItem to calculate the text height
111
112 referenceItem = logItemsAtIndices[currentTopIndex];
113 }
114
115 indexOfSelectedLogEntry = itemIndex;
116 positionOfSelectedLogEntry = itemIndex * logItemHeight;
117 heightOfSelectedLogEntry = referenceItem.CalculateExpandedHeight( collapsedLogEntries[indicesOfEntriesToShow[itemIndex]], ( timestampsOfEntriesToShow != null ) ? timestampsOfEntriesToShow[itemIndex] : (DebugLogEntryTimestamp?) null );
118 deltaHeightOfSelectedLogEntry = heightOfSelectedLogEntry - logItemHeight;
119
120 manager.SetSnapToBottom( false );
121 }
122 else
123 DeselectSelectedLogItem();
124
125 if( indexOfSelectedLogEntry >= currentTopIndex && indexOfSelectedLogEntry <= currentBottomIndex )
126 ColorLogItem( logItemsAtIndices[indexOfSelectedLogEntry], indexOfSelectedLogEntry );
127
128 CalculateContentHeight();
129
130 HardResetItems();
131 UpdateItemsInTheList( true );
132
133 manager.ValidateScrollPosition();
134 }

222 // Calculate the indices of log entries to show
223 // and handle log items accordingly
224 public void UpdateItemsInTheList( bool updateAllVisibleItemContents )
225 {
226 // If there is at least one log entry to show
227 if( indicesOfEntriesToShow.Count > 0 )
228 {
229 float contentPosTop = transformComponent.anchoredPosition.y - 1f;
230 float contentPosBottom = contentPosTop + viewportHeight + 2f;
231
232 if( positionOfSelectedLogEntry <= contentPosBottom )
233 {
234 if( positionOfSelectedLogEntry <= contentPosTop )
235 {
236 contentPosTop -= deltaHeightOfSelectedLogEntry;
237 contentPosBottom -= deltaHeightOfSelectedLogEntry;
238
239 if( contentPosTop < positionOfSelectedLogEntry - 1f )
240 contentPosTop = positionOfSelectedLogEntry - 1f;
241
242 if( contentPosBottom < contentPosTop + 2f )
243 contentPosBottom = contentPosTop + 2f;
244 }
245 else
246 {
247 contentPosBottom -= deltaHeightOfSelectedLogEntry;
248 if( contentPosBottom < positionOfSelectedLogEntry + 1f )
249 contentPosBottom = positionOfSelectedLogEntry + 1f;
250 }
251 }
252
253 int newTopIndex = (int) ( contentPosTop * _1OverLogItemHeight );
254 int newBottomIndex = (int) ( contentPosBottom * _1OverLogItemHeight );
255
256 if( newTopIndex < 0 )
257 newTopIndex = 0;
258
259 if( newBottomIndex > indicesOfEntriesToShow.Count - 1 )
260 newBottomIndex = indicesOfEntriesToShow.Count - 1;
261
262 if( currentTopIndex == -1 )
263 {
264 // There are no log items visible on screen,
265 // just create the new log items
266 updateAllVisibleItemContents = true;
267
268 currentTopIndex = newTopIndex;
269 currentBottomIndex = newBottomIndex;
270
271 CreateLogItemsBetweenIndices( newTopIndex, newBottomIndex );
272 }
273 else
274 {
275 // There are some log items visible on screen
276
277 if( newBottomIndex < currentTopIndex || newTopIndex > currentBottomIndex )
278 {
279 // If user scrolled a lot such that, none of the log items are now within
280 // the bounds of the scroll view, pool all the previous log items and create
281 // new log items for the new list of visible debug entries
282 updateAllVisibleItemContents = true;
283
284 DestroyLogItemsBetweenIndices( currentTopIndex, currentBottomIndex );
285 CreateLogItemsBetweenIndices( newTopIndex, newBottomIndex );
286 }
287 else
288 {
289 // User did not scroll a lot such that, there are still some log items within
290 // the bounds of the scroll view. Don't destroy them but update their content,
291 // if necessary
292 if( newTopIndex > currentTopIndex )
293 DestroyLogItemsBetweenIndices( currentTopIndex, newTopIndex - 1 );
294
295 if( newBottomIndex < currentBottomIndex )
296 DestroyLogItemsBetweenIndices( newBottomIndex + 1, currentBottomIndex );
297
298 if( newTopIndex < currentTopIndex )
299 {
300 CreateLogItemsBetweenIndices( newTopIndex, currentTopIndex - 1 );
301
302 // If it is not necessary to update all the log items,
303 // then just update the newly created log items. Otherwise,
304 // wait for the major update
305 if( !updateAllVisibleItemContents )
306 UpdateLogItemContentsBetweenIndices( newTopIndex, currentTopIndex - 1 );
307 }
308
309 if( newBottomIndex > currentBottomIndex )
310 {
311 CreateLogItemsBetweenIndices( currentBottomIndex + 1, newBottomIndex );
312
313 // If it is not necessary to update all the log items,
314 // then just update the newly created log items. Otherwise,
315 // wait for the major update
316 if( !updateAllVisibleItemContents )
317 UpdateLogItemContentsBetweenIndices( currentBottomIndex + 1, newBottomIndex );
318 }
319 }
320
321 currentTopIndex = newTopIndex;
322 currentBottomIndex = newBottomIndex;
323 }
324
325 if( updateAllVisibleItemContents )
326 {
327 // Update all the log items
328 UpdateLogItemContentsBetweenIndices( currentTopIndex, currentBottomIndex );
329 }
330 }
331 else
332 HardResetItems();
333 }

335 private void CreateLogItemsBetweenIndices( int topIndex, int bottomIndex )
336 {
337 for( int i = topIndex; i <= bottomIndex; i++ )

339 }

341 // Create (or unpool) a log item
342 private void CreateLogItemAtIndex( int index )
343 {
344 DebugLogItem logItem = manager.PopLogItem();
345
346 // Reposition the log item
347 Vector2 anchoredPosition = new Vector2( 1f, -index * logItemHeight );
348 if( index > indexOfSelectedLogEntry )
349 anchoredPosition.y -= deltaHeightOfSelectedLogEntry;
350
351 logItem.Transform.anchoredPosition = anchoredPosition;
352
353 // Color the log item
354 ColorLogItem( logItem, index );
355
356 // To access this log item easily in the future, add it to the dictionary
357 logItemsAtIndices[index] = logItem;
358 }
359
360 private void DestroyLogItemsBetweenIndices( int topIndex, int bottomIndex )
361 {
362 for( int i = topIndex; i <= bottomIndex; i++ )
363 manager.PoolLogItem( logItemsAtIndices[i] );
364 }
365
366 private void UpdateLogItemContentsBetweenIndices( int topIndex, int bottomIndex )
367 {
368 DebugLogItem logItem;
369 for( int i = topIndex; i <= bottomIndex; i++ )
370 {
371 logItem = logItemsAtIndices[i];
372 logItem.SetContent( collapsedLogEntries[indicesOfEntriesToShow[i]], ( timestampsOfEntriesToShow != null ) ? timestampsOfEntriesToShow[i] : (DebugLogEntryTimestamp?) null, i, i == indexOfSelectedLogEntry );
373
374 if( isCollapseOn )
375 logItem.ShowCount();
376 else
377 logItem.HideCount();
378 }
379 }
380
381 // Color a log item using its index
382 private void ColorLogItem( DebugLogItem logItem, int index )
383 {
384 if( index == indexOfSelectedLogEntry )
385 logItem.Image.color = logItemSelectedColor;
386 else if( index % 2 == 0 )
387 logItem.Image.color = logItemNormalColor1;
388 else
389 logItem.Image.color = logItemNormalColor2;
390 }
