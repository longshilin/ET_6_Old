218 public static void Project (TypeDefinition type)
219 {
220 var treatment = TypeDefinitionTreatment.None;
221 var metadata_kind = type.Module.MetadataKind;
222 Collection<MethodDefinition> redirectedMethods = null;
223 Collection<KeyValuePair<InterfaceImplementation, InterfaceImplementation>> redirectedInterfaces = null;
224
225 if (type.IsWindowsRuntime) {
226 if (metadata_kind == MetadataKind.WindowsMetadata) {
227 treatment = GetWellKnownTypeDefinitionTreatment (type);
228 if (treatment != TypeDefinitionTreatment.None) {
229 ApplyProjection (type, new TypeDefinitionProjection (type, treatment, redirectedMethods, redirectedInterfaces));
230 return;
231 }
232
233 var base_type = type.BaseType;
234 if (base_type != null && IsAttribute (base_type)) {
235 treatment = TypeDefinitionTreatment.NormalAttribute;
236 } else {
237 treatment = GenerateRedirectionInformation (type, out redirectedMethods, out redirectedInterfaces);
238 }
239 }
240 else if (metadata_kind == MetadataKind.ManagedWindowsMetadata && NeedsWindowsRuntimePrefix (type))
241 treatment = TypeDefinitionTreatment.PrefixWindowsRuntimeName;
242
243 if (treatment == TypeDefinitionTreatment.PrefixWindowsRuntimeName || treatment == TypeDefinitionTreatment.NormalType)
244 if (!type.IsInterface && HasAttribute (type, "Windows.UI.Xaml", "TreatAsAbstractComposableClassAttribute"))
245 treatment |= TypeDefinitionTreatment.Abstract;
246 }
247 else if (metadata_kind == MetadataKind.ManagedWindowsMetadata && IsClrImplementationType (type))
248 treatment = TypeDefinitionTreatment.UnmangleWindowsRuntimeName;
249
250 if (treatment != TypeDefinitionTreatment.None)
251 ApplyProjection (type, new TypeDefinitionProjection (type, treatment, redirectedMethods, redirectedInterfaces));
252 }
253
254 static TypeDefinitionTreatment GetWellKnownTypeDefinitionTreatment (TypeDefinition type)
255 {
256 ProjectionInfo info;
257 if (!Projections.TryGetValue (type.Name, out info))
258 return TypeDefinitionTreatment.None;
259
260 var treatment = info.Attribute ? TypeDefinitionTreatment.RedirectToClrAttribute : TypeDefinitionTreatment.RedirectToClrType;
261
262 if (type.Namespace == info.ClrNamespace)
263 return treatment;
264
265 if (type.Namespace == info.WinRTNamespace)
266 return treatment | TypeDefinitionTreatment.Internal;
267
268 return TypeDefinitionTreatment.None;
269 }

271 private static TypeDefinitionTreatment GenerateRedirectionInformation (TypeDefinition type, out Collection<MethodDefinition> redirectedMethods, out Collection<KeyValuePair<InterfaceImplementation, InterfaceImplementation>> redirectedInterfaces)
272 {
273 bool implementsProjectedInterface = false;
274 redirectedMethods = null;
275 redirectedInterfaces = null;
276
277 foreach (var implementedInterface in type.Interfaces) {
278 if (IsRedirectedType (implementedInterface.InterfaceType)) {
279 implementsProjectedInterface = true;
280 break;
281 }
282 }
283
284 if (!implementsProjectedInterface)
285 return TypeDefinitionTreatment.NormalType;
286
287 var allImplementedInterfaces = new HashSet<TypeReference> (new TypeReferenceEqualityComparer ());
288 redirectedMethods = new Collection<MethodDefinition> ();
289 redirectedInterfaces = new Collection<KeyValuePair<InterfaceImplementation, InterfaceImplementation>> ();
290
291 foreach (var @interface in type.Interfaces) {
292 var interfaceType = @interface.InterfaceType;
293
294 if (IsRedirectedType (interfaceType)) {
295 allImplementedInterfaces.Add (interfaceType);
296 CollectImplementedInterfaces (interfaceType, allImplementedInterfaces);
297 }
298 }
299
300 foreach (var implementedInterface in type.Interfaces) {
301 var interfaceType = implementedInterface.InterfaceType;
302 if (IsRedirectedType (implementedInterface.InterfaceType)) {
303 var etype = interfaceType.GetElementType ();
304 var unprojectedType = new TypeReference (etype.Namespace, etype.Name, etype.Module, etype.Scope) {
305 DeclaringType = etype.DeclaringType,
306 projection = etype.projection
307 };
308
309 RemoveProjection (unprojectedType);
310
311 var genericInstanceType = interfaceType as GenericInstanceType;
312 if (genericInstanceType != null) {
313 var genericUnprojectedType = new GenericInstanceType (unprojectedType);
314 foreach (var genericArgument in genericInstanceType.GenericArguments)
315 genericUnprojectedType.GenericArguments.Add (genericArgument);
316
317 unprojectedType = genericUnprojectedType;
318 }
319
320 var unprojectedInterface = new InterfaceImplementation (unprojectedType);
321 redirectedInterfaces.Add (new KeyValuePair<InterfaceImplementation, InterfaceImplementation> (implementedInterface, unprojectedInterface));
322 }
323 }
324
325 // Interfaces don't inherit methods of the interfaces they implement
326 if (!type.IsInterface) {
327 foreach (var implementedInterface in allImplementedInterfaces) {
328 RedirectInterfaceMethods (implementedInterface, redirectedMethods);
329 }
330 }
331
332 return TypeDefinitionTreatment.RedirectImplementedMethods;
333 }
334
335 private static void CollectImplementedInterfaces (TypeReference type, HashSet<TypeReference> results)
336 {
337 var typeResolver = TypeResolver.For (type);
338 var typeDef = type.Resolve ();
339
340 foreach (var implementedInterface in typeDef.Interfaces) {
341 var interfaceType = typeResolver.Resolve (implementedInterface.InterfaceType);
342 results.Add (interfaceType);
343 CollectImplementedInterfaces (interfaceType, results);
344 }
345 }
346
347 private static void RedirectInterfaceMethods (TypeReference interfaceType, Collection<MethodDefinition> redirectedMethods)
348 {
349 var typeResolver = TypeResolver.For (interfaceType);
350 var typeDef = interfaceType.Resolve ();
351
352 foreach (var method in typeDef.Methods) {
353 var redirectedMethod = new MethodDefinition (method.Name, MethodAttributes.Public | MethodAttributes.Virtual | MethodAttributes.Final | MethodAttributes.NewSlot, typeResolver.Resolve (method.ReturnType));
354 redirectedMethod.ImplAttributes = MethodImplAttributes.Runtime;
355
356 foreach (var parameter in method.Parameters) {
357 redirectedMethod.Parameters.Add (new ParameterDefinition (parameter.Name, parameter.Attributes, typeResolver.Resolve (parameter.ParameterType)));
358 }
359
360 redirectedMethod.Overrides.Add (typeResolver.Resolve (method));
361 redirectedMethods.Add (redirectedMethod);
362 }
363 }

371 static bool NeedsWindowsRuntimePrefix (TypeDefinition type)
372 {
373 if ((type.Attributes & (TypeAttributes.VisibilityMask | TypeAttributes.Interface)) != TypeAttributes.Public)
374 return false;
375
376 var base_type = type.BaseType;
377 if (base_type == null || base_type.MetadataToken.TokenType != TokenType.TypeRef)
378 return false;
379
380 if (base_type.Namespace == "System")
381 switch (base_type.Name)
382 {
383 case "Attribute":
384 case "MulticastDelegate":
385 case "ValueType":
386 return false;
387 }
388
389 return true;
390 }

392 static bool IsClrImplementationType (TypeDefinition type)
393 {
394 if ((type.Attributes & (TypeAttributes.VisibilityMask | TypeAttributes.SpecialName)) != TypeAttributes.SpecialName)
395 return false;
396 return type.Name.StartsWith ("<CLR>");

399 public static void ApplyProjection (TypeDefinition type, TypeDefinitionProjection projection)
400 {
401 if (projection == null)
402 return;
403
404 var treatment = projection.Treatment;
405
406 switch (treatment & TypeDefinitionTreatment.KindMask) {
407 case TypeDefinitionTreatment.NormalType:
408 type.Attributes |= TypeAttributes.WindowsRuntime | TypeAttributes.Import;
409 break;
410
411 case TypeDefinitionTreatment.NormalAttribute:
412 type.Attributes |= TypeAttributes.WindowsRuntime | TypeAttributes.Sealed;
413 break;
414
415 case TypeDefinitionTreatment.UnmangleWindowsRuntimeName:
416 type.Attributes = type.Attributes & ~TypeAttributes.SpecialName | TypeAttributes.Public;
417 type.Name = type.Name.Substring ("<CLR>".Length);
418 break;
419
420 case TypeDefinitionTreatment.PrefixWindowsRuntimeName:
421 type.Attributes = type.Attributes & ~TypeAttributes.Public | TypeAttributes.Import;
422 type.Name = "<WinRT>" + type.Name;
423 break;
424
425 case TypeDefinitionTreatment.RedirectToClrType:
426 type.Attributes = type.Attributes & ~TypeAttributes.Public | TypeAttributes.Import;
427 break;
428
429 case TypeDefinitionTreatment.RedirectToClrAttribute:
430 type.Attributes = type.Attributes & ~TypeAttributes.Public;
431 break;
432
433 case TypeDefinitionTreatment.RedirectImplementedMethods: {
434 type.Attributes |= TypeAttributes.WindowsRuntime | TypeAttributes.Import;
435
436 foreach (var redirectedInterfacePair in projection.RedirectedInterfaces) {
437 type.Interfaces.Add (redirectedInterfacePair.Value);
438
439 foreach (var customAttribute in redirectedInterfacePair.Key.CustomAttributes)
440 redirectedInterfacePair.Value.CustomAttributes.Add (customAttribute);
441
442 redirectedInterfacePair.Key.CustomAttributes.Clear ();
443
444 foreach (var method in type.Methods) {
445 foreach (var @override in method.Overrides) {
446 if (TypeReferenceEqualityComparer.AreEqual (@override.DeclaringType, redirectedInterfacePair.Key.InterfaceType)) {
447 @override.DeclaringType = redirectedInterfacePair.Value.InterfaceType;
448 }
449 }
450 }
451 }
452
453 foreach (var method in projection.RedirectedMethods) {
454 type.Methods.Add (method);
455 }
456 }
457 break;
458 }
459
460 if ((treatment & TypeDefinitionTreatment.Abstract) != 0)
461 type.Attributes |= TypeAttributes.Abstract;
462
463 if ((treatment & TypeDefinitionTreatment.Internal) != 0)
464 type.Attributes &= ~TypeAttributes.Public;
465
466 type.WindowsRuntimeProjection = projection;
467 }

469 public static TypeDefinitionProjection RemoveProjection (TypeDefinition type)
470 {
471 if (!type.IsWindowsRuntimeProjection)
472 return null;
473
474 var projection = type.WindowsRuntimeProjection;
475 type.WindowsRuntimeProjection = null;
476
477 type.Attributes = projection.Attributes;
478 type.Name = projection.Name;
479
480 if (projection.Treatment == TypeDefinitionTreatment.RedirectImplementedMethods) {
481 foreach (var method in projection.RedirectedMethods) {
482 type.Methods.Remove (method);
483 }
484
485 foreach (var redirectedInterfacePair in projection.RedirectedInterfaces) {
486 foreach (var method in type.Methods) {
487 foreach (var @override in method.Overrides) {
488 if (TypeReferenceEqualityComparer.AreEqual (@override.DeclaringType, redirectedInterfacePair.Value.InterfaceType)) {
489 @override.DeclaringType = redirectedInterfacePair.Key.InterfaceType;
490 }
491 }
492 }
493
494 foreach (var customAttribute in redirectedInterfacePair.Value.CustomAttributes)
495 redirectedInterfacePair.Key.CustomAttributes.Add (customAttribute);
496
497 redirectedInterfacePair.Value.CustomAttributes.Clear ();
498 type.Interfaces.Remove (redirectedInterfacePair.Value);
499 }
500 }
501
502 return projection;
503 }

545 public static void ApplyProjection (TypeReference type, TypeReferenceProjection projection)
546 {
547 if (projection == null)
548 return;
549
550 switch (projection.Treatment)
551 {
552 case TypeReferenceTreatment.SystemDelegate:
553 case TypeReferenceTreatment.SystemAttribute:
554 type.Scope = type.Module.Projections.GetAssemblyReference ("System.Runtime");
555 break;
556
557 case TypeReferenceTreatment.UseProjectionInfo:
558 var info = Projections [type.Name];
559 type.Name = info.ClrName;
560 type.Namespace = info.ClrNamespace;
561 type.Scope = type.Module.Projections.GetAssemblyReference (info.ClrAssembly);
562 break;
563 }
564
565 type.WindowsRuntimeProjection = projection;
566 }

568 public static TypeReferenceProjection RemoveProjection (TypeReference type)
569 {
570 if (!type.IsWindowsRuntimeProjection)
571 return null;
572
573 var projection = type.WindowsRuntimeProjection;
574 type.WindowsRuntimeProjection = null;
575
576 type.Name = projection.Name;
577 type.Namespace = projection.Namespace;
578 type.Scope = projection.Scope;
579
580 return projection;
581 }

583 public static void Project (MethodDefinition method)
584 {
585 var treatment = MethodDefinitionTreatment.None;
586 var other = false;
587 var declaring_type = method.DeclaringType;
588
589 if (declaring_type.IsWindowsRuntime)
590 {
591 if (IsClrImplementationType (declaring_type))
592 treatment = MethodDefinitionTreatment.None;
593 else if (declaring_type.IsNested)
594 treatment = MethodDefinitionTreatment.None;
595 else if (declaring_type.IsInterface)
596 treatment = MethodDefinitionTreatment.Runtime | MethodDefinitionTreatment.InternalCall;
597 else if (declaring_type.Module.MetadataKind == MetadataKind.ManagedWindowsMetadata && !method.IsPublic)
598 treatment = MethodDefinitionTreatment.None;
599 else
600 {
601 other = true;
602
603 var base_type = declaring_type.BaseType;
604 if (base_type != null && base_type.MetadataToken.TokenType == TokenType.TypeRef)
605 {
606 switch (GetSpecialTypeReferenceTreatment(base_type))
607 {
608 case TypeReferenceTreatment.SystemDelegate:
609 treatment = MethodDefinitionTreatment.Runtime | MethodDefinitionTreatment.Public;
610 other = false;
611 break;
612
613 case TypeReferenceTreatment.SystemAttribute:
614 treatment = MethodDefinitionTreatment.Runtime | MethodDefinitionTreatment.InternalCall;
615 other = false;
616 break;
617 }
618 }
619 }
620 }
621
622 if (other)
623 {
624 var seen_redirected = false;
625 var seen_non_redirected = false;
626
627 foreach (var @override in method.Overrides) {
628 if (@override.MetadataToken.TokenType == TokenType.MemberRef && ImplementsRedirectedInterface (@override)) {
629 seen_redirected = true;
630 } else {
631 seen_non_redirected = true;
632 }
633 }
634
635 if (seen_redirected && !seen_non_redirected) {
636 treatment = MethodDefinitionTreatment.Runtime | MethodDefinitionTreatment.InternalCall | MethodDefinitionTreatment.Private;
637 other = false;
638 }
639 }
640
641 if (other)
642 treatment |= GetMethodDefinitionTreatmentFromCustomAttributes(method);
643
644 if (treatment != MethodDefinitionTreatment.None)
645 ApplyProjection (method, new MethodDefinitionProjection (method, treatment));
646 }

745 static bool ImplementsRedirectedInterface (MemberReference member)
746 {
747 var declaring_type = member.DeclaringType;
748 TypeReference type;
749 switch (declaring_type.MetadataToken.TokenType) {
750 case TokenType.TypeRef:
751 type = declaring_type;
752 break;
753
754 case TokenType.TypeSpec:
755 if (!declaring_type.IsGenericInstance)
756 return false;
757
758 type = ((TypeSpecification) declaring_type).ElementType;
759 if (type.MetadataType != MetadataType.Class || type.MetadataToken.TokenType != TokenType.TypeRef)
760 return false;
761
762 break;
763
764 default:
765 return false;
766 }

768 var projection = RemoveProjection (type);
769
770 var found = false;

772 ProjectionInfo info;
773 if (Projections.TryGetValue (type.Name, out info) && type.Namespace == info.WinRTNamespace) {
774 found = true;
775 }
776
777 ApplyProjection (type, projection);
778
779 return found;
