106 public static void SetInfoPlist(string buildPath, XcodeProjectSetting setting)
107 {
108 PlistDocument plist = GetInfoPlist(buildPath);
109 SetPrivacySensiticeData(plist, setting.privacySensiticeData);
110 SetApplicationQueriesSchemes(plist, setting.ApplicationQueriesSchemes);
