24 //过滤.meta文件

28 //

99 public static void AddLocalizedStrings(PBXProject pbxProject, string buildPath, string fileName, string localizedDirectoryPath, string directoryName)
100 {
101 DirectoryInfo dir = new DirectoryInfo(localizedDirectoryPath);
102 if (!dir.Exists)
103 return;
104
105 List<string> locales = new List<string>();
106 var localeDirs = dir.GetDirectories("*.lproj", SearchOption.TopDirectoryOnly);
107
108 foreach (var sub in localeDirs)
109 locales.Add(Path.GetFileNameWithoutExtension(sub.Name));
110
111 AddLocalizedStrings(pbxProject, buildPath, directoryName, fileName, localizedDirectoryPath, locales);
112 }

125 public static void CopyAndReplace(string sourcePath, string copyPath)
126 {
127 Delete(copyPath);
128 Directory.CreateDirectory(copyPath);
129 foreach (var file in Directory.GetFiles(sourcePath))
130 {
131 if (Path.GetExtension(file) == ExtensionName.META)
132 continue;
133 File.Copy(file, Path.Combine(copyPath, Path.GetFileName(file)));
134 }
135
136 foreach (var dir in Directory.GetDirectories(sourcePath))
137 {
138 CopyAndReplace(dir, Path.Combine(copyPath, Path.GetFileName(dir)));
139 }
140 }

142 //删除目标文件夹以及文件夹内的所有文件
143 public static void Delete(string targetDirectoryPath)
144 {
145 if (!Directory.Exists(targetDirectoryPath))
146 return;
147 string[] filePaths = Directory.GetFiles(targetDirectoryPath);
148 foreach (string filePath in filePaths)
149 {
150 File.SetAttributes(filePath, FileAttributes.Normal);
151 File.Delete(filePath);
152 }
153
154 string[] directoryPaths = Directory.GetDirectories(targetDirectoryPath);
155 foreach (string directoryPath in directoryPaths)
156 {
157 Delete(directoryPath);
158 }
159
160 Directory.Delete(targetDirectoryPath, false);
161 }
