32 public GameObjectPool(string poolName, GameObject poolObjectPrefab, GameObject rootPoolObj, int initialCount, PoolInflationType type)
33 {
34 if (poolObjectPrefab == null)
35 {
36#if UNITY_EDITOR
37 Debug.LogError("[ObjPoolManager] null pool object prefab !");
38#endif
39 return;
40 }
41 this.poolName = poolName;
42 this.inflationType = type;
43 this.rootObj = new GameObject(poolName + "Pool");
44 this.rootObj.transform.SetParent(rootPoolObj.transform, false);
45
46 // In case the origin one is Destroyed, we should keep at least one
47 GameObject go = GameObject.Instantiate(poolObjectPrefab);
48 PoolObject po = go.GetComponent<PoolObject>();
49 if (po == null)
50 {
51 po = go.AddComponent<PoolObject>();
52 }
53 po.poolName = poolName;
54 AddObjectToPool(po);
55
56 //populate the pool
57 populatePool(Mathf.Max(initialCount, 1));
58 }

60 //o(1)
61 private void AddObjectToPool(PoolObject po)
62 {
63 //add to pool
64 po.gameObject.SetActive(false);
65 po.gameObject.name = poolName;
66 availableObjStack.Push(po);
67 po.isPooled = true;
68 //add to a root obj
69 po.gameObject.transform.SetParent(rootObj.transform, false);
70 }
71
72 private void populatePool(int initialCount)
73 {
74 for (int index = 0; index < initialCount; index++)
75 {
76 PoolObject po = GameObject.Instantiate(availableObjStack.Peek());
77 AddObjectToPool(po);
78
79 }
80 }

83 public GameObject NextAvailableObject(bool autoActive)
84 {
85 PoolObject po = null;
86 if (availableObjStack.Count > 1)
87 {
88 po = availableObjStack.Pop();
89 }
90 else
91 {
92 int increaseSize = 0;
93 //increment size var, this is for info purpose only
94 if (inflationType == PoolInflationType.INCREMENT)
95 {
96 increaseSize = 1;
97 }
98 else if (inflationType == PoolInflationType.DOUBLE)
99 {
100 increaseSize = availableObjStack.Count + Mathf.Max(objectsInUse, 0);
101 }
102#if UNITY_EDITOR
103 Debug.Log(string.Format("Growing pool {0}: {1} populated", poolName, increaseSize));
104#endif
105 if (increaseSize > 0)
106 {
107 populatePool(increaseSize);
108 po = availableObjStack.Pop();
109 }
110 }
111
112 GameObject result = null;
113 if (po != null)
114 {
115 objectsInUse++;
116 po.isPooled = false;
117 result = po.gameObject;
118 if (autoActive)
119 {
120 result.SetActive(true);
121 }
122 }
123
124 return result;
125 }

129 //o(1)
130 public void ReturnObjectToPool(PoolObject po)
131 {
132 if (poolName.Equals(po.poolName))
133 {
134 objectsInUse--;
135 /* we could have used availableObjStack.Contains(po) to check if this object is in pool.
136 * While that would have been more robust, it would have made this method O(n)
137 */
138 if (po.isPooled)
139 {
140#if UNITY_EDITOR
141 Debug.LogWarning(po.gameObject.name + " is already in pool. Why are you trying to return it again? Check usage.");
142#endif
143 }
144 else
145 {
146 AddObjectToPool(po);
147 }
148 }
149 else
150 {
151 Debug.LogError(string.Format("Trying to add object to incorrect pool {0} {1}", po.poolName, poolName));
152 }
153 }
154 }
