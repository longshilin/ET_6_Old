105 static ParameterDefinition CreateThisParameter (MethodDefinition method)
106 {
107 var parameter_type = method.DeclaringType as TypeReference;
108
109 if (parameter_type.HasGenericParameters) {
110 var instance = new GenericInstanceType (parameter_type, parameter_type.GenericParameters.Count);
111 for (int i = 0; i < parameter_type.GenericParameters.Count; i++)
112 instance.GenericArguments.Add (parameter_type.GenericParameters [i]);
113
114 parameter_type = instance;
115
116 }
117
118 if (parameter_type.IsValueType || parameter_type.IsPrimitive)
119 parameter_type = new ByReferenceType (parameter_type);
120
121 return new ParameterDefinition (parameter_type, method);
122 }

172 void UpdateVariableIndices (int startIndex, int offset, VariableDefinition variableToRemove = null)
173 {
174 for (int i = startIndex; i < size; i++)
175 items [i].index = i + offset;
176
177 var debug_info = method == null ? null : method.debug_info;
178 if (debug_info == null || debug_info.Scope == null)
179 return;
180
181 foreach (var scope in debug_info.GetScopes ()) {
182 if (!scope.HasVariables)
183 continue;
184
185 var variables = scope.Variables;
186 int variableDebugInfoIndexToRemove = -1;
187 for (int i = 0; i < variables.Count; i++) {
188 var variable = variables [i];
189
190 // If a variable is being removed detect if it has debug info counterpart, if so remove that as well.
191 // Note that the debug info can be either resolved (has direct reference to the VariableDefinition)
192 // or unresolved (has only the number index of the variable) - this needs to handle both cases.
193 if (variableToRemove != null &&
194 ((variable.index.IsResolved && variable.index.ResolvedVariable == variableToRemove) ||
195 (!variable.index.IsResolved && variable.Index == variableToRemove.Index))) {
196 variableDebugInfoIndexToRemove = i;
197 continue;
198 }
199
200 // For unresolved debug info updates indeces to keep them pointing to the same variable.
201 if (!variable.index.IsResolved && variable.Index >= startIndex) {
202 variable.index = new VariableIndex (variable.Index + offset);
203 }
204 }
205
206 if (variableDebugInfoIndexToRemove >= 0)
207 variables.RemoveAt (variableDebugInfoIndexToRemove);
208 }
209 }

294 void RemoveSequencePoint (Instruction instruction)

296 var debug_info = method.debug_info;
297 if (debug_info == null || !debug_info.HasSequencePoints)
298 return;

300 var sequence_points = debug_info.sequence_points;

309 void UpdateLocalScopes (Instruction removedInstruction, Instruction existingInstruction)
310 {
311 var debug_info = method.debug_info;
312 if (debug_info == null)
313 return;
314
315 // Local scopes store start/end pair of "instruction offsets". Instruction offset can be either resolved, in which case it
316 // has a reference to Instruction, or unresolved in which case it stores numerical offset (instruction offset in the body).
317 // Typically local scopes loaded from PE/PDB files will be resolved, but it's not a requirement.
318 // Each instruction has its own offset, which is populated on load, but never updated (this would be pretty expensive to do).
319 // Instructions created during the editting will typically have offset 0 (so incorrect).
320 // Local scopes created during editing will also likely be resolved (so no numerical offsets).
321 // So while local scopes which are unresolved are relatively rare if they appear, manipulating them based
322 // on the offsets allone is pretty hard (since we can't rely on correct offsets of instructions).
323 // On the other hand resolved local scopes are easy to maintain, since they point to instructions and thus inserting
324 // instructions is basically a no-op and removing instructions is as easy as changing the pointer.
325 // For this reason the algorithm here is:
326 // - First make sure that all instruction offsets are resolved - if not - resolve them
327 // - First time this will be relatively expensinve as it will walk the entire method body to convert offsets to instruction pointers
328 // Almost all local scopes are stored in the "right" order (sequentially per start offsets), so the code uses a simple one-item
329 // cache instruction<->offset to avoid walking instructions multiple times (that would only happen for scopes which are out of order).
330 // - Subsequent calls should be cheap as it will only walk all local scopes without doing anything
331 // - If there was an edit on local scope which makes some of them unresolved, the cost is proportional
332 // - Then update as necessary by manipulaitng instruction references alone
333
334 InstructionOffsetCache cache = new InstructionOffsetCache () {
335 Offset = 0,
336 Index = 0,
337 Instruction = items [0]
338 };
339
340 UpdateLocalScope (debug_info.Scope, removedInstruction, existingInstruction, ref cache);
341 }
342
343 void UpdateLocalScope (ScopeDebugInformation scope, Instruction removedInstruction, Instruction existingInstruction, ref InstructionOffsetCache cache)
344 {
345 if (scope == null)
346 return;
347
348 if (!scope.Start.IsResolved)
349 scope.Start = ResolveInstructionOffset (scope.Start, ref cache);
350
351 if (!scope.Start.IsEndOfMethod && scope.Start.ResolvedInstruction == removedInstruction)
352 scope.Start = new InstructionOffset (existingInstruction);
353
354 if (scope.HasScopes) {
355 foreach (var subScope in scope.Scopes)
356 UpdateLocalScope (subScope, removedInstruction, existingInstruction, ref cache);
357 }
358
359 if (!scope.End.IsResolved)
360 scope.End = ResolveInstructionOffset (scope.End, ref cache);
361
362 if (!scope.End.IsEndOfMethod && scope.End.ResolvedInstruction == removedInstruction)
363 scope.End = new InstructionOffset (existingInstruction);
364 }
365
366 struct InstructionOffsetCache {
367 public int Offset;
368 public int Index;
369 public Instruction Instruction;
370 }
371
372 InstructionOffset ResolveInstructionOffset(InstructionOffset inputOffset, ref InstructionOffsetCache cache)
373 {
374 if (inputOffset.IsResolved)
375 return inputOffset;
376
377 int offset = inputOffset.Offset;
378
379 if (cache.Offset == offset)
380 return new InstructionOffset (cache.Instruction);
381
382 if (cache.Offset > offset) {
383 // This should be rare - we're resolving offset pointing to a place before the current cache position
384 // resolve by walking the instructions from start and don't cache the result.
385 int size = 0;
386 for (int i = 0; i < items.Length; i++) {
387 if (size == offset)
388 return new InstructionOffset (items [i]);
389
390 if (size > offset)
391 return new InstructionOffset (items [i - 1]);
392
393 size += items [i].GetSize ();
394 }
395
396 // Offset is larger than the size of the body - so it points after the end
397 return new InstructionOffset ();
398 } else {
399 // The offset points after the current cache position - so continue counting and update the cache
400 int size = cache.Offset;
401 for (int i = cache.Index; i < items.Length; i++) {
402 cache.Index = i;
403 cache.Offset = size;
404 cache.Instruction = items [i];
405
406 if (cache.Offset == offset)
407 return new InstructionOffset (cache.Instruction);
408
409 if (cache.Offset > offset)
410 return new InstructionOffset (items [i - 1]);
411
412 size += items [i].GetSize ();
413 }
414
415 return new InstructionOffset ();
416 }
417 }
418 }
