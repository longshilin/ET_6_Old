150 public async ETTask StartRecv()
151 {
152 if (this.IsDisposed)
153 {
154 return;
155 }
156
157 try
158 {
159 while (true)
160 {
161#if NOT_UNITY
162 ValueWebSocketReceiveResult receiveResult;
163#else
164 WebSocketReceiveResult receiveResult;
165#endif
166 int receiveCount = 0;
167 do
168 {
169#if NOT_UNITY
170 receiveResult = await this.webSocket.ReceiveAsync(
171 new Memory<byte>(cache, receiveCount, this.cache.Length - receiveCount),
172 cancellationTokenSource.Token);
173#else
174 receiveResult = await this.webSocket.ReceiveAsync(
175 new ArraySegment<byte>(this.cache, receiveCount, this.cache.Length - receiveCount),
176 cancellationTokenSource.Token);
177#endif
178 if (this.IsDisposed)
179 {
180 return;
181 }
182
183 receiveCount += receiveResult.Count;
184 }
185 while (!receiveResult.EndOfMessage);
186
187 if (receiveResult.MessageType == WebSocketMessageType.Close)
188 {
189 this.OnError(ErrorCore.ERR_WebsocketPeerReset);
190 return;
191 }
192
193 if (receiveResult.Count > ushort.MaxValue)
194 {
195 await this.webSocket.CloseAsync(WebSocketCloseStatus.MessageTooBig, $"message too big: {receiveCount}",
196 cancellationTokenSource.Token);
197 this.OnError(ErrorCore.ERR_WebsocketMessageTooBig);
198 return;
199 }
200
201 this.recvStream.SetLength(receiveCount);
202 this.recvStream.Seek(2, SeekOrigin.Begin);
203 Array.Copy(this.cache, 0, this.recvStream.GetBuffer(), 0, receiveCount);
204 this.OnRead(this.recvStream);
205 }
206 }
207 catch (Exception e)
208 {
209 Log.Error(e);
210 this.OnError(ErrorCore.ERR_WebsocketRecvError);
211 }
