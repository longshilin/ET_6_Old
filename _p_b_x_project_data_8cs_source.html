193 public FileGUIDListBase BuildSectionAny(PBXNativeTargetData target, string path, bool isFolderRef)
194 {
195 string ext = Path.GetExtension(path);
196 var phase = FileTypeUtils.GetFileType(ext, isFolderRef);
197 switch (phase) {
198 case PBXFileType.Framework:
199 foreach (var guid in target.phases)
200 if (frameworks.HasEntry(guid))
201 return frameworks[guid];
202 break;
203 case PBXFileType.Resource:
204 foreach (var guid in target.phases)
205 if (resources.HasEntry(guid))
206 return resources[guid];
207 break;
208 case PBXFileType.Source:
209 foreach (var guid in target.phases)
210 if (sources.HasEntry(guid))
211 return sources[guid];
212 break;
213 case PBXFileType.CopyFile:
214 foreach (var guid in target.phases)
215 if (copyFiles.HasEntry(guid))
216 return copyFiles[guid];
217 break;
218 }
219 return null;
220 }

245 void RefreshMapsForGroupChildren(string projectPath, string realPath, PBXSourceTree realPathTree, PBXGroupData parent)
246 {
247 var children = new List<string>(parent.children);
248 foreach (string guid in children)
249 {
250 PBXFileReferenceData fileRef = fileRefs[guid];
251 string pPath;
252 string rPath;
253 PBXSourceTree rTree;
254
255 if (fileRef != null)
256 {
257 pPath = PBXPath.Combine(projectPath, fileRef.name);
258 PBXPath.Combine(realPath, realPathTree, fileRef.path, fileRef.tree, out rPath, out rTree);
259
260 if (!m_ProjectPathToFileRefMap.ContainsKey(pPath))
261 {
262 m_ProjectPathToFileRefMap.Add(pPath, fileRef);
263 }
264 if (!m_FileRefGuidToProjectPathMap.ContainsKey(fileRef.guid))
265 {
266 m_FileRefGuidToProjectPathMap.Add(fileRef.guid, pPath);
267 }
268 if (!m_RealPathToFileRefMap[rTree].ContainsKey(rPath))
269 {
270 m_RealPathToFileRefMap[rTree].Add(rPath, fileRef);
271 }
272 if (!m_GuidToParentGroupMap.ContainsKey(guid))
273 {
274 m_GuidToParentGroupMap.Add(guid, parent);
275 }
276
277 continue;
278 }
279
280 PBXGroupData gr = groups[guid];
281 if (gr != null)
282 {
283 pPath = PBXPath.Combine(projectPath, gr.name);
284 PBXPath.Combine(realPath, realPathTree, gr.path, gr.tree, out rPath, out rTree);
285
286 if (!m_ProjectPathToGroupMap.ContainsKey(pPath))
287 {
288 m_ProjectPathToGroupMap.Add(pPath, gr);
289 }
290 if (!m_GroupGuidToProjectPathMap.ContainsKey(gr.guid))
291 {
292 m_GroupGuidToProjectPathMap.Add(gr.guid, pPath);
293 }
294 if (!m_GuidToParentGroupMap.ContainsKey(guid))
295 {
296 m_GuidToParentGroupMap.Add(guid, parent);
297 }
298
299 RefreshMapsForGroupChildren(pPath, rPath, rTree, gr);

474 public void ReadFromStream(TextReader sr)
475 {
476 Clear();
477 m_RootElements = ParseContent(sr.ReadToEnd());
478
479 if (!m_RootElements.Contains("objects"))
480 throw new Exception("Invalid PBX project file: no objects element");
481
482 var objects = m_RootElements["objects"].AsDict();
483 m_RootElements.Remove("objects");
484 m_RootElements.SetString("objects", "OBJMARKER");
485
486 if (m_RootElements.Contains("objectVersion"))
487 {
488 m_ObjectVersion = m_RootElements["objectVersion"].AsString();
489 m_RootElements.Remove("objectVersion");
490 }
491
492 var allGuids = new List<string>();
493 string prevSectionName = null;
494 foreach (var kv in objects.values)
495 {
496 allGuids.Add(kv.Key);
497 var el = kv.Value;
498
499 if (!(el is PBXElementDict) || !el.AsDict().Contains("isa"))
500 {
501 m_UnknownObjects.values.Add(kv.Key, el);
502 continue;
503 }
504 var dict = el.AsDict();
505 var sectionName = dict["isa"].AsString();
506
507 if (m_Section.ContainsKey(sectionName))
508 {
509 var section = m_Section[sectionName];
510 section.AddObject(kv.Key, dict);
511 }
512 else
513 {
514 UnknownSection section;
515 if (m_UnknownSections.ContainsKey(sectionName))
516 section = m_UnknownSections[sectionName];
517 else
518 {
519 section = new UnknownSection(sectionName);
520 m_UnknownSections.Add(sectionName, section);
521 }
522 section.AddObject(kv.Key, dict);
523
524 // update section order
525 if (!m_SectionOrder.Contains(sectionName))
526 {
527 int pos = 0;
528 if (prevSectionName != null)
529 {
530 // this never fails, because we already added any previous unknown sections
531 // to m_SectionOrder
532 pos = m_SectionOrder.FindIndex(x => x == prevSectionName);
533 pos += 1;
534 }
535 m_SectionOrder.Insert(pos, sectionName);
536 }
537 }
538 prevSectionName = sectionName;
539 }
540 RepairStructure(allGuids);
541 RefreshAuxMaps();
542 }

545 public string WriteToString()
546 {
547 var commentMap = BuildCommentMap();
548 var emptyChecker = new PropertyCommentChecker();
549 var emptyCommentMap = new GUIDToCommentMap();
550
551 // since we need to add custom comments, the serialization is much more complex
552 StringBuilder objectsSb = new StringBuilder();
553 if (m_ObjectVersion != null) // objectVersion comes right before objects
554 objectsSb.AppendFormat("objectVersion = {0};\n\t", m_ObjectVersion);
555 objectsSb.Append("objects = {");
556 foreach (string sectionName in m_SectionOrder)
557 {
558 if (m_Section.ContainsKey(sectionName))
559 m_Section[sectionName].WriteSection(objectsSb, commentMap);
560 else if (m_UnknownSections.ContainsKey(sectionName))
561 m_UnknownSections[sectionName].WriteSection(objectsSb, commentMap);
562 }
563 foreach (var kv in m_UnknownObjects.values)
564 Serializer.WriteDictKeyValue(objectsSb, kv.Key, kv.Value, 2, false, emptyChecker, emptyCommentMap);
565 objectsSb.Append("\n\t};");
566
567 StringBuilder contentSb = new StringBuilder();
568 contentSb.Append("// !$*UTF8*$!\n");
569 Serializer.WriteDict(contentSb, m_RootElements, 0, false,
570 new PropertyCommentChecker(new string[]{"rootObject/*"}), commentMap);
571 contentSb.Append("\n");
572 string content = contentSb.ToString();
573
574 content = content.Replace("objects = OBJMARKER;", objectsSb.ToString());
575 return content;
576 }

610 /* Removes objects from the given @a section for which @a checker returns true.
611 Also removes the guids of the removed elements from allGuids dictionary.
612 Returns true if any objects were removed.
613 */
614 static bool RemoveObjectsFromSection<T>(KnownSectionBase<T> section,
615 Dictionary<string, bool> allGuids,
616 Func<T, bool> checker) where T : PBXObjectData, new()
617 {
618 List<string> guidsToRemove = null;
619 foreach (var kv in section.GetEntries())
620 {
621 if (checker(kv.Value))
622 {
623 if (guidsToRemove == null)
624 guidsToRemove = new List<string>();
625 guidsToRemove.Add(kv.Key);
626 }
627 }
628 if (guidsToRemove != null)
629 {
630 foreach (var guid in guidsToRemove)
631 {
632 section.RemoveEntry(guid);
633 allGuids.Remove(guid);
634 }
635 return true;
636 }
637 return false;
638 }

640 // Returns true if changes were done and one should call RepairStructureImpl again
641 bool RepairStructureImpl(Dictionary<string, bool> allGuids)
642 {
643 bool changed = false;
644
645 // PBXBuildFile
646 changed |= RemoveObjectsFromSection(buildFiles, allGuids,
647 o => (o.fileRef == null || !allGuids.ContainsKey(o.fileRef)));
648 // PBXFileReference / fileRefs not cleaned
649
650 // PBXGroup
651 changed |= RemoveObjectsFromSection(groups, allGuids, o => o.children == null);
652 foreach (var o in groups.GetObjects())
653 RemoveMissingGuidsFromGuidList(o.children, allGuids);
654
655 // PBXContainerItem / containerItems not cleaned
656 // PBXReferenceProxy / references not cleaned
657
658 // PBXSourcesBuildPhase
659 changed |= RemoveObjectsFromSection(sources, allGuids, o => o.files == null);
660 foreach (var o in sources.GetObjects())
661 RemoveMissingGuidsFromGuidList(o.files, allGuids);
662 // PBXFrameworksBuildPhase
663 changed |= RemoveObjectsFromSection(frameworks, allGuids, o => o.files == null);
664 foreach (var o in frameworks.GetObjects())
665 RemoveMissingGuidsFromGuidList(o.files, allGuids);
666 // PBXResourcesBuildPhase
667 changed |= RemoveObjectsFromSection(resources, allGuids, o => o.files == null);
668 foreach (var o in resources.GetObjects())
669 RemoveMissingGuidsFromGuidList(o.files, allGuids);
670 // PBXCopyFilesBuildPhase
671 changed |= RemoveObjectsFromSection(copyFiles, allGuids, o => o.files == null);
672 foreach (var o in copyFiles.GetObjects())
673 RemoveMissingGuidsFromGuidList(o.files, allGuids);
674 // PBXShellScriptsBuildPhase
675 changed |= RemoveObjectsFromSection(shellScripts, allGuids, o => o.files == null);
676 foreach (var o in shellScripts.GetObjects())
677 RemoveMissingGuidsFromGuidList(o.files, allGuids);

679 // PBXNativeTarget
680 changed |= RemoveObjectsFromSection(nativeTargets, allGuids, o => o.phases == null);

682 RemoveMissingGuidsFromGuidList(o.phases, allGuids);

684 // PBXTargetDependency / targetDependencies not cleaned
