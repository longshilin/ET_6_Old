66 private static void CheckFields(SRMonoBehaviourEx instance, bool justSet = false)
67 {
68 if (_checkedFields == null)
69 {
70 _checkedFields = new Dictionary<Type, IList<FieldInfo>>();
71 }
72
73 var t = instance.GetType();
74
75 IList<FieldInfo> cache;
76
77 if (!_checkedFields.TryGetValue(instance.GetType(), out cache))
78 {
79 cache = ScanType(t);
80
81 _checkedFields.Add(t, cache);
82 }
83
84 PopulateObject(cache, instance, justSet);
85 }

87 private static void PopulateObject(IList<FieldInfo> cache, SRMonoBehaviourEx instance, bool justSet)
88 {
89 for (var i = 0; i < cache.Count; i++)
90 {
91 var f = cache[i];
92
93 if (!EqualityComparer<object>.Default.Equals(f.Field.GetValue(instance), null))
94 {
95 continue;
96 }
97
98 // If import is enabled, use SRServiceManager to import the reference
99 if (f.Import)
100 {
101 var t = f.ImportType ?? f.Field.FieldType;
102
103 var service = SRServiceManager.GetService(t);
104
105 if (service == null)
106 {
107 Debug.LogWarning("Field {0} import failed (Type {1})".Fmt(f.Field.Name, t));
108 continue;
109 }
110
111 f.Field.SetValue(instance, service);
112
113 continue;
114 }
115
116 // If autoset is enabled on field, try and find the component on the GameObject
117
118 if (f.AutoSet)
119 {
120 var newValue = instance.GetComponent(f.Field.FieldType);
121
122 if (!EqualityComparer<object>.Default.Equals(newValue, null))
123 {
124 f.Field.SetValue(instance, newValue);
125 continue;
126 }
127 }
128
129 if (justSet)
130 {
131 continue;
132 }
133
134 if (f.AutoCreate)
135 {
136 var newValue = instance.CachedGameObject.AddComponent(f.Field.FieldType);
137 f.Field.SetValue(instance, newValue);
138 }
139
140 throw new UnassignedReferenceException(
141 "Field {0} is unassigned, but marked with RequiredFieldAttribute".Fmt(f.Field.Name));
142 }
143 }

145 private static List<FieldInfo> ScanType(Type t)
146 {
147 var cache = new List<FieldInfo>();
148
149 // Check for attribute added to the class
150 var globalAttr = SRReflection.GetAttribute<RequiredFieldAttribute>(t);
151
152#if NETFX_CORE
153 var fields = t.GetTypeInfo().DeclaredFields.Where(f => !f.IsStatic);
154#else
155 // Check each field for the attribute
156 var fields = t.GetFields(BindingFlags.Public | BindingFlags.Instance | BindingFlags.NonPublic);
157#endif
158
159 foreach (var f in fields)
160 {
161 var requiredFieldAttribute = SRReflection.GetAttribute<RequiredFieldAttribute>(f);
162 var importAttribute = SRReflection.GetAttribute<ImportAttribute>(f);
163
164 if (globalAttr == null && requiredFieldAttribute == null && importAttribute == null)
165 {
166 continue; // Early out if no attributes found.
167 }
168
169 var info = new FieldInfo();
170 info.Field = f;
171
172 if (importAttribute != null)
173 {
174 info.Import = true;
175 info.ImportType = importAttribute.Service;
176 }
177 else if (requiredFieldAttribute != null)
178 {
179 info.AutoSet = requiredFieldAttribute.AutoSearch;
180 info.AutoCreate = requiredFieldAttribute.AutoCreate;
181 }
182 else
183 {
184 info.AutoSet = globalAttr.AutoSearch;
185 info.AutoCreate = globalAttr.AutoCreate;
186 }
187
188 cache.Add(info);
189 }
190
191 return cache;
192 }
