147 // Parses a string that contains a XML file. No validation is done.
148 internal static XDocument ParseXmlNoDtd(string text)
149 {
150 XmlReaderSettings settings = new XmlReaderSettings();
151 //settings.ProhibitDtd = false;
152 settings.DtdProcessing = DtdProcessing.Ignore;
153 settings.XmlResolver = null; // prevent DTD download
154
155 XmlReader xmlReader = XmlReader.Create(new StringReader(text), settings);
156 return XDocument.Load(xmlReader);
157 }
158
159 // LINQ serializes XML DTD declaration with an explicit empty 'internal subset'
160 // (a pair of square brackets at the end of Doctype declaration).
161 // Even though this is valid XML, XCode does not like it, hence this workaround.
162 internal static string CleanDtdToString(XDocument doc)
163 {
164 // LINQ does not support changing the DTD of existing XDocument instances,
165 // so we create a dummy document for printing of the Doctype declaration.
166 // A single dummy element is added to force LINQ not to omit the declaration.
167 // Also, utf-8 encoding is forced since this is the encoding we use when writing to file in UpdateInfoPlist.
168 if (doc.DocumentType != null)
169 {
170 XDocument tmpDoc =
171 new XDocument(new XDeclaration("1.0", "utf-8", null),
172 new XDocumentType(doc.DocumentType.Name, doc.DocumentType.PublicId, doc.DocumentType.SystemId, null),
173 new XElement(doc.Root.Name));
174 return "" + tmpDoc.Declaration + "\n" + tmpDoc.DocumentType + "\n" + doc.Root;
175 }
176 else
177 {
178 XDocument tmpDoc = new XDocument(new XDeclaration("1.0", "utf-8", null), new XElement(doc.Root.Name));
179 return "" + tmpDoc.Declaration + Environment.NewLine + doc.Root;
180 }
181 }

183 private static string GetText(XElement xml)
184 {
185 return String.Join("", xml.Nodes().OfType<XText>().Select(x => x.Value).ToArray());
186 }

188 private static PlistElement ReadElement(XElement xml)
189 {
190 switch (xml.Name.LocalName)
191 {
192 case "dict":
193 {
194 List<XElement> children = xml.Elements().ToList();
195 var el = new PlistElementDict();
196
197 if (children.Count % 2 == 1)
198 throw new Exception("Malformed plist file");
199
200 for (int i = 0; i < children.Count - 1; i++)
201 {
202 if (children[i].Name != "key")
203 throw new Exception("Malformed plist file");
204 string key = GetText(children[i]).Trim();
205 var newChild = ReadElement(children[i+1]);
206 if (newChild != null)
207 {
208 i++;
209 el[key] = newChild;
210 }
211 }
212 return el;
213 }
214 case "array":
215 {
216 List<XElement> children = xml.Elements().ToList();
217 var el = new PlistElementArray();
218
219 foreach (var childXml in children)
220 {
221 var newChild = ReadElement(childXml);
222 if (newChild != null)
223 el.values.Add(newChild);
224 }
225 return el;
226 }
227 case "string":
228 return new PlistElementString(GetText(xml));
229 case "integer":
230 {
231 int r;
232 if (int.TryParse(GetText(xml), out r))
233 return new PlistElementInteger(r);
234 return null;
235 }
236 case "true":
237 return new PlistElementBoolean(true);
238 case "false":
239 return new PlistElementBoolean(false);
240 default:
241 return null;
242 }
243 }

266 public void ReadFromString(string text)
267 {
268 XDocument doc = ParseXmlNoDtd(text);
269 version = (string) doc.Root.Attribute("version");
270 XElement xml = doc.XPathSelectElement("plist/dict");
271
272 var dict = ReadElement(xml);
273 if (dict == null)
274 throw new Exception("Error parsing plist file");
275 root = dict as PlistElementDict;
276 if (root == null)
277 throw new Exception("Malformed plist file");

280 private static XElement WriteElement(PlistElement el)
281 {
282 if (el is PlistElementBoolean)
283 {
284 var realEl = el as PlistElementBoolean;
285 return new XElement(realEl.value ? "true" : "false");
286 }
287 if (el is PlistElementInteger)
288 {
289 var realEl = el as PlistElementInteger;
290 return new XElement("integer", realEl.value.ToString());
291 }
292 if (el is PlistElementString)
293 {
294 var realEl = el as PlistElementString;
295 return new XElement("string", realEl.value);
296 }
297 if (el is PlistElementDict)
298 {
299 var realEl = el as PlistElementDict;
300 var dictXml = new XElement("dict");
301 foreach (var kv in realEl.values)
302 {
303 var keyXml = new XElement("key", kv.Key);
304 var valueXml = WriteElement(kv.Value);
305 if (valueXml != null)
306 {
307 dictXml.Add(keyXml);
308 dictXml.Add(valueXml);
309 }
310 }
311 return dictXml;
312 }
313 if (el is PlistElementArray)
314 {
315 var realEl = el as PlistElementArray;
316 var arrayXml = new XElement("array");
317 foreach (var v in realEl.values)
318 {
319 var elXml = WriteElement(v);
320 if (elXml != null)
321 arrayXml.Add(elXml);
322 }
323 return arrayXml;
324 }
325 return null;
326 }

339 public string WriteToString()
340 {
341 var el = WriteElement(root);
342 var rootEl = new XElement("plist");
343 rootEl.Add(new XAttribute("version", version));
344 rootEl.Add(el);
345
346 var doc = new XDocument();
347 doc.Add(rootEl);
