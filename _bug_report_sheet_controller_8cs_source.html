84 private IEnumerator SubmitCo()
85 {
86 if (BugReportScreenshotUtil.ScreenshotData == null && Settings.Instance.EnableBugReportScreenshot)
87 {
88 if (TakingScreenshot != null)
89 {
90 TakingScreenshot();
91 }
92
93 yield return new WaitForEndOfFrame();
94
95 yield return StartCoroutine(BugReportScreenshotUtil.ScreenshotCaptureCo());
96
97 if (ScreenshotComplete != null)
98 {
99 ScreenshotComplete();
100 }
101 }
102
103 var s = SRServiceManager.GetService<IBugReportService>();
104
105 var r = new BugReport();
106
107 r.Email = EmailField.text;
108 r.UserDescription = DescriptionField.text;
109
110 r.ConsoleLog = Service.Console.AllEntries.ToList();
111 r.SystemInformation = SRServiceManager.GetService<ISystemInformationService>().CreateReport();
112 r.ScreenshotData = BugReportScreenshotUtil.ScreenshotData;
113
114 BugReportScreenshotUtil.ScreenshotData = null;
115
116 s.SendBugReport(r, OnBugReportComplete, OnBugReportProgress);
117 }
