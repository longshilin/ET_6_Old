55 SourceFile GetSourceFile (Document document)
56 {
57 var url = document.Url;
58
59 SourceFile source_file;
60 if (source_files.TryGetValue (url, out source_file))
61 return source_file;
62
63 var entry = writer.DefineDocument (url, null, document.Hash != null && document.Hash.Length == 16 ? document.Hash : null);
64 var compile_unit = writer.DefineCompilationUnit (entry);
65
66 source_file = new SourceFile (compile_unit, entry);
67 source_files.Add (url, source_file);
68 return source_file;
69 }

71 void Populate (Collection<SequencePoint> sequencePoints, int [] offsets,
72 int [] startRows, int [] endRows, int [] startCols, int [] endCols, out SourceFile file)
73 {
74 SourceFile source_file = null;
75
76 for (int i = 0; i < sequencePoints.Count; i++) {
77 var sequence_point = sequencePoints [i];
78 offsets [i] = sequence_point.Offset;
79
80 if (source_file == null)
81 source_file = GetSourceFile (sequence_point.Document);
82
83 startRows [i] = sequence_point.StartLine;
84 endRows [i] = sequence_point.EndLine;
85 startCols [i] = sequence_point.StartColumn;
86 endCols [i] = sequence_point.EndColumn;
87 }
88
89 file = source_file;
90 }

92 public void Write (MethodDebugInformation info)
93 {
94 var method = new SourceMethod (info.method);
95
96 var sequence_points = info.SequencePoints;
97 int count = sequence_points.Count;
98 if (count == 0)
99 return;
100
101 var offsets = new int [count];
102 var start_rows = new int [count];
103 var end_rows = new int [count];
104 var start_cols = new int [count];
105 var end_cols = new int [count];
106
107 SourceFile file;
108 Populate (sequence_points, offsets, start_rows, end_rows, start_cols, end_cols, out file);
109
110 var builder = writer.OpenMethod (file.CompilationUnit, 0, method);
111
112 for (int i = 0; i < count; i++) {
113 builder.MarkSequencePoint (
114 offsets [i],
115 file.CompilationUnit.SourceFile,
116 start_rows [i],
117 start_cols [i],
118 end_rows [i],
119 end_cols [i],
120 false);
121 }
122
123 if (info.scope != null)
124 WriteRootScope (info.scope, info);
125
126 writer.CloseMethod ();
127 }

137 void WriteScope (ScopeDebugInformation scope, MethodDebugInformation info)
138 {
139 writer.OpenScope (scope.Start.Offset);
140
141 WriteScopeVariables (scope);
142
143 if (scope.HasScopes)
144 WriteScopes (scope.Scopes, info);
145
146 writer.CloseScope (scope.End.IsEndOfMethod ? info.code_size : scope.End.Offset);
147 }
148
149 void WriteScopes (Collection<ScopeDebugInformation> scopes, MethodDebugInformation info)
150 {
151 for (int i = 0; i < scopes.Count; i++)
152 WriteScope (scopes [i], info);
153 }

155 void WriteScopeVariables (ScopeDebugInformation scope)
156 {
157 if (!scope.HasVariables)
158 return;
159
160 foreach (var variable in scope.variables)
161 if (!string.IsNullOrEmpty (variable.Name))
162 writer.DefineLocalVariable (variable.Index, variable.Name);
163 }
