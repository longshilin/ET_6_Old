72 {
73 m = Matrix3x3.identity;
74
75 Vector3 z = viewVec;
76 // compute u0
77 float mag = z.Length();
78 if (mag < Mathf.Epsilon)
79 {
80 return false;
81 }
82 z /= mag;
83
84 Vector3 x = Vector3.Cross(upVec, z);
85 mag = x.Length();
86 if (mag < Mathf.Epsilon)
87 {
88 return false;
89 }
90 x /= mag;
91
92 Vector3 y = Vector3.Cross(z, x);
93 if (!Mathf.CompareApproximate(y.Length(), 1.0F))
94 return false;
95
96 m.SetOrthoNormalBasis(x, y, z);
97 return true;
