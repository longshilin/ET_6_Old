127 public static void MoveForward(this MoveComponent self, bool needCancel)
128 {
129 Unit unit = self.GetParent<Unit>();
130
131 long timeNow = TimeHelper.ClientNow();
132 long moveTime = timeNow - self.StartTime;
133
134 while (true)
135 {
136 if (moveTime <= 0)
137 {
138 return;
139 }
140
141 // 计算位置插值
142 if (moveTime >= self.NeedTime)
143 {
144 unit.Position = self.NextTarget;
145 if (self.TurnTime > 0)
146 {
147 unit.Rotation = self.To;
148 }
149 }
150 else
151 {
152 // 计算位置插值
153 float amount = moveTime * 1f / self.NeedTime;
154 if (amount > 0)
155 {
156 Vector3 newPos = Vector3.Lerp(self.StartPos, self.NextTarget, amount);
157 unit.Position = newPos;
158 }
159
160 // 计算方向插值
161 if (self.TurnTime > 0)
162 {
163 amount = moveTime * 1f / self.TurnTime;
164 Quaternion q = Quaternion.Slerp(self.From, self.To, amount);
165 unit.Rotation = q;
166 }
167 }
168
169 moveTime -= self.NeedTime;
170
171 // 表示这个点还没走完，等下一帧再来
172 if (moveTime < 0)
173 {
174 return;
175 }
176
177 // 到这里说明这个点已经走完
178
179 // 如果是最后一个点
180 if (self.N >= self.Targets.Count - 1)
181 {
182 unit.Position = self.NextTarget;
183 unit.Rotation = self.To;
184
185 Action<bool> callback = self.Callback;
186 self.Callback = null;
187
188 self.Clear();
189 callback?.Invoke(!needCancel);
190 return;
191 }
192
193 self.SetNextTarget();
194 }
195 }

208 private static void SetNextTarget(this MoveComponent self)
209 {
210
211 Unit unit = self.GetParent<Unit>();
212
213 ++self.N;
214
215 // 时间计算用服务端的位置, 但是移动要用客户端的位置来插值
216 Vector3 v = self.GetFaceV();
217 float distance = v.magnitude;
218
219 // 插值的起始点要以unit的真实位置来算
220 self.StartPos = unit.Position;
221
222 self.StartTime += self.NeedTime;
223
224 self.NeedTime = (long) (distance / self.Speed * 1000);
225
226
227 if (self.TurnTime > 0)
228 {
229 // 要用unit的位置
230 Vector3 faceV = self.GetFaceV();
231 if (faceV.sqrMagnitude < 0.0001f)
232 {
233 return;
234 }
235 self.From = unit.Rotation;
236
237 if (self.IsTurnHorizontal)
238 {
239 faceV.y = 0;
240 }
241
242 if (Math.Abs(faceV.x) > 0.01 || Math.Abs(faceV.z) > 0.01)
243 {
244 self.To = Quaternion.LookRotation(faceV, Vector3.up);
245 }
246
247 return;
248 }
249
250 if (self.TurnTime == 0) // turn time == 0 立即转向
251 {
252 Vector3 faceV = self.GetFaceV();
253 if (self.IsTurnHorizontal)
254 {
255 faceV.y = 0;
256 }
257
258 if (Math.Abs(faceV.x) > 0.01 || Math.Abs(faceV.z) > 0.01)
259 {
260 self.To = Quaternion.LookRotation(faceV, Vector3.up);
261 unit.Rotation = self.To;
262 }
263 }
