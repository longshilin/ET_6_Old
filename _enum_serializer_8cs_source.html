123 public object Read(object value, ProtoReader source)
124 {
125 Helpers.DebugAssert(value == null); // since replaces
126 int wireValue = source.ReadInt32();
127 if(map == null) {
128 return WireToEnum(wireValue);
129 }
130 for(int i = 0 ; i < map.Length ; i++) {
131 if(map[i].WireValue == wireValue) {
132 return map[i].TypedValue;
133 }
134 }
135 source.ThrowEnumException(ExpectedType, wireValue);
136 return null; // to make compiler happy
137 }
138 public void Write(object value, ProtoWriter dest)
139 {
140 if (map == null)
141 {
142 ProtoWriter.WriteInt32(EnumToWire(value), dest);
143 }
144 else
145 {
146 for (int i = 0; i < map.Length; i++)
147 {
148 if (object.Equals(map[i].TypedValue, value))
149 {
150 ProtoWriter.WriteInt32(map[i].WireValue, dest);
151 return;
152 }
153 }
154 ProtoWriter.ThrowEnumException(dest, value);
155 }

160 {
161 ProtoTypeCode typeCode = GetTypeCode();

163 {
164 ctx.LoadValue(valueFrom);

195 void IProtoSerializer.EmitRead(Compiler.CompilerContext ctx, Compiler.Local valueFrom)
196 {
197 ProtoTypeCode typeCode = GetTypeCode();
198 if (map == null)
199 {
200 ctx.EmitBasicRead("ReadInt32", ctx.MapType(typeof(int)));
201 ctx.ConvertFromInt32(typeCode, false);
202 }
203 else
204 {
205 int[] wireValues = new int[map.Length];
206 object[] values = new object[map.Length];
207 for (int i = 0; i < map.Length; i++)
208 {
209 wireValues[i] = map[i].WireValue;
210 values[i] = map[i].RawValue;
211 }
212 using (Compiler.Local result = new Compiler.Local(ctx, ExpectedType))
213 using (Compiler.Local wireValue = new Compiler.Local(ctx, ctx.MapType(typeof(int))))
214 {
215 ctx.EmitBasicRead("ReadInt32", ctx.MapType(typeof(int)));
216 ctx.StoreValue(wireValue);
217 Compiler.CodeLabel @continue = ctx.DefineLabel();
218 foreach (BasicList.Group group in BasicList.GetContiguousGroups(wireValues, values))
219 {
220 Compiler.CodeLabel tryNextGroup = ctx.DefineLabel();
221 int groupItemCount = group.Items.Count;
222 if (groupItemCount == 1)
223 {
224 // discreet group; use an equality test
225 ctx.LoadValue(wireValue);
226 ctx.LoadValue(group.First);
227 Compiler.CodeLabel processThisValue = ctx.DefineLabel();
228 ctx.BranchIfEqual(processThisValue, true);
229 ctx.Branch(tryNextGroup, false);
230 WriteEnumValue(ctx, typeCode, processThisValue, @continue, group.Items[0], @result);
231 }
232 else
233 {
234 // implement as a jump-table-based switch
235 ctx.LoadValue(wireValue);
236 ctx.LoadValue(group.First);
237 ctx.Subtract(); // jump-tables are zero-based
238 Compiler.CodeLabel[] jmp = new Compiler.CodeLabel[groupItemCount];
239 for (int i = 0; i < groupItemCount; i++) {
240 jmp[i] = ctx.DefineLabel();
241 }
242 ctx.Switch(jmp);
243 // write the default...
244 ctx.Branch(tryNextGroup, false);
245 for (int i = 0; i < groupItemCount; i++)
246 {
247 WriteEnumValue(ctx, typeCode, jmp[i], @continue, group.Items[i], @result);
248 }
249 }
250 ctx.MarkLabel(tryNextGroup);
251 }
252 // throw source.CreateEnumException(ExpectedType, wireValue);
253 ctx.LoadReaderWriter();
254 ctx.LoadValue(ExpectedType);
255 ctx.LoadValue(wireValue);
256 ctx.EmitCall(ctx.MapType(typeof(ProtoReader)).GetMethod("ThrowEnumException"));
257 ctx.MarkLabel(@continue);
258 ctx.LoadValue(result);
259 }
260 }

262 private static void WriteEnumValue(Compiler.CompilerContext ctx, ProtoTypeCode typeCode, object value)
263 {
264 switch (typeCode)
265 {
266 case ProtoTypeCode.Byte: ctx.LoadValue((int)(byte)value); break;
267 case ProtoTypeCode.SByte: ctx.LoadValue((int)(sbyte)value); break;
268 case ProtoTypeCode.Int16: ctx.LoadValue((int)(short)value); break;
269 case ProtoTypeCode.Int32: ctx.LoadValue((int)(int)value); break;
270 case ProtoTypeCode.Int64: ctx.LoadValue((long)(long)value); break;
271 case ProtoTypeCode.UInt16: ctx.LoadValue((int)(ushort)value); break;
272 case ProtoTypeCode.UInt32: ctx.LoadValue((int)(uint)value); break;
273 case ProtoTypeCode.UInt64: ctx.LoadValue((long)(ulong)value); break;
274 default: throw new InvalidOperationException();
275 }
