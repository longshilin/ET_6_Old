75 public static List<CodeBasicBlock> BuildBasicBlocks(MethodBody body, out Dictionary<Instruction, int> entryMapping)
76 {
77 entryMapping = new Dictionary<Instruction, int>();
78 HashSet<Instruction> branchTargets = new HashSet<Instruction>();
79 foreach (var i in body.Instructions)
80 {
81 switch (i.OpCode.OperandType)
82 {
83 case OperandType.InlineBrTarget:
84 case OperandType.ShortInlineBrTarget:
85 branchTargets.Add((Instruction)i.Operand);
86 break;
87 case OperandType.InlineSwitch:
88 {
89 var arr = i.Operand as Instruction[];
90 foreach (var j in arr)
91 branchTargets.Add(j);
92 }
93 break;
94 }
95 }
96
97 List<CodeBasicBlock> res = new List<CodeBasicBlock>();
98 CodeBasicBlock cur = new CodeBasicBlock();
99 res.Add(cur);
100 foreach (var i in body.Instructions)
101 {
102 if (branchTargets.Contains(i))
103 {
104 if(cur.entry != null && cur.entry != i)
105 {
106 entryMapping[cur.entry] = res.Count - 1;
107 cur = new CodeBasicBlock();
108 res.Add(cur);
109 }
110 }
111 cur.AddInstruction(i);
112 if (i.OpCode.Code == Code.Switch || i.OpCode.Code == Code.Throw || i.OpCode.OperandType == OperandType.InlineBrTarget || i.OpCode.OperandType == OperandType.ShortInlineBrTarget || i.OpCode.Code == Code.Endfinally)
113 {
114 if (cur.entry != null)
115 {
116 if (i.OpCode.OperandType == OperandType.InlineBrTarget || i.OpCode.OperandType == OperandType.ShortInlineBrTarget)
117 {
118 if (cur.entry != (Instruction)i.Operand)
119 {
120 entryMapping[cur.entry] = res.Count - 1;
121 cur = new CodeBasicBlock();
122 res.Add(cur);
123 }
124 }
125 else
126 {
127 if (i.Operand is Instruction)
128 {
129 if (cur.entry != (Instruction)i.Operand)
130 {
131 entryMapping[cur.entry] = res.Count - 1;
132 cur = new CodeBasicBlock();
133 res.Add(cur);
134 }
135 }
136 else
137 {
138 entryMapping[cur.entry] = res.Count - 1;
139 cur = new CodeBasicBlock();
140 res.Add(cur);
141 }
142 }
143 }
144 }
145 }
146 if (cur.entry != null)
147 entryMapping[cur.entry] = res.Count - 1;
148 else
149 res.RemoveAt(res.Count - 1);
150
151 for (int i = 0; i < res.Count; i++)
152 {
153 var block = res[i];
154 var lastIns = block.instructions[block.instructions.Count - 1];
155 switch (lastIns.OpCode.OperandType)
156 {
157 case OperandType.ShortInlineBrTarget:
158 case OperandType.InlineBrTarget:
159 {
160 var dstBlock = res[entryMapping[(Instruction)lastIns.Operand]];
161 dstBlock.prevBlocks.Add(block);
162 block.nextBlocks.Add(dstBlock);
163 switch (lastIns.OpCode.Code)
164 {
165 case Code.Brfalse:
166 case Code.Brfalse_S:
167 case Code.Brtrue:
168 case Code.Brtrue_S:
169 case Code.Beq:
170 case Code.Beq_S:
171 case Code.Bge:
172 case Code.Bge_S:
173 case Code.Bge_Un:
174 case Code.Bge_Un_S:
175 case Code.Bgt:
176 case Code.Bgt_S:
177 case Code.Bgt_Un:
178 case Code.Bgt_Un_S:
179 case Code.Ble:
180 case Code.Ble_S:
181 case Code.Ble_Un:
182 case Code.Ble_Un_S:
183 case Code.Blt:
184 case Code.Blt_S:
185 case Code.Blt_Un:
186 case Code.Blt_Un_S:
187 case Code.Bne_Un:
188 case Code.Bne_Un_S:
189 if (i < res.Count - 1)
190 {
191 var next = res[i + 1];
192 block.nextBlocks.Add(next);
193 next.prevBlocks.Add(block);
194 }
195 break;
196 default:
197 continue;
198 }
199 }
200 break;
201 case OperandType.InlineSwitch:
202 {
203 Instruction[] targets = (Instruction[])lastIns.Operand;
204 foreach(var t in targets)
205 {
206 var dstBlock = res[entryMapping[t]];
207 dstBlock.prevBlocks.Add(block);
208 block.nextBlocks.Add(dstBlock);
209 }
210 }
211 break;
212 }
213 if (i < res.Count - 1)
214 {
215 var next = res[i + 1];
216 block.nextBlocks.Add(next);
217 next.prevBlocks.Add(block);
218 }
219 }
220 return res;
221 }
222 }
