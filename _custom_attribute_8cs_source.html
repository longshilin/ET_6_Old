194 {
195 if (resolved || !HasImage)
196 return;
197
198 lock (Module.SyncRoot) {
199 if (resolved)
200 return;
201
202 Module.Read (this, (attribute, reader) => {
203 try {
204 reader.ReadCustomAttributeSignature (attribute);
205 resolved = true;
206 } catch (ResolutionException) {
207 if (arguments != null)
208 arguments.Clear ();
209 if (fields != null)
210 fields.Clear ();
211 if (properties != null)
