219 void InitParameters()
220 {
221 parameters = new List<IType>();
222 foreach (var i in ParametersCLR)
223 {
224 IType type = appdomain.GetType(i.ParameterType.FullName);
225 if (type == null)
226 type = appdomain.GetType(i.ParameterType.AssemblyQualifiedName);
227 if (i.ParameterType.IsGenericTypeDefinition)
228 {
229 if (type == null)
230 type = appdomain.GetType(i.ParameterType.GetGenericTypeDefinition().FullName);
231 if (type == null)
232 type = appdomain.GetType(i.ParameterType.GetGenericTypeDefinition().AssemblyQualifiedName);
233 }
234 if (i.ParameterType.ContainsGenericParameters)
235 {
236 var t = i.ParameterType;
237 if (t.HasElementType)
238 t = i.ParameterType.GetElementType();
239 else if (t.GetGenericArguments().Length > 0)
240 {
241 t = t.GetGenericArguments()[0];
242 }
243 type = new ILGenericParameterType(t.Name);
244 }
245 if (type == null)
246 throw new TypeLoadException();
247 parameters.Add(type);

272 param[paramCount - i] = obj;
273 }

332 unsafe void FixReference(int paramCount, StackObject* esp, object[] param, IList<object> mStack, object instance, bool hasThis)
333 {
334 var cnt = hasThis ? paramCount + 1 : paramCount;
335 for (int i = cnt; i >= 1; i--)
336 {
337 var p = Minus(esp, i);
338 var val = i <= paramCount ? param[paramCount - i] : instance;
339 switch (p->ObjectType)
340 {
341 case ObjectTypes.StackObjectReference:
342 {
343 var addr = *(long*)&p->Value;
344 var dst = (StackObject*)addr;
345 if (dst->ObjectType >= ObjectTypes.Object)
346 {
347 var obj = val;
348 if (obj is CrossBindingAdaptorType)
349 obj = ((CrossBindingAdaptorType)obj).ILInstance;
350 mStack[dst->Value] = obj;
351 }
352 else
353 {
354 ILIntepreter.UnboxObject(dst, val, mStack, appdomain);
355 }
356 }
357 break;
358 case ObjectTypes.FieldReference:
359 {
360 var obj = mStack[p->Value];
361 if (obj is ILTypeInstance)
362 {
363 ((ILTypeInstance)obj)[p->ValueLow] = val;
364 }
365 else
366 {
367 var t = appdomain.GetType(obj.GetType()) as CLRType;
368 t.GetField(p->ValueLow).SetValue(obj, val);
369 }
370 }
371 break;
372 case ObjectTypes.StaticFieldReference:
373 {
374 var t = appdomain.GetType(p->Value);
375 if (t is ILType)
376 {
377 ((ILType)t).StaticInstance[p->ValueLow] = val;
378 }
379 else
380 {
381 ((CLRType)t).SetStaticFieldValue(p->ValueLow, val);
382 }
383 }
384 break;
385 case ObjectTypes.ArrayReference:
386 {
387 var arr = mStack[p->Value] as Array;
388 arr.SetValue(val, p->ValueLow);
389 }
390 break;
391 }
392 }
