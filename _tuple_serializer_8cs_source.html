110 public object Read(object value, ProtoReader source)
111 {
112 object[] values = new object[members.Length];
113 bool invokeCtor = false;
114 if (value == null)
115 {
116 invokeCtor = true;
117 }
118 for (int i = 0; i < values.Length; i++)
119 values[i] = GetValue(value, i);
120 int field;
121 while ((field = source.ReadFieldHeader()) > 0)
122 {
123 invokeCtor = true;
124 if (field <= tails.Length)
125 {
126 IProtoSerializer tail = tails[field - 1];
127 values[field - 1] = tails[field - 1].Read(tail.RequiresOldValue ? values[field - 1] : null, source);
128 }
129 else
130 {
131 source.SkipField();
132 }
133 }
134 return invokeCtor ? ctor.Invoke(values) : value;
135 }

186 public void EmitRead(Compiler.CompilerContext ctx, Compiler.Local incoming)
187 {
188 using (Compiler.Local objValue = ctx.GetLocalWithValue(ExpectedType, incoming))
189 {
190 Compiler.Local[] locals = new Compiler.Local[members.Length];
191 try
192 {
193 for (int i = 0; i < locals.Length; i++)
194 {
195 Type type = GetMemberType(i);
196 bool store = true;
197 locals[i] = new Compiler.Local(ctx, type);
198 if (!Helpers.IsValueType(ExpectedType))
199 {
200 // value-types always read the old value
201 if (Helpers.IsValueType(type))
202 {
203 switch (Helpers.GetTypeCode(type))
204 {
205 case ProtoTypeCode.Boolean:
206 case ProtoTypeCode.Byte:
207 case ProtoTypeCode.Int16:
208 case ProtoTypeCode.Int32:
209 case ProtoTypeCode.SByte:
210 case ProtoTypeCode.UInt16:
211 case ProtoTypeCode.UInt32:
212 ctx.LoadValue(0);
213 break;
214 case ProtoTypeCode.Int64:
215 case ProtoTypeCode.UInt64:
216 ctx.LoadValue(0L);
217 break;
218 case ProtoTypeCode.Single:
219 ctx.LoadValue(0.0F);
220 break;
221 case ProtoTypeCode.Double:
222 ctx.LoadValue(0.0D);
223 break;
224 case ProtoTypeCode.Decimal:
225 ctx.LoadValue(0M);
226 break;
227 case ProtoTypeCode.Guid:
228 ctx.LoadValue(Guid.Empty);
229 break;
230 default:
231 ctx.LoadAddress(locals[i], type);
232 ctx.EmitCtor(type);
233 store = false;
234 break;
235 }
236 }
237 else
238 {
239 ctx.LoadNullRef();
240 }
241 if (store)
242 {
243 ctx.StoreValue(locals[i]);
244 }
245 }
246 }
247
248 Compiler.CodeLabel skipOld = Helpers.IsValueType(ExpectedType)
249 ? new Compiler.CodeLabel()
250 : ctx.DefineLabel();
251 if (!Helpers.IsValueType(ExpectedType))
252 {
253 ctx.LoadAddress(objValue, ExpectedType);
254 ctx.BranchIfFalse(skipOld, false);
255 }
256 for (int i = 0; i < members.Length; i++)
257 {
258 ctx.LoadAddress(objValue, ExpectedType);
259 if (members[i] is FieldInfo)
260 {
261 ctx.LoadValue((FieldInfo)members[i]);
262 }
263 else if (members[i] is PropertyInfo)
264 {
265 ctx.LoadValue((PropertyInfo)members[i]);
266 }
267 ctx.StoreValue(locals[i]);
268 }
269
270 if (!Helpers.IsValueType(ExpectedType)) ctx.MarkLabel(skipOld);
271
272 using (Compiler.Local fieldNumber = new Compiler.Local(ctx, ctx.MapType(typeof(int))))
273 {
274 Compiler.CodeLabel @continue = ctx.DefineLabel(),
275 processField = ctx.DefineLabel(),
276 notRecognised = ctx.DefineLabel();
277 ctx.Branch(@continue, false);
278
279 Compiler.CodeLabel[] handlers = new Compiler.CodeLabel[members.Length];
280 for (int i = 0; i < members.Length; i++)
281 {
282 handlers[i] = ctx.DefineLabel();
283 }
284
285 ctx.MarkLabel(processField);
286
287 ctx.LoadValue(fieldNumber);
288 ctx.LoadValue(1);
289 ctx.Subtract(); // jump-table is zero-based
290 ctx.Switch(handlers);
291
292 // and the default:
293 ctx.Branch(notRecognised, false);
294 for (int i = 0; i < handlers.Length; i++)
295 {
296 ctx.MarkLabel(handlers[i]);
297 IProtoSerializer tail = tails[i];
298 Compiler.Local oldValIfNeeded = tail.RequiresOldValue ? locals[i] : null;
299 ctx.ReadNullCheckedTail(locals[i].Type, tail, oldValIfNeeded);
300 if (tail.ReturnsValue)
301 {
302 if (Helpers.IsValueType(locals[i].Type))
303 {
304 ctx.StoreValue(locals[i]);
305 }
306 else
307 {
308 Compiler.CodeLabel hasValue = ctx.DefineLabel(), allDone = ctx.DefineLabel();
309
310 ctx.CopyValue();
311 ctx.BranchIfTrue(hasValue, true); // interpret null as "don't assign"
312 ctx.DiscardValue();
313 ctx.Branch(allDone, true);
314 ctx.MarkLabel(hasValue);
315 ctx.StoreValue(locals[i]);
316 ctx.MarkLabel(allDone);
317 }
318 }
319 ctx.Branch(@continue, false);
320 }
321
322 ctx.MarkLabel(notRecognised);
323 ctx.LoadReaderWriter();
324 ctx.EmitCall(ctx.MapType(typeof(ProtoReader)).GetMethod("SkipField"));
325
326 ctx.MarkLabel(@continue);
327 ctx.EmitBasicRead("ReadFieldHeader", ctx.MapType(typeof(int)));
328 ctx.CopyValue();
329 ctx.StoreValue(fieldNumber);
330 ctx.LoadValue(0);
331 ctx.BranchIfGreater(processField, false);
332 }
333 for (int i = 0; i < locals.Length; i++)
334 {
335 ctx.LoadValue(locals[i]);
336 }
337
338 ctx.EmitCtor(ctor);
339 ctx.StoreValue(objValue);
340 }
341 finally
342 {
343 for (int i = 0; i < locals.Length; i++)
344 {
345 if (locals[i] != null)
346 locals[i].Dispose(); // release for re-use
347 }
348 }
349 }
