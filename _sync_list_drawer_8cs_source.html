53 SirenixEditorGUI.BeginHorizontalToolbar();
54 this.visible.Value = SirenixEditorGUI.Foldout(this.visible.Value, GUIHelper.TempContent("SyncList " + label.text + " [" + typeof(TList).Name + "]"));
55 EditorGUILayout.LabelField(GUIHelper.TempContent(minCount == maxCount ? (minCount == 0 ? "Empty" : minCount + " items") : minCount + " (" + maxCount + ") items"), SirenixGUIStyles.RightAlignedGreyMiniLabel);
56 SirenixEditorGUI.EndHorizontalToolbar();

60 GUIHelper.PushGUIEnabled(false);
