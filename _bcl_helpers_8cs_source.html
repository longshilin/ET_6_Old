93 private static void WriteTimeSpanImpl(TimeSpan timeSpan, ProtoWriter dest, DateTimeKind kind)
94 {
95 if (dest == null) throw new ArgumentNullException("dest");
96 long value;
97 switch(dest.WireType)
98 {
99 case WireType.String:
100 case WireType.StartGroup:
101 TimeSpanScale scale;
102 value = timeSpan.Ticks;
103 if (timeSpan == TimeSpan.MaxValue)
104 {
105 value = 1;
106 scale = TimeSpanScale.MinMax;
107 }
108 else if (timeSpan == TimeSpan.MinValue)
109 {
110 value = -1;
111 scale = TimeSpanScale.MinMax;
112 }
113 else if (value % TimeSpan.TicksPerDay == 0)
114 {
115 scale = TimeSpanScale.Days;
116 value /= TimeSpan.TicksPerDay;
117 }
118 else if (value % TimeSpan.TicksPerHour == 0)
119 {
120 scale = TimeSpanScale.Hours;
121 value /= TimeSpan.TicksPerHour;
122 }
123 else if (value % TimeSpan.TicksPerMinute == 0)
124 {
125 scale = TimeSpanScale.Minutes;
126 value /= TimeSpan.TicksPerMinute;
127 }
128 else if (value % TimeSpan.TicksPerSecond == 0)
129 {
130 scale = TimeSpanScale.Seconds;
131 value /= TimeSpan.TicksPerSecond;
132 }
133 else if (value % TimeSpan.TicksPerMillisecond == 0)
134 {
135 scale = TimeSpanScale.Milliseconds;
136 value /= TimeSpan.TicksPerMillisecond;
137 }
138 else
139 {
140 scale = TimeSpanScale.Ticks;
141 }
142
143 SubItemToken token = ProtoWriter.StartSubItem(null, dest);
144
145 if(value != 0) {
146 ProtoWriter.WriteFieldHeader(FieldTimeSpanValue, WireType.SignedVariant, dest);
147 ProtoWriter.WriteInt64(value, dest);
148 }
149 if(scale != TimeSpanScale.Days) {
150 ProtoWriter.WriteFieldHeader(FieldTimeSpanScale, WireType.Variant, dest);
151 ProtoWriter.WriteInt32((int)scale, dest);
152 }
153 if(kind != DateTimeKind.Unspecified)
154 {
155 ProtoWriter.WriteFieldHeader(FieldTimeSpanKind, WireType.Variant, dest);
156 ProtoWriter.WriteInt32((int)kind, dest);
157 }
158 ProtoWriter.EndSubItem(token, dest);
159 break;
160 case WireType.Fixed64:
161 ProtoWriter.WriteInt64(timeSpan.Ticks, dest);
162 break;
163 default:
164 throw new ProtoException("Unexpected wire-type: " + dest.WireType.ToString());
165 }

332 private static long ReadTimeSpanTicks(ProtoReader source, out DateTimeKind kind) {
333 kind = DateTimeKind.Unspecified;
334 switch (source.WireType)
335 {
336 case WireType.String:
337 case WireType.StartGroup:
338 SubItemToken token = ProtoReader.StartSubItem(source);
339 int fieldNumber;
340 TimeSpanScale scale = TimeSpanScale.Days;
341 long value = 0;
342 while ((fieldNumber = source.ReadFieldHeader()) > 0)
343 {
344 switch (fieldNumber)
345 {
346 case FieldTimeSpanScale:
347 scale = (TimeSpanScale)source.ReadInt32();
348 break;
349 case FieldTimeSpanValue:
350 source.Assert(WireType.SignedVariant);
351 value = source.ReadInt64();
352 break;
353 case FieldTimeSpanKind:
354 kind = (DateTimeKind)source.ReadInt32();
355 switch(kind)
356 {
357 case DateTimeKind.Unspecified:
358 case DateTimeKind.Utc:
359 case DateTimeKind.Local:
360 break; // fine
361 default:
362 throw new ProtoException("Invalid date/time kind: " + kind.ToString());
363 }
364 break;
365 default:
366 source.SkipField();
367 break;
368 }
369 }
370 ProtoReader.EndSubItem(token, source);
371 switch (scale)
372 {
373 case TimeSpanScale.Days:
374 return value * TimeSpan.TicksPerDay;
375 case TimeSpanScale.Hours:
376 return value * TimeSpan.TicksPerHour;
377 case TimeSpanScale.Minutes:
378 return value * TimeSpan.TicksPerMinute;
379 case TimeSpanScale.Seconds:
380 return value * TimeSpan.TicksPerSecond;
381 case TimeSpanScale.Milliseconds:
382 return value * TimeSpan.TicksPerMillisecond;
383 case TimeSpanScale.Ticks:
384 return value;
385 case TimeSpanScale.MinMax:
386 switch (value)
387 {
388 case 1: return long.MaxValue;
389 case -1: return long.MinValue;
390 default: throw new ProtoException("Unknown min/max value: " + value.ToString());
391 }
392 default:
393 throw new ProtoException("Unknown timescale: " + scale.ToString());
394 }
395 case WireType.Fixed64:
396 return source.ReadInt64();
397 default:
398 throw new ProtoException("Unexpected wire-type: " + source.WireType.ToString());
399 }

407 public static decimal ReadDecimal(ProtoReader reader)
408 {
409 ulong low = 0;
410 uint high = 0;
411 uint signScale = 0;
412
413 int fieldNumber;
414 SubItemToken token = ProtoReader.StartSubItem(reader);
415 while ((fieldNumber = reader.ReadFieldHeader()) > 0)
416 {
417 switch (fieldNumber)
418 {
419 case FieldDecimalLow: low = reader.ReadUInt64(); break;
420 case FieldDecimalHigh: high = reader.ReadUInt32(); break;
421 case FieldDecimalSignScale: signScale = reader.ReadUInt32(); break;
422 default: reader.SkipField(); break;
423 }
424
425 }
426 ProtoReader.EndSubItem(token, reader);
427
428 if (low == 0 && high == 0) return decimal.Zero;
429
430 int lo = (int)(low & 0xFFFFFFFFL),
431 mid = (int)((low >> 32) & 0xFFFFFFFFL),
432 hi = (int)high;
433 bool isNeg = (signScale & 0x0001) == 0x0001;
434 byte scale = (byte)((signScale & 0x01FE) >> 1);
435 return new decimal(lo, mid, hi, isNeg, scale);
436 }
440 public static void WriteDecimal(decimal value, ProtoWriter writer)
441 {
442 int[] bits = decimal.GetBits(value);
443 ulong a = ((ulong)bits[1]) << 32, b = ((ulong)bits[0]) & 0xFFFFFFFFL;
444 ulong low = a | b;
445 uint high = (uint)bits[2];
446 uint signScale = (uint)(((bits[3] >> 15) & 0x01FE) | ((bits[3] >> 31) & 0x0001));
447
448 SubItemToken token = ProtoWriter.StartSubItem(null, writer);
449 if (low != 0) {

453 if (high != 0)
454 {
455 ProtoWriter.WriteFieldHeader(FieldDecimalHigh, WireType.Variant, writer);
456 ProtoWriter.WriteUInt32(high, writer);
457 }
458 if (signScale != 0)
459 {
460 ProtoWriter.WriteFieldHeader(FieldDecimalSignScale, WireType.Variant, writer);
461 ProtoWriter.WriteUInt32(signScale, writer);

549 public static object ReadNetObject(object value, ProtoReader source, int key, Type type, NetObjectOptions options)
550 {
551#if FEAT_IKVM
552 throw new NotSupportedException();
553#else
554 SubItemToken token = ProtoReader.StartSubItem(source);
555 int fieldNumber;
556 int newObjectKey = -1, newTypeKey = -1, tmp;
557 while ((fieldNumber = source.ReadFieldHeader()) > 0)
558 {
559 switch (fieldNumber)
560 {
561 case FieldExistingObjectKey:
562 tmp = source.ReadInt32();
563 value = source.NetCache.GetKeyedObject(tmp);
564 break;
565 case FieldNewObjectKey:
566 newObjectKey = source.ReadInt32();
567 break;
568 case FieldExistingTypeKey:
569 tmp = source.ReadInt32();
570 type = (Type)source.NetCache.GetKeyedObject(tmp);
571 key = source.GetTypeKey(ref type);
572 break;
573 case FieldNewTypeKey:
574 newTypeKey = source.ReadInt32();
575 break;
576 case FieldTypeName:
577 string typeName = source.ReadString();
578 type = source.DeserializeType(typeName);
579 if(type == null)
580 {
581 throw new ProtoException("Unable to resolve type: " + typeName + " (you can use the TypeModel.DynamicTypeFormatting event to provide a custom mapping)");
582 }
583 if (type == typeof(string))
584 {
585 key = -1;
586 }
587 else
588 {
589 key = source.GetTypeKey(ref type);
590 if (key < 0)
591 throw new InvalidOperationException("Dynamic type is not a contract-type: " + type.Name);
592 }
593 break;
594 case FieldObject:
595 bool isString = type == typeof(string);
596 bool wasNull = value == null;
597 bool lateSet = wasNull && (isString || ((options & NetObjectOptions.LateSet) != 0));
598
599 if (newObjectKey >= 0 && !lateSet)
600 {
601 if (value == null)
602 {
603 source.TrapNextObject(newObjectKey);
604 }
605 else
606 {
607 source.NetCache.SetKeyedObject(newObjectKey, value);
608 }
609 if (newTypeKey >= 0) source.NetCache.SetKeyedObject(newTypeKey, type);
610 }
611 object oldValue = value;
612 if (isString)
613 {
614 value = source.ReadString();
615 }
616 else
617 {
618 value = ProtoReader.ReadTypedObject(oldValue, key, source, type);
619 }
620
621 if (newObjectKey >= 0)
622 {
623 if(wasNull && !lateSet)
624 { // this both ensures (via exception) that it *was* set, and makes sure we don't shout
625 // about changed references
626 oldValue = source.NetCache.GetKeyedObject(newObjectKey);
627 }
628 if (lateSet)
629 {
630 source.NetCache.SetKeyedObject(newObjectKey, value);
631 if (newTypeKey >= 0) source.NetCache.SetKeyedObject(newTypeKey, type);
632 }
633 }
634 if (newObjectKey >= 0 && !lateSet && !ReferenceEquals(oldValue, value))
635 {
636 throw new ProtoException("A reference-tracked object changed reference during deserialization");
637 }
638 if (newObjectKey < 0 && newTypeKey >= 0)
639 { // have a new type, but not a new object
640 source.NetCache.SetKeyedObject(newTypeKey, type);
641 }
642 break;
643 default:
644 source.SkipField();
645 break;
646 }
647 }
648 if(newObjectKey >= 0 && (options & NetObjectOptions.AsReference) == 0)
649 {
650 throw new ProtoException("Object key in input stream, but reference-tracking was not expected");
651 }
652 ProtoReader.EndSubItem(token, source);
653
654 return value;
655#endif
656 }
660 public static void WriteNetObject(object value, ProtoWriter dest, int key, NetObjectOptions options)
661 {
662#if FEAT_IKVM
663 throw new NotSupportedException();
664#else
665 if (dest == null) throw new ArgumentNullException("dest");
666 bool dynamicType = (options & NetObjectOptions.DynamicType) != 0,
667 asReference = (options & NetObjectOptions.AsReference) != 0;
668 WireType wireType = dest.WireType;
669 SubItemToken token = ProtoWriter.StartSubItem(null, dest);
670 bool writeObject = true;
671 if (asReference)
672 {
673 bool existing;
674 int objectKey = dest.NetCache.AddObjectKey(value, out existing);
675 ProtoWriter.WriteFieldHeader(existing ? FieldExistingObjectKey : FieldNewObjectKey, WireType.Variant, dest);
676 ProtoWriter.WriteInt32(objectKey, dest);
677 if (existing)
678 {
679 writeObject = false;
680 }
681 }
682
683 if (writeObject)
684 {
685 if (dynamicType)
686 {
687 bool existing;
688 Type type = value.GetType();
689
690 if (!(value is string))
691 {
692 key = dest.GetTypeKey(ref type);
693 if (key < 0) throw new InvalidOperationException("Dynamic type is not a contract-type: " + type.Name);
694 }
695 int typeKey = dest.NetCache.AddObjectKey(type, out existing);
696 ProtoWriter.WriteFieldHeader(existing ? FieldExistingTypeKey : FieldNewTypeKey, WireType.Variant, dest);
697 ProtoWriter.WriteInt32(typeKey, dest);
698 if (!existing)
699 {
700 ProtoWriter.WriteFieldHeader(FieldTypeName, WireType.String, dest);
701 ProtoWriter.WriteString(dest.SerializeType(type), dest);
702 }
703
704 }
705 ProtoWriter.WriteFieldHeader(FieldObject, wireType, dest);
706 if (value is string)
707 {
708 ProtoWriter.WriteString((string)value, dest);
709 }
710 else {
711 ProtoWriter.WriteObject(value, key, dest);
712 }
713 }
714 ProtoWriter.EndSubItem(token, dest);
715#endif
716 }
