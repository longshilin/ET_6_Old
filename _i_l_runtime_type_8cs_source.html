32 {
33 if(type.TypeDefinition == null)
34 {
35 customAttributes = new Attribute[0];
36 attributeTypes = new Type[0];
37 return;
38 }
39 customAttributes = new Attribute[type.TypeDefinition.CustomAttributes.Count];
40 attributeTypes = new Type[customAttributes.Length];
41 for (int i = 0; i < type.TypeDefinition.CustomAttributes.Count; i++)
42 {
43 var attribute = type.TypeDefinition.CustomAttributes[i];
44 var at = appdomain.GetType(attribute.AttributeType, type, null);
45 try
46 {
47 Attribute ins = attribute.CreateInstance(at, appdomain) as Attribute;
48
49 attributeTypes[i] = at.ReflectionType is ILRuntimeWrapperType ? at.TypeForCLR : at.ReflectionType;
50 customAttributes[i] = ins;

60 void InitializeProperties()
61 {
62 if (type.TypeDefinition == null)
63 {
64 properties = new ILRuntimePropertyInfo[0];
65 return;
66 }
67 int cnt = type.TypeDefinition.HasProperties ? type.TypeDefinition.Properties.Count : 0;
68 properties = new ILRuntimePropertyInfo[cnt];

72 ILRuntimePropertyInfo pi = new ILRuntimePropertyInfo(pd, type);
73 properties[i] = pi;

92 {
93 if (ILType.TypeReference.HasGenericParameters)

96 for (int i = 0; i < ga.Length; i++)

99 if (typeArguments[i] is ILRuntimeType)

101 else

110 void InitializeFields()
111 {
112 int staticCnt = type.StaticFieldTypes != null ? type.StaticFieldTypes.Length : 0;
113 fields = new ILRuntimeFieldInfo[type.TotalFieldCount + staticCnt];
114 for (int i = 0; i < type.TotalFieldCount; i++)
115 {
116 Mono.Cecil.FieldDefinition fd;
117 var t = type.GetField(i, out fd);
118 fields[i] = new ILRuntimeFieldInfo(fd, this, i, t);
119 }
120 for (int i = type.TotalFieldCount; i < type.TotalFieldCount + staticCnt; i++)
121 {
122 fields[i] = new ILRuntimeFieldInfo(type.StaticFieldDefinitions[i - type.TotalFieldCount], this, true, i - type.TotalFieldCount);
