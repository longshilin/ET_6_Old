80 static void Write (ModuleDefinition module, Disposable<Stream> stream, WriterParameters parameters)
81 {
82 if ((module.Attributes & ModuleAttributes.ILOnly) == 0)
83 throw new NotSupportedException ("Writing mixed-mode assemblies is not supported");
84
85 if (module.HasImage && module.ReadingMode == ReadingMode.Deferred) {
86 var immediate_reader = new ImmediateModuleReader (module.Image);
87 immediate_reader.ReadModule (module, resolve_attributes: false);
88 immediate_reader.ReadSymbols (module);
89 }
90
91 module.MetadataSystem.Clear ();
92
93 if (module.symbol_reader != null)
94 module.symbol_reader.Dispose ();
95
96 var name = module.assembly != null ? module.assembly.Name : null;
97 var fq_name = stream.value.GetFileName ();
98 var timestamp = parameters.Timestamp ?? module.timestamp;
99 var symbol_writer_provider = parameters.SymbolWriterProvider;
100
101 if (symbol_writer_provider == null && parameters.WriteSymbols)
102 symbol_writer_provider = new DefaultSymbolWriterProvider ();
103
104 if (parameters.HasStrongNameKey && name != null) {
105 name.PublicKey = CryptoService.GetPublicKey (parameters);
106 module.Attributes |= ModuleAttributes.StrongNameSigned;
107 }
108
109 if (parameters.DeterministicMvid)
110 module.Mvid = Guid.Empty;
111
112 var metadata = new MetadataBuilder (module, fq_name, timestamp, symbol_writer_provider);
113 try {
114 module.metadata_builder = metadata;
115
116 using (var symbol_writer = GetSymbolWriter (module, fq_name, symbol_writer_provider, parameters)) {
117 metadata.SetSymbolWriter (symbol_writer);
118 BuildMetadata (module, metadata);
119
120 if (parameters.DeterministicMvid)
121 metadata.ComputeDeterministicMvid ();
122
123 var writer = ImageWriter.CreateWriter (module, metadata, stream);
124 stream.value.SetLength (0);
125 writer.WriteImage ();
126
127 if (parameters.HasStrongNameKey)
128 CryptoService.StrongName (stream.value, writer, parameters);
129 }
130 } finally {
131 module.metadata_builder = null;
132 }
133 }

135 static void BuildMetadata (ModuleDefinition module, MetadataBuilder metadata)
136 {
137 if (!module.HasImage) {
138 metadata.BuildMetadata ();
139 return;
140 }
141
142 module.Read (metadata, (builder, _) => {
143 builder.BuildMetadata ();
144 return builder;
145 });
146 }
147
148 static ISymbolWriter GetSymbolWriter (ModuleDefinition module, string fq_name, ISymbolWriterProvider symbol_writer_provider, WriterParameters parameters)
149 {
150 if (symbol_writer_provider == null)
151 return null;
152
153 if (parameters.SymbolStream != null)
154 return symbol_writer_provider.GetSymbolWriter (module, parameters.SymbolStream);
155
156 return symbol_writer_provider.GetSymbolWriter (module, fq_name);
157 }

185 abstract class MetadataTable<TRow> : MetadataTable where TRow : struct {
186
187 internal TRow [] rows = new TRow [2];
188 internal int length;
189
190 public sealed override int Length {
191 get { return length; }
192 }
193
194 public int AddRow (TRow row)
195 {
196 if (rows.Length == length)
197 Grow ();
198
199 rows [length++] = row;
200 return length;
201 }
202
203 void Grow ()
204 {
205 var rows = new TRow [this.rows.Length * 2];
206 Array.Copy (this.rows, rows, this.rows.Length);
207 this.rows = rows;
208 }
209
210 public override void Sort ()
211 {
212 }
213 }

223 {
224 return x == y ? 0 : x > y ? 1 : -1;
225 }
