146 private IProtoSerializer GetMoreSpecificSerializer(object value)
147 {
148 if (!CanHaveInheritance) return null;

255 private object InvokeCallback(MethodInfo method, object obj, SerializationContext context)
256 {
257 object result = null;
258 object[] args;
259 if (method != null)
260 { // pass in a streaming context if one is needed, else null
261 bool handled;
262 ParameterInfo[] parameters = method.GetParameters();
263 switch (parameters.Length)
264 {
265 case 0:
266 args = null;
267 handled = true;
268 break;
269 default:
270 args = new object[parameters.Length];
271 handled = true;
272 for (int i = 0; i < args.Length; i++)
273 {
274 object val;
275 Type paramType = parameters[i].ParameterType;
276 if (paramType == typeof(SerializationContext)) val = context;
277 else if (paramType == typeof(System.Type)) val = constructType;
278#if PLAT_BINARYFORMATTER || (SILVERLIGHT && NET_4_0)
279 else if (paramType == typeof(System.Runtime.Serialization.StreamingContext)) val = (System.Runtime.Serialization.StreamingContext)context;
280#endif
281 else
282 {
283 val = null;
284 handled = false;
285 }
286 args[i] = val;
287 }
288 break;
289 }
290 if(handled)
291 {
292 result = method.Invoke(obj, args);
293 }
294 else
295 {
296 throw Meta.CallbackSet.CreateInvalidCallbackSignature(method);
297 }
298
299 }
300 return result;
301 }
302 object CreateInstance(ProtoReader source, bool includeLocalCallback)
303 {
304 //Helpers.DebugWriteLine("* creating : " + forType.FullName);
305 object obj;

309 }

323 }

326 if (obj == null)

334 void IProtoSerializer.EmitWrite(Compiler.CompilerContext ctx, Compiler.Local valueFrom)
335 {
336 Type expected = ExpectedType;
337 using (Compiler.Local loc = ctx.GetLocalWithValue(expected, valueFrom))
338 {
339 // pre-callbacks
340 EmitCallbackIfNeeded(ctx, loc, TypeModel.CallbackType.BeforeSerialize);
341
342 Compiler.CodeLabel startFields = ctx.DefineLabel();
343 // inheritance
344 if (CanHaveInheritance)
345 {
346 for (int i = 0; i < serializers.Length; i++)
347 {
348 IProtoSerializer ser = serializers[i];
349 Type serType = ser.ExpectedType;
350 if (serType != forType)
351 {
352 Compiler.CodeLabel ifMatch = ctx.DefineLabel(), nextTest = ctx.DefineLabel();
353 ctx.LoadValue(loc);
354 ctx.TryCast(serType);
355 ctx.CopyValue();
356 ctx.BranchIfTrue(ifMatch, true);
357 ctx.DiscardValue();
358 ctx.Branch(nextTest, true);
359 ctx.MarkLabel(ifMatch);
360 ser.EmitWrite(ctx, null);
361 ctx.Branch(startFields, false);
362 ctx.MarkLabel(nextTest);
363 }
364 }
365
366
367 if (constructType != null && constructType != forType)
368 {
369 using(Compiler.Local actualType = new Compiler.Local(ctx, ctx.MapType(typeof(System.Type))))
370 {
371 // would have jumped to "fields" if an expected sub-type, so two options:
372 // a: *exactly* that type, b: an *unexpected* type
373 ctx.LoadValue(loc);
374 ctx.EmitCall(ctx.MapType(typeof(object)).GetMethod("GetType"));
375 ctx.CopyValue();
376 ctx.StoreValue(actualType);
377 ctx.LoadValue(forType);
378 ctx.BranchIfEqual(startFields, true);
379
380 ctx.LoadValue(actualType);
381 ctx.LoadValue(constructType);
382 ctx.BranchIfEqual(startFields, true);
383 }
384 }
385 else
386 {
387 // would have jumped to "fields" if an expected sub-type, so two options:
388 // a: *exactly* that type, b: an *unexpected* type
389 ctx.LoadValue(loc);
390 ctx.EmitCall(ctx.MapType(typeof(object)).GetMethod("GetType"));
391 ctx.LoadValue(forType);
392 ctx.BranchIfEqual(startFields, true);
393 }
394 // unexpected, then... note that this *might* be a proxy, which
395 // is handled by ThrowUnexpectedSubtype
396 ctx.LoadValue(forType);
397 ctx.LoadValue(loc);
398 ctx.EmitCall(ctx.MapType(typeof(object)).GetMethod("GetType"));
399 ctx.EmitCall(ctx.MapType(typeof(TypeModel)).GetMethod("ThrowUnexpectedSubtype",
400 BindingFlags.Public | BindingFlags.NonPublic | BindingFlags.Static));
401
402 }
403 // fields
404
405 ctx.MarkLabel(startFields);
406 for (int i = 0; i < serializers.Length; i++)
407 {
408 IProtoSerializer ser = serializers[i];
409 if (ser.ExpectedType == forType) ser.EmitWrite(ctx, loc);
410 }
411
412 // extension data
413 if (isExtensible)
414 {
415 ctx.LoadValue(loc);
416 ctx.LoadReaderWriter();
417 ctx.EmitCall(ctx.MapType(typeof(ProtoWriter)).GetMethod("AppendExtensionData"));
418 }
419 // post-callbacks
420 EmitCallbackIfNeeded(ctx, loc, TypeModel.CallbackType.AfterSerialize);
421 }

423 static void EmitInvokeCallback(Compiler.CompilerContext ctx, MethodInfo method, bool copyValue, Type constructType, Type type)
424 {
425 if (method != null)
426 {
427 if(copyValue) ctx.CopyValue(); // assumes the target is on the stack, and that we want to *retain* it on the stack
428 ParameterInfo[] parameters = method.GetParameters();
429 bool handled = true;
430
431 for (int i = 0; i < parameters.Length; i++)
432 {
433 Type parameterType = parameters[i].ParameterType;
434 if (parameterType == ctx.MapType(typeof(SerializationContext)))
435 {
436 ctx.LoadSerializationContext();
437 }
438 else if (parameterType == ctx.MapType(typeof(System.Type)))
439 {
440 Type tmp = constructType;
441 if (tmp == null) tmp = type; // no ?? in some C# profiles
442 ctx.LoadValue(tmp);
443 }
444#if PLAT_BINARYFORMATTER
445 else if (parameterType == ctx.MapType(typeof(System.Runtime.Serialization.StreamingContext)))
446 {
447 ctx.LoadSerializationContext();
448 MethodInfo op = ctx.MapType(typeof(SerializationContext)).GetMethod("op_Implicit", new Type[] { ctx.MapType(typeof(SerializationContext)) });
449 if (op != null)
450 { // it isn't always! (framework versions, etc)
451 ctx.EmitCall(op);
452 handled = true;
453 }
454 }
455#endif
456 else
457 {
458 handled = false;
459 }
460 }
461 if (handled)
462 {
463 ctx.EmitCall(method);
464 if (constructType != null)
465 {
466 if (method.ReturnType == ctx.MapType(typeof(object)))
467 {
468 ctx.CastFromObject(type);
469 }
470 }
471 }
472 else
473 {
474 throw Meta.CallbackSet.CreateInvalidCallbackSignature(method);
475 }

485 void IProtoTypeSerializer.EmitCallback(Compiler.CompilerContext ctx, Compiler.Local valueFrom, TypeModel.CallbackType callbackType)
486 {
487 bool actuallyHasInheritance = false;
488 if (CanHaveInheritance)
489 {
490
491 for (int i = 0; i < serializers.Length; i++)
492 {
493 IProtoSerializer ser = serializers[i];
494 if (ser.ExpectedType != forType && ((IProtoTypeSerializer)ser).HasCallbacks(callbackType))
495 {
496 actuallyHasInheritance = true;
497 }
498 }
499 }
500
501 Helpers.DebugAssert(((IProtoTypeSerializer)this).HasCallbacks(callbackType), "Shouldn't be calling this if there is nothing to do");
502 MethodInfo method = callbacks == null ? null : callbacks[callbackType];
503 if(method == null && !actuallyHasInheritance)
504 {
505 return;
506 }
507 ctx.LoadAddress(valueFrom, ExpectedType);
508 EmitInvokeCallback(ctx, method, actuallyHasInheritance, null, forType);
509
510 if (actuallyHasInheritance)
511 {
512 Compiler.CodeLabel @break = ctx.DefineLabel();
513 for (int i = 0; i < serializers.Length; i++)
514 {
515 IProtoSerializer ser = serializers[i];
516 IProtoTypeSerializer typeser;
517 Type serType = ser.ExpectedType;
518 if (serType != forType &&
519 (typeser = (IProtoTypeSerializer) ser).HasCallbacks(callbackType))
520 {
521 Compiler.CodeLabel ifMatch = ctx.DefineLabel(), nextTest = ctx.DefineLabel();
522 ctx.CopyValue();
523 ctx.TryCast(serType);
524 ctx.CopyValue();
525 ctx.BranchIfTrue(ifMatch, true);
526 ctx.DiscardValue();
527 ctx.Branch(nextTest, false);
528 ctx.MarkLabel(ifMatch);
529 typeser.EmitCallback(ctx, null, callbackType);
530 ctx.Branch(@break, false);
531 ctx.MarkLabel(nextTest);
532 }
533 }
534 ctx.MarkLabel(@break);
535 ctx.DiscardValue();
536 }

539 void IProtoSerializer.EmitRead(Compiler.CompilerContext ctx, Compiler.Local valueFrom)
540 {
541 Type expected = ExpectedType;
542 Helpers.DebugAssert(valueFrom != null);
543
544 using (Compiler.Local loc = ctx.GetLocalWithValue(expected, valueFrom))
545 using (Compiler.Local fieldNumber = new Compiler.Local(ctx, ctx.MapType(typeof(int))))
546 {
547 // pre-callbacks
548 if (HasCallbacks(TypeModel.CallbackType.BeforeDeserialize))
549 {
550 if(Helpers.IsValueType(ExpectedType))
551 {
552 EmitCallbackIfNeeded(ctx, loc, TypeModel.CallbackType.BeforeDeserialize);
553 }
554 else
555 { // could be null
556 Compiler.CodeLabel callbacksDone = ctx.DefineLabel();
557 ctx.LoadValue(loc);
558 ctx.BranchIfFalse(callbacksDone, false);
559 EmitCallbackIfNeeded(ctx, loc, TypeModel.CallbackType.BeforeDeserialize);
560 ctx.MarkLabel(callbacksDone);
561 }
562 }
563
564 Compiler.CodeLabel @continue = ctx.DefineLabel(), processField = ctx.DefineLabel();
565 ctx.Branch(@continue, false);
566
567 ctx.MarkLabel(processField);
568 foreach (BasicList.Group group in BasicList.GetContiguousGroups(fieldNumbers, serializers))
569 {
570 Compiler.CodeLabel tryNextField = ctx.DefineLabel();
571 int groupItemCount = group.Items.Count;
572 if (groupItemCount == 1)
573 {
574 // discreet group; use an equality test
575 ctx.LoadValue(fieldNumber);
576 ctx.LoadValue(group.First);
577 Compiler.CodeLabel processThisField = ctx.DefineLabel();
578 ctx.BranchIfEqual(processThisField, true);
579 ctx.Branch(tryNextField, false);
580 WriteFieldHandler(ctx, expected, loc, processThisField, @continue, (IProtoSerializer)group.Items[0]);
581 }
582 else
583 { // implement as a jump-table-based switch
584 ctx.LoadValue(fieldNumber);
585 ctx.LoadValue(group.First);
586 ctx.Subtract(); // jump-tables are zero-based
587 Compiler.CodeLabel[] jmp = new Compiler.CodeLabel[groupItemCount];
588 for (int i = 0; i < groupItemCount; i++) {
589 jmp[i] = ctx.DefineLabel();
590 }
591 ctx.Switch(jmp);
592 // write the default...
593 ctx.Branch(tryNextField, false);
594 for (int i = 0; i < groupItemCount; i++)
595 {
596 WriteFieldHandler(ctx, expected, loc, jmp[i], @continue, (IProtoSerializer)group.Items[i]);
597 }
598 }
599 ctx.MarkLabel(tryNextField);
600 }
601
602 EmitCreateIfNull(ctx, loc);
603 ctx.LoadReaderWriter();
604 if (isExtensible)
605 {
606 ctx.LoadValue(loc);
607 ctx.EmitCall(ctx.MapType(typeof(ProtoReader)).GetMethod("AppendExtensionData"));
608 }
609 else
610 {
611 ctx.EmitCall(ctx.MapType(typeof(ProtoReader)).GetMethod("SkipField"));
612 }
613
614 ctx.MarkLabel(@continue);
615 ctx.EmitBasicRead("ReadFieldHeader", ctx.MapType(typeof(int)));
616 ctx.CopyValue();
617 ctx.StoreValue(fieldNumber);
618 ctx.LoadValue(0);
619 ctx.BranchIfGreater(processField, false);
620
621 EmitCreateIfNull(ctx, loc);
622 // post-callbacks
623 EmitCallbackIfNeeded(ctx, loc, TypeModel.CallbackType.AfterDeserialize);
624
625 if (valueFrom != null && !loc.IsSame(valueFrom))
626 {
627 ctx.LoadValue(loc);
628 ctx.Cast(valueFrom.Type);
629 ctx.StoreValue(valueFrom);
630 }
631 }

634 private void WriteFieldHandler(
635 Compiler.CompilerContext ctx, Type expected, Compiler.Local loc,
636 Compiler.CodeLabel handler, Compiler.CodeLabel @continue, IProtoSerializer serializer)
637 {
638 ctx.MarkLabel(handler);
639 Type serType = serializer.ExpectedType;
640 if (serType == forType) {
641 EmitCreateIfNull(ctx, loc);
642 serializer.EmitRead(ctx, loc);
643 }
644 else {
645 //RuntimeTypeModel rtm = (RuntimeTypeModel)ctx.Model;
646 if (((IProtoTypeSerializer)serializer).CanCreateInstance())
647 {
648 Compiler.CodeLabel allDone = ctx.DefineLabel();
649
650 ctx.LoadValue(loc);
651 ctx.BranchIfFalse(allDone, false); // null is always ok
652
653 ctx.LoadValue(loc);
654 ctx.TryCast(serType);
655 ctx.BranchIfTrue(allDone, false); // not null, but of the correct type
656
657 // otherwise, need to convert it
658 ctx.LoadReaderWriter();
659 ctx.LoadValue(loc);
660 ((IProtoTypeSerializer)serializer).EmitCreateInstance(ctx);
661 ctx.EmitCall(ctx.MapType(typeof(ProtoReader)).GetMethod("Merge"));
662 ctx.Cast(expected);
663 ctx.StoreValue(loc); // Merge always returns a value
664
665 // nothing needs doing
666 ctx.MarkLabel(allDone);
667 }
668 ctx.LoadValue(loc);
669 ctx.Cast(serType);
670 serializer.EmitRead(ctx, null);
671
672 }
673
674 if (serializer.ReturnsValue)
675 { // update the variable
676 ctx.StoreValue(loc);
677 }
678 ctx.Branch(@continue, false); // "continue"
679 }

681 void IProtoTypeSerializer.EmitCreateInstance(Compiler.CompilerContext ctx)
682 {
683 // different ways of creating a new instance
684 bool callNoteObject = true;
685 if (factory != null)
686 {
687 EmitInvokeCallback(ctx, factory, false, constructType, forType);
688 }
689 else if (!useConstructor)
690 { // DataContractSerializer style
691 ctx.LoadValue(constructType);
692 ctx.EmitCall(ctx.MapType(typeof(BclHelpers)).GetMethod("GetUninitializedObject"));
693 ctx.Cast(forType);
694 }
695 else if (Helpers.IsClass(constructType) && hasConstructor)
696 { // XmlSerializer style
697 ctx.EmitCtor(constructType);
698 }
699 else
700 {
701 ctx.LoadValue(ExpectedType);
702 ctx.EmitCall(ctx.MapType(typeof(TypeModel)).GetMethod("ThrowCannotCreateInstance",
703 BindingFlags.Static | BindingFlags.Public));
704 ctx.LoadNullRef();
705 callNoteObject = false;
706 }
707 if (callNoteObject)
708 {
709 // track root object creation
710 ctx.CopyValue();
711 ctx.LoadReaderWriter();
712 ctx.EmitCall(ctx.MapType(typeof(ProtoReader)).GetMethod("NoteObject",
713 BindingFlags.Static | BindingFlags.Public));
714 }
715 if (baseCtorCallbacks != null)
716 {
717 for (int i = 0; i < baseCtorCallbacks.Length; i++)
718 {
719 EmitInvokeCallback(ctx, baseCtorCallbacks[i], true, null, forType);
720 }
721 }
