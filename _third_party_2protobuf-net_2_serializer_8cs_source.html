183 public static void Merge<T>(System.Xml.XmlReader reader, T instance) where T : System.Xml.Serialization.IXmlSerializable
184 {
185 if (reader == null) throw new ArgumentNullException("reader");
186 if (instance == null) throw new ArgumentNullException("instance");
187
188 const int LEN = 4096;
189 byte[] buffer = new byte[LEN];
190 int read;
191 using (MemoryStream ms = new MemoryStream())
192 {
193 int depth = reader.Depth;
194 while(reader.Read() && reader.Depth > depth)
195 {
196 if (reader.NodeType == System.Xml.XmlNodeType.Text)
197 {
198 while ((read = reader.ReadContentAsBase64(buffer, 0, LEN)) > 0)
199 {
200 ms.Write(buffer, 0, read);
201 }
202 if (reader.Depth <= depth) break;
203 }
204 }
205 ms.Position = 0;
206 Serializer.Merge(ms, instance);
207 }
