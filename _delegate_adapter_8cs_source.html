921 {

928 public unsafe StackObject* ILInvoke(ILIntepreter intp, StackObject* esp, IList<object> mStack)
929 {
930 var ebp = esp;
931 esp = ILInvokeSub(intp, esp, mStack);
932 return ClearStack(intp, esp, ebp, mStack);

934
935 unsafe StackObject* ILInvokeSub(ILIntepreter intp, StackObject* esp, IList<object> mStack)
936 {
937 var ebp = esp;
938 bool unhandled;
939 if (method.HasThis)
940 esp = ILIntepreter.PushObject(esp, mStack, instance);
