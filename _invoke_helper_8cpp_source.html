25int32_t InitNav(const char* buffer, int32_t n, dtNavMesh*& navMesh)
26{
27 int index = 0;
28 // Read header.
29 NavMeshSetHeader header;
30
31 int count = sizeof(NavMeshSetHeader);
32 if (index + count > n)
33 {
34 return -1;
35 }
36 memcpy(&header, buffer + index, count);
37 index += count;
38
39 if (header.magic != NAVMESHSET_MAGIC)
40 {
41 return -2;
42 }
43 if (header.version != NAVMESHSET_VERSION)
44 {
45 return -3;
46 }
47
48 dtNavMesh* mesh = dtAllocNavMesh();
49 if (!mesh)
50 {
51 return -4;
52 }
53 dtStatus status = mesh->init(&header.params);
54 if (dtStatusFailed(status))
55 {
56 return -5;
57 }
58
59 // Read tiles.
60 for (int i = 0; i < header.numTiles; ++i)
61 {
62 NavMeshTileHeader tileHeader;
63
64 count = sizeof(NavMeshTileHeader);
65 if (index + count > n)
66 {
67 return -6;
68 }
69 memcpy(&tileHeader, buffer + index, count);
70 index += count;
71
72 if (!tileHeader.tileRef || !tileHeader.dataSize)
73 break;
74
75 unsigned char* data = (unsigned char*)dtAlloc(tileHeader.dataSize, DT_ALLOC_PERM);
76 if (!data) break;
77 memset(data, 0, tileHeader.dataSize);
78
79 count = tileHeader.dataSize;
80 if (index + count > n)
81 {
82 return -7;
83 }
84 memcpy(data, buffer + index, count);
85 index += count;
86
87 mesh->addTile(data, tileHeader.dataSize, DT_TILE_FREE_DATA, tileHeader.tileRef, 0);
88 }
89 navMesh = mesh;
90 return 0;
91}
