128 _clipBounds = new Rect(0, 0, graphWidth, graphHeight);

130 var targetFps = TargetFps;
131
132 if (Application.isPlaying && TargetFpsUseApplication && Application.targetFrameRate > 0)
133 {
134 targetFps = Application.targetFrameRate;
135 }
136
137 var maxValue = 1f/targetFps;
138
139 // Holds the index of the nearest 'useful' FPS step
140 var fpsStep = -1;
141
142 var maxFrameTime = FloatingScale ? CalculateMaxFrameTime() : 1f/targetFps;
143
144 if (FloatingScale)
145 {
146 for (var i = 0; i < ScaleSteps.Length; i++)
147 {
148 var step = ScaleSteps[i];
149
150 if (maxFrameTime < step*1.1f)
151 {
152 maxValue = step;
153 fpsStep = i;
154 break;
155 }
156 }
157
158 // Fall back on the largest one
159 if (fpsStep < 0)
160 {
161 fpsStep = ScaleSteps.Length - 1;
162 maxValue = ScaleSteps[fpsStep];
163 }
164 }
165 else
166 {
167 // Search for the next scale step after the user-provided step
168 for (var i = 0; i < ScaleSteps.Length; i++)
169 {
170 var step = ScaleSteps[i];
171
172 if (maxFrameTime > step)
173 {
174 fpsStep = i;
175 }
176 }
177 }
178
179 var verticalScale = (graphHeight - (VerticalPadding*2))/maxValue;
180
181 // Number of data points that can fit into the graph space
182 var availableDataPoints = CalculateVisibleDataPointCount();
183
184 // Reallocate vertex array if insufficient length (or not yet created)
185 var sampleCount = GetFrameBufferCurrentSize();
186
187 for (var i = 0; i < sampleCount; i++)
188 {
189 // Break loop if all visible data points have been drawn
190 if (i >= availableDataPoints)
191 {
192 break;
193 }
194
195 // When using right-alignment, read from the end of the profiler buffer
196 var frame = GetFrame(sampleCount - i - 1);
197
198 // Left-hand x coord
199 var lx = graphWidth - DataPointWidth*i - DataPointWidth - graphWidth/2f;
200
201 DrawDataPoint(lx, verticalScale, frame);
202 }
203
204 if (DrawAxes)
205 {
206 if (!FloatingScale)
207 {
208 DrawAxis(maxValue, maxValue*verticalScale, GetAxisLabel(0));
209 }
210
211 var axisCount = 2;
212 var j = 0;
213
214 if (!FloatingScale)
215 {
216 j++;
217 }
218
219 for (var i = fpsStep; i >= 0; --i)
220 {
221 if (j >= axisCount)
222 {
223 break;
224 }
225
226 DrawAxis(ScaleSteps[i], ScaleSteps[i]*verticalScale, GetAxisLabel(j));

228 }

241 protected void DrawDataPoint(float xPosition, float verticalScale, ProfilerFrame frame)
242 {
243 // Right-hand x-coord
244 var rx = Mathf.Min(_clipBounds.width/2f, xPosition + DataPointWidth - DataPointMargin);
245
246 var currentLineHeight = 0f;
247
248 for (var j = 0; j < LineCount; j++)
249 {
250 var lineIndex = j;
251
252 var value = 0f;
253
254 if (j == 0)
255 {
256 value = (float) frame.UpdateTime;
257 }
258 else if (j == 1)
259 {
260 value = (float) frame.RenderTime;
261 }
262 else if (j == 2)
263 {
264 value = (float) frame.OtherTime;
265 }
266
267 value *= verticalScale;
268
269 if (value.ApproxZero() || value - DataPointVerticalMargin*2f < 0f)
270 {
271 continue;
272 }
273
274 // Lower y-coord
275 var ly = currentLineHeight + DataPointVerticalMargin - rectTransform.rect.height/2f;
276
277 if (VerticalAlignment == VerticalAlignments.Top)
278 {
279 ly = rectTransform.rect.height/2f - currentLineHeight - DataPointVerticalMargin;
280 }
281
282 // Upper y-coord
283 var uy = ly + value - DataPointVerticalMargin;
284
285 if (VerticalAlignment == VerticalAlignments.Top)
286 {
287 uy = ly - value + DataPointVerticalMargin;
288 }
289
290 var c = LineColours[lineIndex];
291
292 AddRect(new Vector3(Mathf.Max(-_clipBounds.width/2f, xPosition), ly),
293 new Vector3(Mathf.Max(-_clipBounds.width/2f, xPosition), uy), new Vector3(rx, uy),
294 new Vector3(rx, ly), c);
295
296 currentLineHeight += value;
297 }

414 protected float CalculateMaxFrameTime()
415 {
416 var frameCount = GetFrameBufferCurrentSize();
417 var c = Mathf.Min(CalculateVisibleDataPointCount(), frameCount);
418
419 var max = 0d;
420
421 for (var i = 0; i < c; i++)
422 {
423 var frameNumber = frameCount - i - 1;
424
425 var t = GetFrame(frameNumber);
426
427 if (t.FrameTime > max)
428 {
429 max = t.FrameTime;
430 }
431 }
432
433 return (float) max;
