13 /* The argument is an array of matcher strings each of which determine
14 whether a property with a certain path needs to be decorated with a
15 comment.
16
17 A path is a number of path elements concatenated by '/'. The last path
18 element is either:
19 the key if we're referring to a dict key
20 the value if we're referring to a value in an array
21 the value if we're referring to a value in a dict
22 All other path elements are either:
23 the key if the container is dict
24 '*' if the container is array
25
26 Path matcher has the same structure as a path, except that any of the
27 elements may be '*'. Matcher matches a path if both have the same number
28 of elements and for each pair matcher element is the same as path element
29 or is '*'.
30
31 a/b/c matches a/b/c but not a/b nor a/b/c/d
32 a/* /c matches a/d/c but not a/b nor a/b/c/d
33 * /* /* matches any path from three elements
34 */
35 protected PropertyCommentChecker(int level, List<List<string>> props)
36 {
37 m_Level = level;
38 m_All = false;
39 m_Props = props;
40 }
41
42 public PropertyCommentChecker()
43 {
44 m_Level = 0;
45 m_All = false;
46 m_Props = new List<List<string>>();
47 }
48
49 public PropertyCommentChecker(IEnumerable<string> props)
50 {
51 m_Level = 0;
52 m_All = false;
53 m_Props = new List<List<string>>();
54 foreach (var prop in props)
55 {
56 m_Props.Add(new List<string>(prop.Split('/')));
57 }
58 }
59
60 bool CheckContained(string prop)
61 {
62 if (m_All)
63 return true;
64 foreach (var list in m_Props)
65 {
66 if (list.Count == m_Level+1)
67 {
68 if (list[m_Level] == prop)
69 return true;
70 if (list[m_Level] == "*")
71 {
72 m_All = true; // short-circuit all at this level
73 return true;
74 }
75 }
76 }
77 return false;
78 }
79
80 public bool CheckStringValueInArray(string value) { return CheckContained(value); }
81 public bool CheckKeyInDict(string key) { return CheckContained(key); }
82
83 public bool CheckStringValueInDict(string key, string value)
84 {
85 foreach (var list in m_Props)
86 {
87 if (list.Count == m_Level + 2)
88 {
89 if ((list[m_Level] == "*" || list[m_Level] == key) &&
90 list[m_Level+1] == "*" || list[m_Level+1] == value)
91 return true;
92 }
93 }
94 return false;
95 }
96
97 public PropertyCommentChecker NextLevel(string prop)
98 {
99 var newList = new List<List<string>>();
100 foreach (var list in m_Props)
101 {
102 if (list.Count <= m_Level+1)
103 continue;
104 if (list[m_Level] == "*" || list[m_Level] == prop)
105 newList.Add(list);
106 }
107 return new PropertyCommentChecker(m_Level + 1, newList);
108 }
109 }

113 public static PBXElementDict ParseTreeAST(TreeAST ast, TokenList tokens, string text)
114 {
115 var el = new PBXElementDict();
116 foreach (var kv in ast.values)
117 {
118 PBXElementString key = ParseIdentifierAST(kv.key, tokens, text);
119 PBXElement value = ParseValueAST(kv.value, tokens, text);
120 el[key.value] = value;
121 }
122 return el;
123 }
124
125 public static PBXElementArray ParseArrayAST(ArrayAST ast, TokenList tokens, string text)
126 {
127 var el = new PBXElementArray();
128 foreach (var v in ast.values)
129 {
130 el.values.Add(ParseValueAST(v, tokens, text));
131 }
132 return el;
133 }
134
135 public static PBXElement ParseValueAST(ValueAST ast, TokenList tokens, string text)
136 {
137 if (ast is TreeAST)
138 return ParseTreeAST((TreeAST)ast, tokens, text);
139 if (ast is ArrayAST)
140 return ParseArrayAST((ArrayAST)ast, tokens, text);
141 if (ast is IdentifierAST)
142 return ParseIdentifierAST((IdentifierAST)ast, tokens, text);
143 return null;
144 }
145
146 public static PBXElementString ParseIdentifierAST(IdentifierAST ast, TokenList tokens, string text)
147 {
148 Token tok = tokens[ast.value];
149 string value;
150 switch (tok.type)
151 {
152 case TokenType.String:
153 value = text.Substring(tok.begin, tok.end - tok.begin);
154 return new PBXElementString(value);
155 case TokenType.QuotedString:
156 value = text.Substring(tok.begin, tok.end - tok.begin);
157 value = PBXStream.UnquoteString(value);
158 return new PBXElementString(value);
159 default:
160 throw new Exception("Internal parser error");
161 }
162 }

179 public static void WriteDictKeyValue(StringBuilder sb, string key, PBXElement value, int indent, bool compact,
180 PropertyCommentChecker checker, GUIDToCommentMap comments)
181 {
182 if (!compact)
183 {
184 sb.Append("\n");
185 sb.Append(GetIndent(indent));
186 }
187 WriteStringImpl(sb, key, checker.CheckKeyInDict(key), comments);
188 sb.Append(" = ");
189
190 if (value is PBXElementString)
191 WriteStringImpl(sb, value.AsString(), checker.CheckStringValueInDict(key, value.AsString()), comments);
192 else if (value is PBXElementDict)
193 WriteDict(sb, value.AsDict(), indent, compact, checker.NextLevel(key), comments);
194 else if (value is PBXElementArray)
195 WriteArray(sb, value.AsArray(), indent, compact, checker.NextLevel(key), comments);
196 sb.Append(";");
197 if (compact)
198 sb.Append(" ");
199 }
200
201 public static void WriteDict(StringBuilder sb, PBXElementDict el, int indent, bool compact,
202 PropertyCommentChecker checker, GUIDToCommentMap comments)
203 {
204 sb.Append("{");
205
206 if (el.Contains("isa"))
207 WriteDictKeyValue(sb, "isa", el["isa"], indent+1, compact, checker, comments);
208 var keys = new List<string>(el.values.Keys);
209 keys.Sort(StringComparer.Ordinal);
210 foreach (var key in keys)
211 {
212 if (key != "isa")
213 WriteDictKeyValue(sb, key, el[key], indent+1, compact, checker, comments);
214 }
215 if (!compact)
216 {
217 sb.Append("\n");
218 sb.Append(GetIndent(indent));
219 }
220 sb.Append("}");
221 }
222
223 public static void WriteArray(StringBuilder sb, PBXElementArray el, int indent, bool compact,
224 PropertyCommentChecker checker, GUIDToCommentMap comments)
225 {
226 sb.Append("(");
227 foreach (var value in el.values)
228 {
229 if (!compact)
230 {
231 sb.Append("\n");
232 sb.Append(GetIndent(indent+1));
233 }
234
235 if (value is PBXElementString)
236 WriteStringImpl(sb, value.AsString(), checker.CheckStringValueInArray(value.AsString()), comments);
237 else if (value is PBXElementDict)
238 WriteDict(sb, value.AsDict(), indent+1, compact, checker.NextLevel("*"), comments);
239 else if (value is PBXElementArray)
240 WriteArray(sb, value.AsArray(), indent+1, compact, checker.NextLevel("*"), comments);
241 sb.Append(",");
242 if (compact)
243 sb.Append(" ");
244 }
245
246 if (!compact)
247 {
248 sb.Append("\n");
249 sb.Append(GetIndent(indent));
250 }
251 sb.Append(")");
252 }
