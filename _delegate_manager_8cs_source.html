232 internal IDelegateAdapter FindDelegateAdapter(CLRType type, ILTypeInstance ins, ILMethod ilMethod)
233 {
234 IDelegateAdapter dele;
235 if (ins != null)
236 {
237 dele = (ins).GetDelegateAdapter(ilMethod);
238 if (dele == null)
239 {
240 var invokeMethod =
241 type.GetMethod("Invoke",
242 ilMethod.ParameterCount);
243 if (invokeMethod == null && ilMethod.IsExtend)
244 {
245 invokeMethod = type.GetMethod("Invoke", ilMethod.ParameterCount - 1);
246 }
247 dele = appdomain.DelegateManager.FindDelegateAdapter(
248 ins, ilMethod, invokeMethod);
249 }
250 }
251 else
252 {
253 if (ilMethod.DelegateAdapter == null)
254 {
255 var invokeMethod = type.GetMethod("Invoke", ilMethod.ParameterCount);
256 ilMethod.DelegateAdapter = appdomain.DelegateManager.FindDelegateAdapter(null, ilMethod, invokeMethod);
257 }
258 dele = ilMethod.DelegateAdapter;
259 }
260 return dele;
261 }

271 internal IDelegateAdapter FindDelegateAdapter(ILTypeInstance instance, ILMethod ilMethod, IMethod method)
272 {
273 IDelegateAdapter res;
274 var parameterCount = method.ParameterCount;
275 var returnTypeForCLR = method.ReturnType.TypeForCLR;
276 if (method.ReturnType == appdomain.VoidType)
277 {
278 if (parameterCount == 0)
279 {
280 res = zeroParamMethodAdapter.Instantiate(appdomain, instance, ilMethod);
281 if (instance != null)
282 instance.SetDelegateAdapter(ilMethod, res);
283 return res;
284 }
285 foreach (var i in methods)
286 {
287 var parameterTypes = i.ParameterTypes;
288 if (parameterTypes.Length == parameterCount)
289 {
290 bool match = true;
291 for (int j = 0; j < parameterCount; j++)
292 {
293 if (parameterTypes[j] != method.Parameters[j].TypeForCLR)
294 {
295 match = false;
296 break;
297 }
298 }
299 if (match)
300 {
301 res = i.Adapter.Instantiate(appdomain, instance, ilMethod);
302 if (instance != null)
303 instance.SetDelegateAdapter(ilMethod, res);
304 return res;
305 }
306 }
307 }
308 }
309 else
310 {
311
312 foreach (var i in functions)
313 {
314 var parameterTypes = i.ParameterTypes;
315 if (parameterTypes.Length == parameterCount + 1)
316 {
317 bool match = true;
318 for (int j = 0; j < parameterCount; j++)
319 {
320 if (parameterTypes[j] != method.Parameters[j].TypeForCLR)
321 {
322 match = false;
323 break;
324 }
325 }
326 if (match)
327 {
328 if (returnTypeForCLR == parameterTypes[parameterCount])
329 {
330 res = i.Adapter.Instantiate(appdomain, instance, ilMethod);
331 if (instance != null)
332 instance.SetDelegateAdapter(ilMethod, res);
333 return res;
334 }
335 }
336 }
337 }
338 }
339
340 res = dummyAdapter.Instantiate(appdomain, instance, ilMethod);
341 if (instance != null)
342 instance.SetDelegateAdapter(ilMethod, res);
343 return res;
344 }
