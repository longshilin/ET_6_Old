189 public static string AddAppExtension(this PBXProject proj, string mainTargetGuid,
190 string name, string bundleId, string infoPlistPath)
191 {
192 string ext = ".appex";
193 var newTargetGuid = proj.AddTarget(name, ext, "com.apple.product-type.app-extension");
194
195 foreach (var configName in proj.BuildConfigNames())
196 {
197 var configGuid = proj.BuildConfigByName(newTargetGuid, configName);
198 if (configName.Contains("Debug"))
199 SetDefaultAppExtensionDebugBuildFlags(proj, configGuid);
200 else
201 SetDefaultAppExtensionReleaseBuildFlags(proj, configGuid);
202 proj.SetBuildPropertyForConfig(configGuid, "INFOPLIST_FILE", infoPlistPath);
203 proj.SetBuildPropertyForConfig(configGuid, "PRODUCT_BUNDLE_IDENTIFIER", bundleId);
204 }
205
206 proj.AddSourcesBuildPhase(newTargetGuid);
207 proj.AddResourcesBuildPhase(newTargetGuid);
208 proj.AddFrameworksBuildPhase(newTargetGuid);
209 string copyFilesPhaseGuid = proj.AddCopyFilesBuildPhase(mainTargetGuid, "Embed App Extensions", "", "13");
210 proj.AddFileToBuildSection(mainTargetGuid, copyFilesPhaseGuid, proj.GetTargetProductFileRef(newTargetGuid));
211
212 proj.AddTargetDependency(mainTargetGuid, newTargetGuid);
213
214 return newTargetGuid;
215 }

227 public static string AddWatchApp(this PBXProject proj, string mainTargetGuid, string watchExtensionTargetGuid,
228 string name, string bundleId, string infoPlistPath)
229 {
230 var newTargetGuid = proj.AddTarget(name, ".app", "com.apple.product-type.application.watchapp2");
231
232 var isbcModuleName = proj.nativeTargets[watchExtensionTargetGuid].name.Replace(" ", "_");
233
234 foreach (var configName in proj.BuildConfigNames())
235 {
236 var configGuid = proj.BuildConfigByName(newTargetGuid, configName);
237 if (configName.Contains("Debug"))
238 SetDefaultWatchAppDebugBuildFlags(proj, configGuid);
239 else
240 SetDefaultWatchAppReleaseBuildFlags(proj, configGuid);
241 proj.SetBuildPropertyForConfig(configGuid, "PRODUCT_BUNDLE_IDENTIFIER", bundleId);
242 proj.SetBuildPropertyForConfig(configGuid, "INFOPLIST_FILE", infoPlistPath);
243 proj.SetBuildPropertyForConfig(configGuid, "IBSC_MODULE", isbcModuleName);
244 }
245
246 proj.AddResourcesBuildPhase(newTargetGuid);
247 string copyFilesGuid = proj.AddCopyFilesBuildPhase(newTargetGuid, "Embed App Extensions", "", "13");
248 proj.AddFileToBuildSection(newTargetGuid, copyFilesGuid, proj.GetTargetProductFileRef(watchExtensionTargetGuid));
249
250 string copyWatchFilesGuid = proj.AddCopyFilesBuildPhase(mainTargetGuid, "Embed Watch Content", "$(CONTENTS_FOLDER_PATH)/Watch", "16");
251 proj.AddFileToBuildSection(mainTargetGuid, copyWatchFilesGuid, proj.GetTargetProductFileRef(newTargetGuid));
252
253 proj.AddTargetDependency(newTargetGuid, watchExtensionTargetGuid);
254 proj.AddTargetDependency(mainTargetGuid, newTargetGuid);
255
256 return newTargetGuid;
257 }

308 public static void AddFileToEmbedFrameworks(this PBXProject proj, string targetGuid, string fileGuid)
309 {
310 PBXNativeTargetData target = proj.nativeTargets[targetGuid];
311
312 var phaseGuid = proj.AddCopyFilesBuildPhase(targetGuid, "Embed Frameworks", "", "10");
313 var phase = proj.copyFiles[phaseGuid];
314 var frameworkEmbedFileData = proj.FindFrameworkByFileGuid(phase, fileGuid);
315
316 if (frameworkEmbedFileData == null)
317 {
318 frameworkEmbedFileData = PBXBuildFileData.CreateFromFile(fileGuid, false, null);
319 proj.BuildFilesAdd(targetGuid, frameworkEmbedFileData);
320
321 phase.files.AddGUID(frameworkEmbedFileData.guid);
322 }
323
324 frameworkEmbedFileData.codeSignOnCopy = true;
325 frameworkEmbedFileData.removeHeadersOnCopy = true;
326 }
