16 public enum FieldAttributes : ushort {

18 CompilerControlled = 0x0000, // Member not referenceable

20 FamANDAssem = 0x0002, // Accessible by sub-types only in this assembly
21 Assembly = 0x0003, // Accessible by anyone in the Assembly
22 Family = 0x0004, // Accessible only by type and sub-types
23 FamORAssem = 0x0005, // Accessible by sub-types anywhere, plus anyone in the assembly
24 Public = 0x0006, // Accessible by anyone who has visibility to this scope field contract attributes
25
26 Static = 0x0010, // Defined on type, else per instance
27 InitOnly = 0x0020, // Field may only be initialized, not written after init
28 Literal = 0x0040, // Value is compile time constant
29 NotSerialized = 0x0080, // Field does not have to be serialized when type is remoted
30 SpecialName = 0x0200, // Field is special
31
32 // Interop Attributes
33 PInvokeImpl = 0x2000, // Implementation is forwarded through PInvoke
34

36 RTSpecialName = 0x0400, // CLI provides 'special' behavior, depending upon the name of the field
