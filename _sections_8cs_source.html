18 internal class KnownSectionBase<T> : SectionBase where T : PBXObjectData, new()
19 {
20 private Dictionary<string, T> m_Entries = new Dictionary<string, T>();
21
22 private string m_Name;
23
24 public KnownSectionBase(string sectionName)
25 {
26 m_Name = sectionName;
27 }
28
29 public IEnumerable<KeyValuePair<string, T>> GetEntries()
30 {
31 return m_Entries;
32 }
33
34 public IEnumerable<string> GetGuids()
35 {
36 return m_Entries.Keys;
37 }
38
39 public IEnumerable<T> GetObjects()
40 {
41 return m_Entries.Values;
42 }
43
44 public override void AddObject(string key, PBXElementDict value)
45 {
46 T obj = new T();
47 obj.guid = key;
48 obj.SetPropertiesWhenSerializing(value);
49 obj.UpdateVars();
50 m_Entries[obj.guid] = obj;
51 }
52
53 public override void WriteSection(StringBuilder sb, GUIDToCommentMap comments)
54 {
55 if (m_Entries.Count == 0)
56 return; // do not write empty sections
57
58 sb.AppendFormat("\n\n/* Begin {0} section */", m_Name);
59 var keys = new List<string>(m_Entries.Keys);
60 keys.Sort(StringComparer.Ordinal);
61 foreach (string key in keys)
62 {
63 T obj = m_Entries[key];
64 obj.UpdateProps();
65 sb.Append("\n\t\t");
66 comments.WriteStringBuilder(sb, obj.guid);
67 sb.Append(" = ");
68 Serializer.WriteDict(sb, obj.GetPropertiesWhenSerializing(), 2,
69 obj.shouldCompact, obj.checker, comments);
70 sb.Append(";");
71 }
72 sb.AppendFormat("\n/* End {0} section */", m_Name);
73 }
74
75 // returns null if not found
76 public T this[string guid]
77 {
78 get {
79 if (m_Entries.ContainsKey(guid))
80 return m_Entries[guid];
81 return null;
82 }
83 }
84
85 public bool HasEntry(string guid)
86 {
87 return m_Entries.ContainsKey(guid);
88 }
89
90 public void AddEntry(T obj)
91 {
92 m_Entries[obj.guid] = obj;
93 }
94
95 public void RemoveEntry(string guid)
96 {
97 if (m_Entries.ContainsKey(guid))
98 m_Entries.Remove(guid);
99 }
