67 public CircularBuffer(int capacity, T[] items)
68 {
69 if (capacity < 1)
70 {
71 throw new ArgumentException(
72 "Circular buffer cannot have negative or zero capacity.", "capacity");
73 }
74 if (items == null)
75 {
76 throw new ArgumentNullException("items");
77 }
78 if (items.Length > capacity)
79 {
80 throw new ArgumentException(
81 "Too many items to fit circular buffer", "items");
82 }
83
84 _buffer = new T[capacity];
85
86 Array.Copy(items, _buffer, items.Length);
87 _count = items.Length;
88
89 _start = 0;
90 _end = _count == capacity ? 0 : _count;
91 }

355 // The array is composed by at most two non-contiguous segments,

357
358 private ArraySegment<T> ArrayOne()
359 {
360 if (_start <= _end)
361 {
362 return new ArraySegment<T>(_buffer, _start, _end - _start);
363 }
364 return new ArraySegment<T>(_buffer, _start, _buffer.Length - _start);
365 }
366
367 private ArraySegment<T> ArrayTwo()
368 {
369 if (_start < _end)
370 {
371 return new ArraySegment<T>(_buffer, _end, 0);
372 }
373 return new ArraySegment<T>(_buffer, 0, _end);
374 }
