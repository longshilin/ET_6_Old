32 public static unsafe object ToObject(StackObject* esp, ILRuntime.Runtime.Enviorment.AppDomain appdomain, IList<object> mStack)
33 {
34 switch (esp->ObjectType)
35 {
36 case ObjectTypes.Integer:
37 return esp->Value;
38 case ObjectTypes.Long:
39 {
40 return *(long*)&esp->Value;
41 }
42 case ObjectTypes.Float:
43 {
44 return *(float*)&esp->Value;
45 }
46 case ObjectTypes.Double:
47 {
48 return *(double*)&esp->Value;
49 }
50 case ObjectTypes.Object:
51 return mStack[esp->Value];
52 case ObjectTypes.FieldReference:
53 {
54 ILTypeInstance instance = mStack[esp->Value] as ILTypeInstance;
55 if (instance != null)
56 {
57 return instance[esp->ValueLow];
58 }
59 else
60 {
61 var obj = mStack[esp->Value];
62 IType t = null;
63 if (obj is CrossBindingAdaptorType)
64 {
65 t = appdomain.GetType(((CrossBindingAdaptor)((CrossBindingAdaptorType)obj).ILInstance.Type.FirstCLRBaseType).BaseCLRType);
66 }
67 else
68 t = appdomain.GetType(obj.GetType());
69
70 return ((CLRType)t).GetFieldValue(esp->ValueLow, obj);
71 }
72 }
73 case ObjectTypes.ArrayReference:
74 {
75 Array instance = mStack[esp->Value] as Array;
76 return instance.GetValue(esp->ValueLow);
77 }
78 case ObjectTypes.StaticFieldReference:
79 {
80 var t = appdomain.GetType(esp->Value);
81 if (t is CLR.TypeSystem.ILType)
82 {
83 CLR.TypeSystem.ILType type = (CLR.TypeSystem.ILType)t;
84 return type.StaticInstance[esp->ValueLow];
85 }
86 else
87 {
88 CLR.TypeSystem.CLRType type = (CLR.TypeSystem.CLRType)t;
89 return type.GetFieldValue(esp->ValueLow, null);
90 }
91 }
92 case ObjectTypes.StackObjectReference:
93 {
94 return ToObject((ILIntepreter.ResolveReference(esp)), appdomain, mStack);
95 }
96 case ObjectTypes.ValueTypeObjectReference:
97 {
98 StackObject* dst = ILIntepreter.ResolveReference(esp);
99 IType type = appdomain.GetTypeByIndex(dst->Value);
100 if (type is ILType)
101 {
102 ILType iltype = (ILType)type;
103 var ins = iltype.Instantiate(false);
104 for (int i = 0; i < dst->ValueLow; i++)
105 {
106 var addr = ILIntepreter.Minus(dst, i + 1);
107 ins.AssignFromStack(i, addr, appdomain, mStack);
108 }
109 return ins;
110 }
111 else
112 {
113 return ((CLRType)type).ValueTypeBinder.ToObject(dst, mStack);
114 }
115 }
116 case ObjectTypes.Null:
117 return null;
118 default:
119 throw new NotImplementedException();
120 }
121 }

123 public unsafe static void Initialized(ref StackObject esp, int idx, IType fieldType, IList<object> mStack)
124 {
125 if (fieldType.IsPrimitive)

129 else
130 {

140 esp.Value = 0;
141 esp.ValueLow = 0;
142 mStack[idx] = null;
143 }
144 else
145 mStack[idx] = ((CLRType)fieldType).CreateDefaultInstance();
146 }
147 else
148 {
149 if (((ILType)fieldType).IsEnum)
150 {
151 esp.ObjectType = ObjectTypes.Integer;
152 esp.Value = 0;
153 esp.ValueLow = 0;
154 mStack[idx] = null;
155 }
156 else
157 mStack[idx] = ((ILType)fieldType).Instantiate();
158 }
159 }
160 else

164 }
165 }
