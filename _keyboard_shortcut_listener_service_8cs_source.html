20 protected override void Awake()

22 base.Awake();
23
24 CachedTransform.SetParent(Hierarchy.Get("SRDebugger"));
25
26 _shortcuts = new List<Settings.KeyboardShortcut>(Settings.Instance.KeyboardShortcuts);
27
28#if ENABLE_INPUT_SYSTEM
29
30 foreach (var s in _shortcuts)
31 {
32 // Cache the actual keycode so we don't have to use strings each time we want to use it.
33 string keyName = s.Key.ToString();
34 KeyControl keyControl = Keyboard.current[keyName] as KeyControl;
35
36 if (keyControl == null)
37 {
38 Debug.LogErrorFormat(
39 "[SRDebugger] Input System: Unable to find shortcut key: {0}. Shortcut ({1}) will not be functional.",
40 keyName, s.Action);
41 s.Cached_KeyCode = Key.None;
42 }
43
44 // Find the index for this key control
45 for (var index = 0; index < Keyboard.current.allKeys.Count; index++)
46 {
47 if (Keyboard.current.allKeys[index] == keyControl)
48 {
49 s.Cached_KeyCode = (Key) (index + 1);
50 break;
51 }
52 }
53 }
54
55#endif

178#if ENABLE_INPUT_SYSTEM
179 private void UpdateInputSystem()
180 {
181 var keyboard = Keyboard.current;
182
183 if (Settings.Instance.KeyboardEscapeClose && keyboard.escapeKey.isPressed && Service.Panel.IsVisible)
184 {
185 SRDebug.Instance.HideDebugPanel();
186 }
187
188 var ctrl = keyboard.leftCtrlKey.isPressed || keyboard.rightCtrlKey.isPressed;
189 var alt = keyboard.leftAltKey.isPressed || keyboard.rightAltKey.isPressed;
190 var shift = keyboard.leftShiftKey.isPressed || keyboard.rightShiftKey.isPressed;
191
192 for (var i = 0; i < _shortcuts.Count; i++)
193 {
194 var s = _shortcuts[i];
195
196 if (s.Control && !ctrl)
197 {
198 continue;
199 }
200
201 if (s.Shift && !shift)
202 {
203 continue;
204 }
205
206 if (s.Alt && !alt)
207 {
208 continue;
209 }
210
211 if (!s.Cached_KeyCode.HasValue)
212 {
213 continue; // We can't use this shortcut since we didn't find the keycode.
214 }
215
216 if (keyboard[s.Cached_KeyCode.Value].wasPressedThisFrame)
217 {
218 ExecuteShortcut(s);
219 break;
220 }
221 }
222 }
223#endif

225#if ENABLE_LEGACY_INPUT_MANAGER || (!ENABLE_INPUT_SYSTEM && !UNITY_2019_3_OR_NEWER)

234 var alt = Input.GetKey(KeyCode.LeftAlt) || Input.GetKey(KeyCode.RightAlt);
235 var shift = Input.GetKey(KeyCode.LeftShift) || Input.GetKey(KeyCode.RightShift);
236
237 for (var i = 0; i < _shortcuts.Count; i++)
238 {

240

255
