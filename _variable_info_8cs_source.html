143 public static VariableInfo FromObject(object obj, bool retriveType = false)
144 {
145 VariableInfo info = new VariableInfo();
146 info.Name = "";
147 if (obj != null)
148 {
149 info.Value = obj.ToString();
150 if(obj is int)
151 {
152 info.ValueType = ValueTypes.Integer;
153 }
154 else if(obj is bool)
155 {
156 info.ValueType = ValueTypes.Boolean;
157 }
158 else if(obj is string)
159 {
160 info.ValueType = ValueTypes.String;
161 }
162 if (retriveType)
163 {
164 if (obj is Runtime.Intepreter.ILTypeInstance)
165 {
166 info.TypeName = ((Intepreter.ILTypeInstance)obj).Type.FullName;
167 }
168 else if (obj is Enviorment.CrossBindingAdaptorType)
169 {
170 info.TypeName = ((Enviorment.CrossBindingAdaptorType)obj).ILInstance.Type.FullName;
171 }
172 else
173 {
174 info.TypeName = obj.GetType().FullName;
175 }
176 }
177 info.Expandable = !obj.GetType().IsPrimitive && !(obj is string);
178 }
179 else
180 {
181 info.Value = "null";
182 info.ValueType = ValueTypes.Null;
183 }
184
185 return info;
186 }
