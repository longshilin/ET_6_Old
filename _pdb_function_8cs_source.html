236 internal PdbFunction(/*string module, */ManProcSym proc, BitAccess bits) {
237 this.token = proc.token;
238 //this.module = module;
239 //this.name = proc.name;
240 //this.flags = proc.flags;
241 this.segment = proc.seg;
242 this.address = proc.off;
243 this.length = proc.len;
244
245 if (proc.seg != 1) {
246 throw new PdbDebugException("Segment is {0}, not 1.", proc.seg);
247 }
248 if (proc.parent != 0 || proc.next != 0) {
249 throw new PdbDebugException("Warning parent={0}, next={1}",
250 proc.parent, proc.next);
251 }
252 //if (proc.dbgStart != 0 || proc.dbgEnd != 0) {
253 // throw new PdbDebugException("Warning DBG start={0}, end={1}",
254 // proc.dbgStart, proc.dbgEnd);
255 //}
256
257 int constantCount;
258 int scopeCount;
259 int slotCount;
260 int usedNamespacesCount;
261 CountScopesAndSlots(bits, proc.end, out constantCount, out scopeCount, out slotCount, out usedNamespacesCount);
262 int scope = constantCount > 0 || slotCount > 0 || usedNamespacesCount > 0 ? 1 : 0;
263 int slot = 0;
264 int constant = 0;
265 int usedNs = 0;
266 scopes = new PdbScope[scopeCount+scope];
267 slots = new PdbSlot[slotCount];
268 constants = new PdbConstant[constantCount];
269 usedNamespaces = new string[usedNamespacesCount];
270
271 if (scope > 0)
272 scopes[0] = new PdbScope(this.address, proc.len, slots, constants, usedNamespaces);
273
274 while (bits.Position < proc.end) {
275 ushort siz;
276 ushort rec;
277
278 bits.ReadUInt16(out siz);
279 int star = bits.Position;
280 int stop = bits.Position + siz;
281 bits.Position = star;
282 bits.ReadUInt16(out rec);
283
284 switch ((SYM)rec) {
285 case SYM.S_OEM: { // 0x0404
286 OemSymbol oem;
287
288 bits.ReadGuid(out oem.idOem);
289 bits.ReadUInt32(out oem.typind);
290 // internal byte[] rgl; // user data, force 4-byte alignment
291
292 if (oem.idOem == msilMetaData) {
293 string name = bits.ReadString();
294 if (name == "MD2") {
295 ReadMD2CustomMetadata(bits);
296 } else if (name == "asyncMethodInfo") {
297 this.synchronizationInformation = new PdbSynchronizationInformation(bits);
298 }
299 bits.Position = stop;
300 break;
301 } else {
302 throw new PdbDebugException("OEM section: guid={0} ti={1}",
303 oem.idOem, oem.typind);
304 // bits.Position = stop;
305 }
306 }
307
308 case SYM.S_BLOCK32: {
309 BlockSym32 block = new BlockSym32();
310
311 bits.ReadUInt32(out block.parent);
312 bits.ReadUInt32(out block.end);
313 bits.ReadUInt32(out block.len);
314 bits.ReadUInt32(out block.off);
315 bits.ReadUInt16(out block.seg);
316 bits.SkipCString(out block.name);
317 bits.Position = stop;
318
319 scopes[scope++] = new PdbScope(this.address, block, bits, out slotToken);
320 bits.Position = (int)block.end;
321 break;
322 }
323
324 case SYM.S_MANSLOT:
325 slots[slot++] = new PdbSlot(bits);
326 bits.Position = stop;
327 break;
328
329 case SYM.S_MANCONSTANT:
330 constants[constant++] = new PdbConstant(bits);
331 bits.Position = stop;
332 break;
333
334 case SYM.S_UNAMESPACE:
335 bits.ReadCString(out usedNamespaces[usedNs++]);
336 bits.Position = stop;
337 break;
338
339 case SYM.S_END:
340 bits.Position = stop;
341 break;
342
343 default: {
344 //throw new PdbDebugException("Unknown SYM: {0}", (SYM)rec);
345 bits.Position = stop;
346 break;
347 }
348 }
349 }
350
351 if (bits.Position != proc.end) {
352 throw new PdbDebugException("Not at S_END");
353 }
354
355 ushort esiz;
356 ushort erec;
357 bits.ReadUInt16(out esiz);
358 bits.ReadUInt16(out erec);
359
360 if (erec != (ushort)SYM.S_END) {
361 throw new PdbDebugException("Missing S_END");
362 }
363 }

378 private void ReadCustomMetadata(BitAccess bits)
379 {
380 int savedPosition = bits.Position;
381 byte version;
382 bits.ReadUInt8(out version);
383 byte kind;
384 bits.ReadUInt8(out kind);
385 bits.Position += 2; // 2-bytes padding
386 uint numberOfBytesInItem;
387 bits.ReadUInt32(out numberOfBytesInItem);
388 if (version == 4)
389 {
390 switch (kind)
391 {
392 case 0: this.ReadUsingInfo(bits); break;
393 case 1: this.ReadForwardInfo(bits); break;
394 case 2: break; // this.ReadForwardedToModuleInfo(bits); break;
395 case 3: this.ReadIteratorLocals(bits); break;
396 case 4: this.ReadForwardIterator(bits); break;
397 case 5: break; // dynamic locals - see http://index/#Microsoft.VisualStudio.LanguageServices/Shared/CustomDebugInfoReader.cs,a3031f7681d76e93
398 case 6: break; // EnC data
399 case 7: break; // EnC data for lambdas and closures
400 // ignore any other unknown record types that may be added in future, instead of throwing an exception
401 // see more details here: https://github.com/tmat/roslyn/blob/portable-pdb/docs/specs/PortablePdb-Metadata.md
402 default: break; // throw new PdbDebugException("Unknown custom metadata item kind: {0}", kind);
403 }
404 }
405 bits.Position = savedPosition + (int)numberOfBytesInItem;
406 }
