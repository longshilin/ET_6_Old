17
18 public bool Equals (ByteBuffer x, ByteBuffer y)
19 {
20 if (x.length != y.length)
21 return false;
22
23 var x_buffer = x.buffer;
24 var y_buffer = y.buffer;
25
26 for (int i = 0; i < x.length; i++)
27 if (x_buffer [i] != y_buffer [i])
28 return false;
29
30 return true;
31 }
32
33 public int GetHashCode (ByteBuffer buffer)
34 {
35 // See http://en.wikipedia.org/wiki/Fowler%E2%80%93Noll%E2%80%93Vo_hash_function
36 const int fnv_offset_bias = unchecked((int)2166136261);
37 const int fnv_prime = 16777619;
38
39 var hash_code = fnv_offset_bias;
40 var bytes = buffer.buffer;
41
42 for (int i = 0; i < buffer.length; i++)
43 hash_code = unchecked ((hash_code ^ bytes [i]) * fnv_prime);
44
45 return hash_code;
46 }
47 }
