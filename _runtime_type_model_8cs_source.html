147 public override string GetSchema(Type type, ProtoSyntax syntax)
148 {
149 BasicList requiredTypes = new BasicList();
150 MetaType primaryType = null;
151 bool isInbuiltType = false;
152 if (type == null)
153 { // generate for the entire model
154 foreach(MetaType meta in types)
155 {
156 MetaType tmp = meta.GetSurrogateOrBaseOrSelf(false);
157 if (!requiredTypes.Contains(tmp))
158 { // ^^^ note that the type might have been added as a descendent
159 requiredTypes.Add(tmp);
160 CascadeDependents(requiredTypes, tmp);
161 }
162 }
163 }
164 else
165 {
166 Type tmp = Helpers.GetUnderlyingType(type);
167 if (tmp != null) type = tmp;
168
169 WireType defaultWireType;
170 isInbuiltType = (ValueMember.TryGetCoreSerializer(this, DataFormat.Default, type, out defaultWireType, false, false, false, false) != null);
171 if (!isInbuiltType)
172 {
173 //Agenerate just relative to the supplied type
174 int index = FindOrAddAuto(type, false, false, false);
175 if (index < 0) throw new ArgumentException("The type specified is not a contract-type", "type");
176
177 // get the required types
178 primaryType = ((MetaType)types[index]).GetSurrogateOrBaseOrSelf(false);
179 requiredTypes.Add(primaryType);
180 CascadeDependents(requiredTypes, primaryType);
181 }
182 }
183
184 // use the provided type's namespace for the "package"
185 StringBuilder headerBuilder = new StringBuilder();
186 string package = null;
187
188 if (!isInbuiltType)
189 {
190 IEnumerable typesForNamespace = primaryType == null ? types : requiredTypes;
191 foreach (MetaType meta in typesForNamespace)
192 {
193 if (meta.IsList) continue;
194 string tmp = meta.Type.Namespace;
195 if (!Helpers.IsNullOrEmpty(tmp))
196 {
197 if (tmp.StartsWith("System.")) continue;
198 if (package == null)
199 { // haven't seen any suggestions yet
200 package = tmp;
201 }
202 else if (package == tmp)
203 { // that's fine; a repeat of the one we already saw
204 }
205 else
206 { // something else; have confliucting suggestions; abort
207 package = null;
208 break;
209 }
210 }
211 }
212 }
213 switch(syntax)
214 {
215 case ProtoSyntax.Proto2:
216 headerBuilder.AppendLine(@"syntax = ""proto2"";");
217 break;
218 case ProtoSyntax.Proto3:
219 headerBuilder.AppendLine(@"syntax = ""proto3"";");
220 break;
221 default:
222 throw new ArgumentOutOfRangeException(nameof(syntax));
223 }
224
225 if (!Helpers.IsNullOrEmpty(package))
226 {
227 headerBuilder.Append("package ").Append(package).Append(';');
228 Helpers.AppendLine(headerBuilder);
229 }
230
231 var imports = CommonImports.None;
232 StringBuilder bodyBuilder = new StringBuilder();
233 // sort them by schema-name
234 MetaType[] metaTypesArr = new MetaType[requiredTypes.Count];
235 requiredTypes.CopyTo(metaTypesArr, 0);
236 Array.Sort(metaTypesArr, MetaType.Comparer.Default);
237
238 // write the messages
239 if (isInbuiltType)
240 {
241 Helpers.AppendLine(bodyBuilder).Append("message ").Append(type.Name).Append(" {");
242 MetaType.NewLine(bodyBuilder, 1).Append(syntax == ProtoSyntax.Proto2 ? "optional " : "").Append(GetSchemaTypeName(type, DataFormat.Default, false, false, ref imports))
243 .Append(" value = 1;");
244 Helpers.AppendLine(bodyBuilder).Append('}');
245 }
246 else
247 {
248 for (int i = 0; i < metaTypesArr.Length; i++)
249 {
250 MetaType tmp = metaTypesArr[i];
251 if (tmp.IsList && tmp != primaryType) continue;
252 tmp.WriteSchema(bodyBuilder, 0, ref imports, syntax);
253 }
254 }
255 if ((imports & CommonImports.Bcl) != 0)
256 {
257 headerBuilder.Append("import \"protobuf-net/bcl.proto\"; // schema for protobuf-net's handling of core .NET types");
258 Helpers.AppendLine(headerBuilder);
259 }
260 if ((imports & CommonImports.Protogen) != 0)
261 {
262 headerBuilder.Append("import \"protobuf-net/protogen.proto\"; // custom protobuf-net options");
263 Helpers.AppendLine(headerBuilder);
264 }
265 if ((imports & CommonImports.Timestamp) != 0)
266 {
267 headerBuilder.Append("import \"google/protobuf/timestamp.proto\";");
268 Helpers.AppendLine(headerBuilder);
269 }
270 if ((imports & CommonImports.Duration) != 0)
271 {
272 headerBuilder.Append("import \"google/protobuf/duration.proto\";");
273 Helpers.AppendLine(headerBuilder);
274 }
275 return Helpers.AppendLine(headerBuilder.Append(bodyBuilder)).ToString();
276 }

286 private void CascadeDependents(BasicList list, MetaType metaType)
287 {
288 MetaType tmp;
289 if (metaType.IsList)
290 {
291 Type itemType = TypeModel.GetListItemType(this, metaType.Type);
292 WireType defaultWireType;
293 IProtoSerializer coreSerializer = ValueMember.TryGetCoreSerializer(this, DataFormat.Default, itemType, out defaultWireType, false, false, false, false);
294 if (coreSerializer == null)
295 {
296 int index = FindOrAddAuto(itemType, false, false, false);
297 if (index >= 0)
298 {
299 tmp = ((MetaType)types[index]).GetSurrogateOrBaseOrSelf(false);
300 if (!list.Contains(tmp))
301 { // could perhaps also implement as a queue, but this should work OK for sane models
302 list.Add(tmp);
303 CascadeDependents(list, tmp);
304 }
305 }
306 }
307 }
308 else
309 {
310 if (metaType.IsAutoTuple)
311 {
312 MemberInfo[] mapping;
313 if(MetaType.ResolveTupleConstructor(metaType.Type, out mapping) != null)
314 {
315 for (int i = 0; i < mapping.Length; i++)
316 {
317 Type type = null;
318 if (mapping[i] is PropertyInfo) type = ((PropertyInfo)mapping[i]).PropertyType;
319 else if (mapping[i] is FieldInfo) type = ((FieldInfo)mapping[i]).FieldType;
320
321 WireType defaultWireType;
322 IProtoSerializer coreSerializer = ValueMember.TryGetCoreSerializer(this, DataFormat.Default, type, out defaultWireType, false, false, false, false);
323 if (coreSerializer == null)
324 {
325 int index = FindOrAddAuto(type, false, false, false);
326 if (index >= 0)
327 {
328 tmp = ((MetaType)types[index]).GetSurrogateOrBaseOrSelf(false);
329 if (!list.Contains(tmp))
330 { // could perhaps also implement as a queue, but this should work OK for sane models
331 list.Add(tmp);
332 CascadeDependents(list, tmp);
333 }
334 }
335 }
336 }
337 }
338 }
339 else
340 {
341 foreach (ValueMember member in metaType.Fields)
342 {
343 Type type = member.ItemType;
344 if(member.IsMap)
345 {
346 Type dictionaryType, keyType;
347 member.ResolveMapTypes(out dictionaryType, out keyType, out type); // don't need key-type
348 }
349 if (type == null) type = member.MemberType;
350 WireType defaultWireType;
351 IProtoSerializer coreSerializer = ValueMember.TryGetCoreSerializer(this, DataFormat.Default, type, out defaultWireType, false, false, false, false);
352 if (coreSerializer == null)
353 {
354 // is an interesting type
355 int index = FindOrAddAuto(type, false, false, false);
356 if (index >= 0)
357 {
358 tmp = ((MetaType)types[index]).GetSurrogateOrBaseOrSelf(false);
359 if (!list.Contains(tmp))
360 { // could perhaps also implement as a queue, but this should work OK for sane models
361 list.Add(tmp);
362 CascadeDependents(list, tmp);
363 }
364 }
365 }
366 }
367 }
368 if (metaType.HasSubtypes)
369 {
370 foreach (SubType subType in metaType.GetSubtypes())
371 {
372 tmp = subType.DerivedType.GetSurrogateOrSelf(); // note: exclude base-types!
373 if (!list.Contains(tmp))
374 {
375 list.Add(tmp);
376 CascadeDependents(list, tmp);
377 }
378 }
379 }
380 tmp = metaType.BaseType;
381 if (tmp != null) tmp = tmp.GetSurrogateOrSelf(); // note: already walking base-types; exclude base
382 if (tmp != null && !list.Contains(tmp))
383 {
384 list.Add(tmp);
385 CascadeDependents(list, tmp);
386 }
387 }

533 internal int FindOrAddAuto(Type type, bool demand, bool addWithContractOnly, bool addEvenIfAutoDisabled)
534 {
535 int key = types.IndexOf(MetaTypeFinder, type);
536 MetaType metaType;
537
538 // the fast happy path: meta-types we've already seen
539 if (key >= 0)
540 {
541 metaType = (MetaType)types[key];
542 if (metaType.Pending)
543 {
544 WaitOnLock(metaType);
545 }
546 return key;
547 }
548
549 // the fast fail path: types that will never have a meta-type
550 bool shouldAdd = AutoAddMissingTypes || addEvenIfAutoDisabled;
551
552 if (!Helpers.IsEnum(type) && TryGetBasicTypeSerializer(type) != null)
553 {
554 if (shouldAdd && !addWithContractOnly) throw MetaType.InbuiltType(type);
555 return -1; // this will never be a meta-type
556 }
557
558 // otherwise: we don't yet know
559
560 // check for proxy types
561 Type underlyingType = ResolveProxies(type);
562 if (underlyingType != null)
563 {
564 key = types.IndexOf(MetaTypeFinder, underlyingType);
565 type = underlyingType; // if new added, make it reflect the underlying type
566 }
567
568 if (key < 0)
569 {
570 int opaqueToken = 0;
571 try
572 {
573 TakeLock(ref opaqueToken);
574 // try to recognise a few familiar patterns...
575 if ((metaType = RecogniseCommonTypes(type)) == null)
576 { // otherwise, check if it is a contract
577 MetaType.AttributeFamily family = MetaType.GetContractFamily(this, type, null);
578 if (family == MetaType.AttributeFamily.AutoTuple)
579 {
580 shouldAdd = addEvenIfAutoDisabled = true; // always add basic tuples, such as KeyValuePair
581 }
582
583 if (!shouldAdd || (
584 !Helpers.IsEnum(type) && addWithContractOnly && family == MetaType.AttributeFamily.None)
585 )
586 {
587 if (demand) ThrowUnexpectedType(type);
588 return key;
589 }
590 metaType = Create(type);
591 }
592 metaType.Pending = true;
593 bool weAdded = false;
594
595 // double-checked
596 int winner = types.IndexOf(MetaTypeFinder, type);
597 if (winner < 0)
598 {
599 ThrowIfFrozen();
600 key = types.Add(metaType);
601 weAdded = true;
602 }
603 else
604 {
605 key = winner;
606 }
607 if (weAdded)
608 {
609 metaType.ApplyDefaultBehaviour();
610 metaType.Pending = false;
611 }
612 }
613 finally
614 {
615 ReleaseLock(opaqueToken);
616 }
617 }
618 return key;
619 }

669 public MetaType Add(Type type, bool applyDefaultBehaviour)
670 {
671 if (type == null) throw new ArgumentNullException("type");
672 MetaType newType = FindWithoutAdd(type);
673 if (newType != null) return newType; // return existing
674 int opaqueToken = 0;
675
676#if WINRT || COREFX
677 System.Reflection.TypeInfo typeInfo = System.Reflection.IntrospectionExtensions.GetTypeInfo(type);
678 if (typeInfo.IsInterface && MetaType.ienumerable.IsAssignableFrom(typeInfo)
679#else
680 if (type.IsInterface && MapType(MetaType.ienumerable).IsAssignableFrom(type)
681#endif
682 && GetListItemType(this, type) == null)
683 {
684 throw new ArgumentException("IEnumerable[<T>] data cannot be used as a meta-type unless an Add method can be resolved");
685 }
686 try
687 {
688 newType = RecogniseCommonTypes(type);
689 if(newType != null)
690 {
691 if(!applyDefaultBehaviour) {
692 throw new ArgumentException(
693 "Default behaviour must be observed for certain types with special handling; " + type.FullName,
694 "applyDefaultBehaviour");
695 }
696 // we should assume that type is fully configured, though; no need to re-run:
697 applyDefaultBehaviour = false;
698 }
699 if(newType == null) newType = Create(type);
700 newType.Pending = true;
701 TakeLock(ref opaqueToken);
702 // double checked
703 if (FindWithoutAdd(type) != null) throw new ArgumentException("Duplicate type", "type");
704 ThrowIfFrozen();
705 types.Add(newType);
706 if (applyDefaultBehaviour) { newType.ApplyDefaultBehaviour(); }
707 newType.Pending = false;
708 }
709 finally
710 {
711 ReleaseLock(opaqueToken);
712 }
713
714 return newType;

895 internal sealed class SerializerPair : IComparable

897 int IComparable.CompareTo(object obj)
898 {
899 if (obj == null) throw new ArgumentException("obj");
900 SerializerPair other = (SerializerPair)obj;
901
902 // we want to bunch all the items with the same base-type together, but we need the items with a
903 // different base **first**.
904 if (this.BaseKey == this.MetaKey)
905 {
906 if (other.BaseKey == other.MetaKey)
907 { // neither is a subclass
908 return this.MetaKey.CompareTo(other.MetaKey);
909 }
910 else
911 { // "other" (only) is involved in inheritance; "other" should be first
912 return 1;
913 }
914 }
915 else
916 {
917 if (other.BaseKey == other.MetaKey)
918 { // "this" (only) is involved in inheritance; "this" should be first
919 return -1;
920 }
921 else
922 { // both are involved in inheritance
923 int result = this.BaseKey.CompareTo(other.BaseKey);
924 if (result == 0) result = this.MetaKey.CompareTo(other.MetaKey);
925 return result;
926 }
927 }

1357 private Compiler.CompilerContext WriteSerializeDeserialize(string assemblyName, TypeBuilder type, SerializerPair[] methodPairs, Compiler.CompilerContext.ILVersion ilVersion, ref ILGenerator il)
1358 {
1359 il = Override(type, "Serialize");
1360 Compiler.CompilerContext ctx = new Compiler.CompilerContext(il, false, true, methodPairs, this, ilVersion, assemblyName, MapType(typeof(object)), "Serialize " + type.Name);
1361 // arg0 = this, arg1 = key, arg2=obj, arg3=dest
1362 Compiler.CodeLabel[] jumpTable = new Compiler.CodeLabel[types.Count];
1363 for (int i = 0; i < jumpTable.Length; i++)
1364 {
1365 jumpTable[i] = ctx.DefineLabel();
1366 }
1367 il.Emit(OpCodes.Ldarg_1);
1368 ctx.Switch(jumpTable);
1369 ctx.Return();
1370 for (int i = 0; i < jumpTable.Length; i++)
1371 {
1372 SerializerPair pair = methodPairs[i];
1373 ctx.MarkLabel(jumpTable[i]);
1374 il.Emit(OpCodes.Ldarg_2);
1375 ctx.CastFromObject(pair.Type.Type);
1376 il.Emit(OpCodes.Ldarg_3);
1377 il.EmitCall(OpCodes.Call, pair.Serialize, null);
1378 ctx.Return();
1379 }
1380
1381 il = Override(type, "Deserialize");
1382 ctx = new Compiler.CompilerContext(il, false, false, methodPairs, this, ilVersion, assemblyName, MapType(typeof(object)), "Deserialize " + type.Name);
1383 // arg0 = this, arg1 = key, arg2=obj, arg3=source
1384 for (int i = 0; i < jumpTable.Length; i++)
1385 {
1386 jumpTable[i] = ctx.DefineLabel();
1387 }
1388 il.Emit(OpCodes.Ldarg_1);
1389 ctx.Switch(jumpTable);
1390 ctx.LoadNullRef();
1391 ctx.Return();
1392 for (int i = 0; i < jumpTable.Length; i++)
1393 {
1394 SerializerPair pair = methodPairs[i];
1395 ctx.MarkLabel(jumpTable[i]);
1396 Type keyType = pair.Type.Type;
1397 if (Helpers.IsValueType(keyType))
1398 {
1399 il.Emit(OpCodes.Ldarg_2);
1400 il.Emit(OpCodes.Ldarg_3);
1401 il.EmitCall(OpCodes.Call, EmitBoxedSerializer(type, i, keyType, methodPairs, this, ilVersion, assemblyName), null);
1402 ctx.Return();
1403 }
1404 else
1405 {
1406 il.Emit(OpCodes.Ldarg_2);
1407 ctx.CastFromObject(keyType);
1408 il.Emit(OpCodes.Ldarg_3);
1409 il.EmitCall(OpCodes.Call, pair.Deserialize, null);
1410 ctx.Return();
1411 }
1412 }
1413 return ctx;

1416 private const int KnownTypes_Array = 1, KnownTypes_Dictionary = 2, KnownTypes_Hashtable = 3, KnownTypes_ArrayCutoff = 20;
1417 private void WriteGetKeyImpl(TypeBuilder type, bool hasInheritance, SerializerPair[] methodPairs, Compiler.CompilerContext.ILVersion ilVersion, string assemblyName, out ILGenerator il, out int knownTypesCategory, out FieldBuilder knownTypes, out Type knownTypesLookupType)
1418 {
1419
1420 il = Override(type, "GetKeyImpl");
1421 Compiler.CompilerContext ctx = new Compiler.CompilerContext(il, false, false, methodPairs, this, ilVersion, assemblyName, MapType(typeof(System.Type), true), "GetKeyImpl");
1422
1423
1424 if (types.Count <= KnownTypes_ArrayCutoff)
1425 {
1426 knownTypesCategory = KnownTypes_Array;
1427 knownTypesLookupType = MapType(typeof(System.Type[]), true);
1428 }
1429 else
1430 {
1431#if NO_GENERICS
1432 knownTypesLookupType = null;
1433#else
1434 knownTypesLookupType = MapType(typeof(System.Collections.Generic.Dictionary<System.Type, int>), false);
1435#endif
1436
1437#if !COREFX
1438 if (knownTypesLookupType == null)
1439 {
1440 knownTypesLookupType = MapType(typeof(Hashtable), true);
1441 knownTypesCategory = KnownTypes_Hashtable;
1442 }
1443 else
1444#endif
1445 {
1446 knownTypesCategory = KnownTypes_Dictionary;
1447 }
1448 }
1449 knownTypes = type.DefineField("knownTypes", knownTypesLookupType, FieldAttributes.Private | FieldAttributes.InitOnly | FieldAttributes.Static);
1450
1451 switch (knownTypesCategory)
1452 {
1453 case KnownTypes_Array:
1454 {
1455 il.Emit(OpCodes.Ldsfld, knownTypes);
1456 il.Emit(OpCodes.Ldarg_1);
1457 // note that Array.IndexOf is not supported under CF
1458 il.EmitCall(OpCodes.Callvirt, MapType(typeof(IList)).GetMethod(
1459 "IndexOf", new Type[] { MapType(typeof(object)) }), null);
1460 if (hasInheritance)
1461 {
1462 il.DeclareLocal(MapType(typeof(int))); // loc-0
1463 il.Emit(OpCodes.Dup);
1464 il.Emit(OpCodes.Stloc_0);
1465
1466 BasicList getKeyLabels = new BasicList();
1467 int lastKey = -1;
1468 for (int i = 0; i < methodPairs.Length; i++)
1469 {
1470 if (methodPairs[i].MetaKey == methodPairs[i].BaseKey) break;
1471 if (lastKey == methodPairs[i].BaseKey)
1472 { // add the last label again
1473 getKeyLabels.Add(getKeyLabels[getKeyLabels.Count - 1]);
1474 }
1475 else
1476 { // add a new unique label
1477 getKeyLabels.Add(ctx.DefineLabel());
1478 lastKey = methodPairs[i].BaseKey;
1479 }
1480 }
1481 Compiler.CodeLabel[] subtypeLabels = new Compiler.CodeLabel[getKeyLabels.Count];
1482 getKeyLabels.CopyTo(subtypeLabels, 0);
1483
1484 ctx.Switch(subtypeLabels);
1485 il.Emit(OpCodes.Ldloc_0); // not a sub-type; use the original value
1486 il.Emit(OpCodes.Ret);
1487
1488 lastKey = -1;
1489 // now output the different branches per sub-type (not derived type)
1490 for (int i = subtypeLabels.Length - 1; i >= 0; i--)
1491 {
1492 if (lastKey != methodPairs[i].BaseKey)
1493 {
1494 lastKey = methodPairs[i].BaseKey;
1495 // find the actual base-index for this base-key (i.e. the index of
1496 // the base-type)
1497 int keyIndex = -1;
1498 for (int j = subtypeLabels.Length; j < methodPairs.Length; j++)
1499 {
1500 if (methodPairs[j].BaseKey == lastKey && methodPairs[j].MetaKey == lastKey)
1501 {
1502 keyIndex = j;
1503 break;
1504 }
1505 }
1506 ctx.MarkLabel(subtypeLabels[i]);
1507 Compiler.CompilerContext.LoadValue(il, keyIndex);
1508 il.Emit(OpCodes.Ret);
1509 }
1510 }
1511 }
1512 else
1513 {
1514 il.Emit(OpCodes.Ret);
1515 }
1516 }
1517 break;
1518 case KnownTypes_Dictionary:
1519 {
1520 LocalBuilder result = il.DeclareLocal(MapType(typeof(int)));
1521 Label otherwise = il.DefineLabel();
1522 il.Emit(OpCodes.Ldsfld, knownTypes);
1523 il.Emit(OpCodes.Ldarg_1);
1524 il.Emit(OpCodes.Ldloca_S, result);
1525 il.EmitCall(OpCodes.Callvirt, knownTypesLookupType.GetMethod("TryGetValue", BindingFlags.Instance | BindingFlags.Public), null);
1526 il.Emit(OpCodes.Brfalse_S, otherwise);
1527 il.Emit(OpCodes.Ldloc_S, result);
1528 il.Emit(OpCodes.Ret);
1529 il.MarkLabel(otherwise);
1530 il.Emit(OpCodes.Ldc_I4_M1);
1531 il.Emit(OpCodes.Ret);
1532 }
1533 break;
1534 case KnownTypes_Hashtable:
1535 {
1536 Label otherwise = il.DefineLabel();
1537 il.Emit(OpCodes.Ldsfld, knownTypes);
1538 il.Emit(OpCodes.Ldarg_1);
1539 il.EmitCall(OpCodes.Callvirt, knownTypesLookupType.GetProperty("Item").GetGetMethod(), null);
1540 il.Emit(OpCodes.Dup);
1541 il.Emit(OpCodes.Brfalse_S, otherwise);
1542#if FX11
1543 il.Emit(OpCodes.Unbox, MapType(typeof(int)));
1544 il.Emit(OpCodes.Ldobj, MapType(typeof(int)));
1545#else
1546 if (ilVersion == Compiler.CompilerContext.ILVersion.Net1)
1547 {
1548 il.Emit(OpCodes.Unbox, MapType(typeof(int)));
1549 il.Emit(OpCodes.Ldobj, MapType(typeof(int)));
1550 }
1551 else
1552 {
1553 il.Emit(OpCodes.Unbox_Any, MapType(typeof(int)));
1554 }
1555#endif
1556 il.Emit(OpCodes.Ret);
1557 il.MarkLabel(otherwise);
1558 il.Emit(OpCodes.Pop);
1559 il.Emit(OpCodes.Ldc_I4_M1);
1560 il.Emit(OpCodes.Ret);
1561 }
1562 break;
1563 default:
1564 throw new InvalidOperationException();
1565 }

1617#endif
