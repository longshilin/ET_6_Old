142 private static async ETTask<IActorResponse> CallInner(this ActorLocationSenderComponent self, ActorLocationSender actorLocationSender, int rpcId, MemoryStream memoryStream)
143 {
144 int failTimes = 0;
145 long instanceId = actorLocationSender.InstanceId;
146 actorLocationSender.LastSendOrRecvTime = TimeHelper.ServerNow();
147
148 while (true)
149 {
150 if (actorLocationSender.ActorId == 0)
151 {
152 actorLocationSender.ActorId = await LocationProxyComponent.Instance.Get(actorLocationSender.Id);
153 if (actorLocationSender.InstanceId != instanceId)
154 {
155 throw new RpcException(ErrorCore.ERR_ActorLocationSenderTimeout2, $"{memoryStream.ToActorMessage()}");
156 }
157 }
158
159 if (actorLocationSender.ActorId == 0)
160 {
161 IActorRequest iActorRequest = (IActorRequest)memoryStream.ToActorMessage();
162 return ActorHelper.CreateResponse(iActorRequest, ErrorCore.ERR_NotFoundActor);
163 }
164 IActorResponse response = await ActorMessageSenderComponent.Instance.Call(actorLocationSender.ActorId, rpcId, memoryStream, false);
165 if (actorLocationSender.InstanceId != instanceId)
166 {
167 throw new RpcException(ErrorCore.ERR_ActorLocationSenderTimeout3, $"{memoryStream.ToActorMessage()}");
168 }
169
170 switch (response.Error)
171 {
172 case ErrorCore.ERR_NotFoundActor:
173 {
174 // 如果没找到Actor,重试
175 ++failTimes;
176 if (failTimes > 20)
177 {
178 Log.Debug($"actor send message fail, actorid: {actorLocationSender.Id}");
179 actorLocationSender.Error = ErrorCore.ERR_NotFoundActor;
180 // 这里不能删除actor，要让后面等待发送的消息也返回ERR_NotFoundActor，直到超时删除
181 return response;
182 }
183
184 // 等待0.5s再发送
185 await TimerComponent.Instance.WaitAsync(500);
186 if (actorLocationSender.InstanceId != instanceId)
187 {
188 throw new RpcException(ErrorCore.ERR_ActorLocationSenderTimeout4, $"{memoryStream.ToActorMessage()}");
189 }
190
191 actorLocationSender.ActorId = 0;
192 continue;
193 }
194 case ErrorCore.ERR_ActorTimeout:
195 {
196 throw new RpcException(response.Error, $"{memoryStream.ToActorMessage()}");
197 }
198 }
199
200 if (ErrorCore.IsRpcNeedThrowException(response.Error))
201 {
202 throw new RpcException(response.Error, $"Message: {response.Message} Request: {memoryStream.ToActorMessage()}");
203 }
204
205 return response;
206 }
207 }
